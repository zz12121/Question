```sql
CREATE TABLE `idempotent_record` (

  `biz_id` VARCHAR(64) NOT NULL COMMENT '业务单号',
  `idempotent_key` VARCHAR(128) NOT NULL COMMENT '幂等唯一KEY',

  `update_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `expire_at` DATETIME NOT NULL COMMENT '过期时间，TTL字段',
  UNIQUE KEY `uk_biz_key` (`biz_type`, `idempotent_key`),
  KEY `idx_expire_at` (`expire_at`),
  KEY `idx_biz_id` (`biz_id`)

-- ========== 方案二：业务专属幂等表 ==========
CREATE TABLE `payment_idempotent` (
  `id` BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  `trade_no` VARCHAR(64) NOT NULL COMMENT '支付流水号',
  `order_no` VARCHAR(64) NOT NULL COMMENT '订单号',
  `amount` DECIMAL(10,2) NOT NULL COMMENT '金额',
  `status` TINYINT NOT NULL DEFAULT 0 COMMENT '状态',
  `create_time` DATETIME NOT NULL,
  `update_time` DATETIME NOT NULL,
  `expire_time` DATETIME NOT NULL COMMENT '过期时间',
  UNIQUE KEY `uk_trade_no` (`trade_no`),
  KEY `idx_order_no` (`order_no`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ========== MySQL TTL 设置（MySQL 8.0+） ==========
-- MySQL 本身不支持自动TTL清理，但可以用事件调度器
```

### 2.2 幂等 Key 生成策略

```java
// ========== 幂等 Key 生成工具 ==========
public class IdempotentKeyGenerator {
    
    /**
     * 支付回调幂等Key
     * 格式: PAY_{渠道}_{支付流水号}
     * 例如: PAY_ALIPAY_20230101123456
     */
    public static String forPayment(String channel, String tradeNo) {
        return String.format("PAY_%s_%s", channel.toUpperCase(), tradeNo);
    }
    
    /**
     * 消息消费幂等Key
     * 格式: MSG_{topic}_{consumerGroup}_{messageId}
     */
    public static String forMessage(String topic, String consumerGroup, String messageId) {
        return String.format("MSG_%s_%s_%s", topic, consumerGroup, messageId);
    }
    
    /**
     * 接口提交幂等Key
     * 格式: SUBMIT_{业务}_{用户ID}_{时间戳分钟}
     * 支持N分钟内幂等
     */
    public static String forSubmit(String bizType, Long userId, int minuteSlot) {
        long slot = System.currentTimeMillis() / (minuteSlot * 60 * 1000);
        return String.format("SUBMIT_%s_%d_%d", bizType, userId, slot);
    }
    
    /**
     * 表单重复提交幂等Key（防抖）
     * 格式: NONCE_{用户ID}_{nonce随机串}
     * nonce由前端生成，一次性使用
     */
    public static String forNonce(Long userId, String nonce) {
        return String.format("NONCE_%d_%s", userId, nonce);
    }
}
```

### 2.3 通用幂等 Service

```java
// ========== 通用幂等处理器 ==========
@Service
public class IdempotentService {
    
    @Autowired
    private IdempotentRecordMapper idempotentMapper;
    
    @Autowired
    private StringRedisTemplate redisTemplate;
    
    // Redis 锁前缀（防止并发时同时写入DB）
    private static final String LOCK_PREFIX = "idempotent:lock:";
    // 默认TTL: 24小时
    private static final Duration DEFAULT_TTL = Duration.ofHours(24);
    
    /**
     * 通用幂等校验（Redis + DB 双写）
     * @param bizType 业务类型
     * @param idempotentKey 幂等key
     * @param expireTime 过期时间
     * @return true=首次执行（需要处理业务），false=重复请求（直接返回）
     */
    public boolean tryAcquire(String bizType, String idempotentKey, Duration expireTime) {
        // Step1: Redis SETNX 快速判断
        String redisKey = buildRedisKey(bizType, idempotentKey);
        Boolean success = redisTemplate.opsForValue()
            .setIfAbsent(redisKey, "PROCESSING", expireTime);
        
        if (Boolean.TRUE.equals(success)) {
            // 首次请求，获取分布式锁，准备写DB
            return true;
        }
        
        // Step2: Redis 命中，检查状态
        String status = redisTemplate.opsForValue().get(redisKey);
        if ("SUCCESS".equals(status)) {
            // 已成功处理，直接返回
            return false;
        }
        
        // Step3: 可能是处理中的脏数据（锁过期但业务还在执行）
        // 降级到DB查询
        IdempotentRecord record = idempotentMapper.selectByBizKey(bizType, idempotentKey);
        return record == null || record.getStatus() != 1;
    }
    
    /**
     * 标记处理成功
     */
    public void markSuccess(String bizType, String idempotentKey, String resultMsg) {
        String redisKey = buildRedisKey(bizType, idempotentKey);
        
        // Redis 标记成功
        redisTemplate.opsForValue().set(redisKey, "SUCCESS", Duration.ofHours(1));
        
        // DB 更新状态
        idempotentMapper.updateStatus(bizType, idempotentKey, 1, resultMsg);
    }
    
    /**
     * 标记处理失败（可重试）
     */
    public void markFailed(String bizType, String idempotentKey, String errorMsg) {
        idempotentMapper.updateStatus(bizType, idempotentKey, 2, errorMsg);
        // 失败时删除Redis，让下次可以重试
        redisTemplate.delete(buildRedisKey(bizType, idempotentKey));
    }
    
    /**
     * 释放锁（用于长事务场景，提前释放）
     */
    public void releaseLock(String bizType, String idempotentKey) {
        redisTemplate.delete(buildRedisKey(bizType, idempotentKey));
    }
    
    private String buildRedisKey(String bizType, String idempotentKey) {
        return String.format("idempotent:%s:%s", bizType, idempotentKey);
    }
}
```

### 2.4 三大场景实战

```java
// ========== 场景一：支付回调 ==========
@Service
public class PaymentCallbackService {
    
    @Autowired
    private IdempotentService idempotentService;
    
    @Autowired
    private OrderService orderService;
    
    public void handleCallback(PaymentCallbackDTO callback) {
        // 1. 生成幂等Key
        String idempotentKey = IdempotentKeyGenerator.forPayment(
            callback.getChannel(), 
            callback.getTradeNo()
        );
        
        // 2. 幂等校验（支付回调建议TTL=7天）
        boolean shouldProcess = idempotentService.tryAcquire(
            "PAY", 
            idempotentKey, 
            Duration.ofDays(7)
        );
        
        if (!shouldProcess) {
            log.info("支付回调重复，忽略: tradeNo={}", callback.getTradeNo());
            return;
        }
        
        try {
            // 3. 业务处理
            orderService.confirmOrder(callback.getOrderNo(), callback.getAmount());
            
            // 4. 标记成功
            idempotentService.markSuccess("PAY", idempotentKey, "success");
            
        } catch (Exception e) {
            // 5. 失败时不标记，允许重试
            idempotentService.markFailed("PAY", idempotentKey, e.getMessage());
            throw e;
        }
    }
}

// ========== 场景二：消息消费 ==========
@Service
public class OrderMessageConsumer {
    
    @Autowired
    private IdempotentService idempotentService;
    
    @KafkaListener(topics = "order-topic", groupId = "order-service")
    public void consumeOrderMessage(OrderMessage message, Acknowledgment ack) {
        // 1. 生成幂等Key
        String idempotentKey = IdempotentKeyGenerator.forMessage(
            "order-topic",
            "order-service",
            message.getMessageId()
        );
        
        // 2. 幂等校验（消息消费建议TTL=1天）
        boolean shouldProcess = idempotentService.tryAcquire(
            "MSG",
            idempotentKey,
            Duration.ofHours(24)
        );
        
        if (!shouldProcess) {
            log.info("消息重复消费，忽略: messageId={}", message.getMessageId());
            ack.acknowledge();  // 仍然ACK，避免无限重试
            return;
        }
        
        try {
            // 3. 业务处理
            processOrder(message);
            
            // 4. 标记成功
            idempotentService.markSuccess("MSG", idempotentKey, null);
            
            // 5. 手动ACK
            ack.acknowledge();
            
        } catch (Exception e) {
            idempotentService.markFailed("MSG", idempotentKey, e.getMessage());
            // 不ACK，触发重试
            throw e;
        }
    }
}

// ========== 场景三：接口幂等提交 ==========
@Service
public class OrderSubmitService {
    
    @Autowired
    private IdempotentService idempotentService;
    
    // 前端生成一次性 nonce
    public Result submitOrder(OrderSubmitDTO dto, @RequestHeader("X-Idempotent-Nonce") String nonce) {
        // 1. 生成幂等Key（用户+nonce，一次性）
        String idempotentKey = IdempotentKeyGenerator.forNonce(dto.getUserId(), nonce);
        
        // 2. 幂等校验（接口提交建议TTL=10分钟）
        boolean shouldProcess = idempotentService.tryAcquire(
            "ORDER",
            idempotentKey,
            Duration.ofMinutes(10)
        );
        
        if (!shouldProcess) {
            return Result.error("请求已提交，请勿重复操作");
        }
        
        try {
            // 3. 业务处理
            Order order = createOrder(dto);
            
            // 4. 标记成功
            idempotentService.markSuccess("ORDER", idempotentKey, order.getOrderId());
            
            return Result.success(order);
            
        } catch (Exception e) {
            idempotentService.markFailed("ORDER", idempotentKey, e.getMessage());
            throw e;
        }
    }
}
```

## 🔥 扩展追问 1：Redis 方案 vs 数据库方案怎么选？

> **A:** 两种方案各有优劣，要根据场景选：

| 特性 | Redis 方案 | 数据库幂等表 |
|------|-----------|------------|
| 性能 | ✅ 极高，纳秒级 | ❌ 毫秒级，有IO |
| 持久化 | ❌ Redis 挂了可能丢数据 | ✅ 肯定不会丢 |
| 分布式 | ✅ 天然支持 | ✅ 支持 |
| 超时控制 | ✅ TTL 天然支持 | ❌ 需额外清理 |
| 适用场景 | 高并发、允许极端情况丢一条 | 金融/支付等强幂等 |

**最佳实践：Redis + DB 双写**
```java

```

## 🔥 扩展追问 2：状态机 + 乐观锁怎么配合？

> **A:** 幂等的核心是"只执行一次"，但很多业务需要"只成功一次、失败可重试"。这就需要状态机配合：

```sql
-- 订单表加状态字段
ALTER TABLE orders ADD COLUMN status INT DEFAULT 0;
-- 0=待支付 1=已支付 2=支付失败

-- 乐观锁更新，只有状态正确才能更新
UPDATE orders 
SET status = 1, 
    pay_time = NOW(),
    version = version + 1
WHERE order_no = #{orderNo} 
  AND status = 0  -- 关键：只有待支付才能变已支付
  AND version = #{version};
-- 影响行数=1才算成功，=0说明已被其他请求更新
```

**代码层面：**
```java
public boolean confirmOrder(String orderNo, BigDecimal amount) {
    int rows = orderMapper.confirmIfPending(orderNo, amount);
    if (rows == 0) {
        // 影响0行：要么不存在，要么状态不对
        Order order = orderMapper.selectByOrderNo(orderNo);
        if (order.getStatus() == 1) {
            log.info("订单已支付，幂等忽略: {}", orderNo);
            return true;
        }
        return false;
    }
    return true;
}
```

## 🔥 扩展追问 3：分布式锁 vs 幂等表，哪个更好？

> **A:** 这两个解决的是不同问题：

| 场景 | 分布式锁 | 幂等表 |
|------|---------|--------|
| 并发控制 | ✅ 互斥，同一时刻只能一个请求执行 | ❌ 不能防并发，只能防重复 |
| 失败重试 | ❌ 锁超时自动释放，可能重复执行 | ✅ 有记录，可以识别重复 |
| 超时控制 | ✅ 自动过期 | ✅ TTL 字段可控 |
| 适用 | 扣库存、余额扣减等原子操作 | 回调、消息、提交等可重试场景 |

**结论：可以结合用**
```java
// 分布式锁 + 幂等表
public void process() {
    String lockKey = "order:" + orderId;
    try {
        if (!redisLock.tryLock(lockKey, 30, TimeUnit.SECONDS)) {
            throw new BizException("系统繁忙");
        }
        
        // 加锁成功后，幂等校验
        if (!idempotentService.tryAcquire("ORDER", orderId, Duration.ofHours(24))) {
            return; // 重复请求
        }
        
        // 业务处理
        doBusiness();
        
    } finally {
        redisLock.unlock(lockKey);
    }
}
```

## 🔥 扩展追问 4：如何防止幂等 Key 被恶意刷？

```java
// 1. 对 nonce 做签名验证
public class IdempotentTokenManager {
    
    private static final String SECRET = "your-secret-key";
    
    public String generateToken(Long userId, String action) {
        String nonce = UUID.randomUUID().toString();
        String sign = MD5(userId + action + nonce + SECRET);
        return nonce + ":" + sign;
    }
    
    public boolean validateToken(Long userId, String action, String token) {
        String[] parts = token.split(":");
        String nonce = parts[0];
        String sign = parts[1];
        return MD5(userId + action + nonce + SECRET).equals(sign);
    }
}
```

## 🔥 扩展追问 5：TTL 到底设置多久合理？

> **A:** TTL 设置原则是"业务允许的最长时间"

| 场景 | 推荐TTL | 理由 |
|------|--------|------|
| 支付回调 | 7天 | 银行可能延迟回调，月底对账 |
| 消息消费 | 1-3天 | 消息积压时可能延迟消费 |
| 接口幂等 | 5-30分钟 | 防重复提交，时间窗口够用就行 |
| 异步任务 | 1-2小时 | 任务失败重试的时间窗口 |
| 查询操作 | 10分钟 | 一般不需要幂等 |

**Redis TTL + DB expire_at 字段配合：**
```java
public void saveIdempotentRecord(String bizType, String key, Duration ttl) {
    // Redis TTL
    redisTemplate.expire(redisKey, ttl);
    
    // DB expire_at 字段（兜底清理）
    IdempotentRecord record = new IdempotentRecord();
    record.setExpireAt(LocalDateTime.now().plus(ttl));
    idempotentMapper.insert(record);
}
```

## 一句话总结

> **幂等设计核心是"唯一 Key + 状态校验 + 防重表"，通过 Redis SETNX 快速判断 + DB 持久化兜底，配合状态机/乐观锁做业务校验，TTL 根据业务容忍度设置，金融类7天、消息1天、接口5分钟。**

**面试加分话术：**

> "之前公司被薅过一次羊毛，有个用户用脚本并发提交了 200 多次相同订单，库存扣了 200 份。后来查出来就是接口没做幂等。我加了两层防护：第一层是前端生成 UUID 随机串做 nonce，后端 Redis SETNX 5 分钟过期快速拦截；第二层是订单表用乐观锁 `WHERE status=0`，只有待支付状态才能转已支付，这样即使绕过第一层也会被数据库兜住。另外注意分布式锁和幂等表不要混用，锁解决的是并发问题，幂等解决的是重复问题，场景不同。"
