```
│  ┌─────────────────────────────────────────────────────────────┐   │

│  │  消费失败？消息直接丢弃！                                   │   │

│  └─────────────────────────────────────────────────────────────┘   │
│                                                                     │

│  │                      │                      │               │   │
│  │                      │                      ▼               │   │
│  │                      │              [Pending Entry List]   │   │
│  │                      │                      │               │   │

│                                                                     │

│  ┌─────────────────────────────────────────────────────────────┐   │
│  │                                                             │   │
│  │  Stream ──▶ [msg1, msg2, msg3, msg4, msg5]               │   │

│  │  Group A ──▶ consumer-2: [msg3]                            │   │
│  │                                                             │   │

│  │                                                             │   │
│  │  ✅ 每条消息只会分给组内一个消费者                           │   │

│  │                                                             │   │
│  └─────────────────────────────────────────────────────────────┘   │
│                                                                     │
└─────────────────────────────────────────────────────────────────────┘
```

```bash

PUBLISH alert:error '{"level":"error","msg":"服务挂了"}'
PUBLISH alert:warn '{"level":"warn","msg":"内存告警"}'

# 订阅（客户端）

# 订阅多个频道
PSUBSCRIBE alert:*

# 1. 创建 Stream 和消费者组
XGROUP CREATE notification:stream workers $ MKSTREAM

# 2. 发送消息

# 3. 读取新消息
XREADGROUP GROUP workers consumer-1 STREAMS notification:stream ">"

# 4. 读取 Pending 消息（之前获取但未确认）
XREADGROUP GROUP workers consumer-1 STREAMS notification:stream "0"

# 5. 确认消息
XACK notification:stream workers 1527854169173-0

# 6. 查看 Pending 列表
XPENDING notification:stream workers

# 7. 认领超时消息
XCLAIM notification:stream workers consumer-2 60000 1527854169173-0

# 8. 裁剪 Stream（保留最近 1000 条）
```

```java
// Pub/Sub 发布者
public class PubSubPublisher {
    
    private JedisPool jedisPool;
    
    public void publishAlert(String level, String message) {
        try (Jedis jedis = jedisPool.getResource()) {
            String channel = "alert:" + level;
            
            String payload = JSON.toJSONString(new AlertEvent(
                level, message, System.currentTimeMillis()
            ));
            
            // 返回订阅者数量（可以为 0）
            long subscribers = jedis.publish(channel, payload);
            System.out.println("推送给 " + subscribers + " 个订阅者");
        }
    }
}

// Pub/Sub 订阅者
public class PubSubSubscriber {
    
    public void subscribe() {
        JedisPool jedisPool = new JedisPool();
        Jedis jedis = jedisPool.getResource();
        
        JedisPubSub pubSub = new JedisPubSub() {
            @Override
            public void onMessage(String channel, String message) {
                try {
                    System.out.println("收到 [" + channel + "]: " + message);
                    // 处理消息...
                } catch (Exception e) {
                    e.printStackTrace();
                }
            }
            
            @Override
            public void onSubscribe(String channel, int subscribedChannels) {
                System.out.println("订阅频道: " + channel);
            }
        };
        
        // 订阅多个频道
        jedis.subscribe(pubSub, "alert:error", "alert:warn");
        
        // 或者使用通配符
        // jedis.psubscribe(pubSub, "alert:*");
    }
}

// Stream 可靠消息实现
public class StreamMessageService {
    
    private JedisPool jedisPool;
    private static final String STREAM_KEY = "notification:stream";
    private static final String GROUP_NAME = "notification-workers";
    
    /**
     * 初始化：创建 Stream 和消费者组
     */
    public void init() {
        try (Jedis jedis = jedisPool.getResource()) {
            try {
                // $ 表示只接收新消息，0 表示接收所有历史消息
                jedis.xgroupCreate(STREAM_KEY, GROUP_NAME, "0", true);
                System.out.println("消费者组创建成功");
            } catch (JedisDataException e) {
                if (e.getMessage().contains("BUSYGROUP")) {
                    System.out.println("消费者组已存在");
                } else {
                    throw e;
                }
            }
        }
    }
    
    /**
     * 发送消息
     */
    public String sendNotification(String userId, String title, String content) {
        try (Jedis jedis = jedisPool.getResource()) {
            Map<String, String> fields = new HashMap<>();
            fields.put("userId", userId);
            fields.put("title", title);
            fields.put("content", content);
            fields.put("timestamp", String.valueOf(System.currentTimeMillis()));
            fields.put("status", "pending");
            
            String messageId = jedis.xadd(
                STREAM_KEY, 
                StreamEntryID.NEW_ENTRY,
                fields
            );
            
            System.out.println("消息已发送，ID: " + messageId);
            return messageId;
        }
    }
    
    /**
     * 消费消息
     */
    public List<Map<String, String>> consume(int batchSize, String consumerName) {
        try (Jedis jedis = jedisPool.getResource()) {
            Map<String, List<Map.Entry<List<String>, List<Map<String, String>>>>> 
                result = jedis.xreadGroup(
                    ConsumerGroup.from(GROUP_NAME, consumerName),
                    StreamReadGroupOptions.Count(batchSize)
                        .block(Duration.ofSeconds(1))
                        .count(batchSize),
                    new StreamOffset[]{StreamOffset.create(STREAM_KEY, ReadOffset.from(">"))}
                );
            
            List<Map<String, String>> messages = new ArrayList<>();
            
            if (result != null && !result.isEmpty()) {
                for (Map.Entry<List<String>, List<Map<String, String>>> entry : 
                     result.get(STREAM_KEY)) {
                    messages.addAll(entry.getValue());
                }
            }
            
            return messages;
        }
    }
    
    /**
     * 确认消息
     */
    public void ackMessage(String messageId) {
        try (Jedis jedis = jedisPool.getResource()) {
            jedis.xack(STREAM_KEY, GROUP_NAME, messageId);
            System.out.println("消息 " + messageId + " 已确认");
        }
    }
    
    /**
     * 处理超时未确认消息（自动重试）
     */
    public void retryPendingMessages(int idleTimeoutMs) {
        try (Jedis jedis = jedisPool.getResource()) {
            // 查看 Pending 消息
            List<Map<String, Object>> pending = jedis.xpendingAll(STREAM_KEY, GROUP_NAME);
            
            for (Map<String, Object> info : pending) {
                String messageId = (String) info.get("message_id");
                String consumer = (String) info.get("consumer");
                long idleTime = ((Number) info.get("time_since_delivered")).longValue();
                
                // 超过阈值，认领并重新处理
                if (idleTime > idleTimeoutMs) {
                    System.out.println("消息 " + messageId + " 空闲 " + idleTime + 
                                      "ms，由 " + consumer + " 持有，认领重试");
                    
                    jedis.xclaim(
                        STREAM_KEY,
                        GROUP_NAME,
                        "retry-consumer",  // 新消费者
                        idleTimeoutMs,
                        new StreamEntryID(messageId)
                    );
                }
            }
        }
    }
    
    /**
     * 完整消息处理流程
     */
    public void processMessages(String consumerName) {
        while (true) {
            try {
                // 读取消息
                List<Map<String, String>> messages = consume(10, consumerName);
                
                for (Map<String, String> msg : messages) {
                    String messageId = msg.get("messageId");
                    String userId = msg.get("userId");
                    String title = msg.get("title");
                    
                    try {
                        // 执行业务逻辑
                        sendEmail(userId, title);
                        
                        // 成功：ACK
                        ackMessage(messageId);
                        
                    } catch (Exception e) {
                        // 失败：不 ACK，会被重新分发
                        e.printStackTrace();
                    }
                }
                
                Thread.sleep(100);
                
            } catch (Exception e) {
                e.printStackTrace();
            }
        }
    }
    
    private void sendEmail(String userId, String title) {
        System.out.println("发送邮件给 " + userId + ": " + title);
    }
}
```

## 🔥 扩展追问 1：Pub/Sub 有哪些局限性？

**Pub/Sub 的三大致命问题**

```
┌─────────────────────────────────────────────────────────────────┐
│                    Pub/Sub 核心局限                               │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  问题一：无消息持久化                                            │
│  ┌─────────────────────────────────────────────────────────┐   │
│  │ • 消息即发即弃，不存储                                     │   │
│  │ • 订阅者不在线 → 消息丢失                               │   │
│  │ • 网络抖动 → 消息丢失                                    │   │
│  └─────────────────────────────────────────────────────────┘   │
│                                                                 │
│  问题二：无消费确认                                               │
│  ┌─────────────────────────────────────────────────────────┐   │
│  │ • 消息发出后不关心是否被消费                             │   │
│  │ • 消费失败 → 消息丢失                                   │   │
│  │ • 没有重试机制                                           │   │
│  └─────────────────────────────────────────────────────────┘   │
│                                                                 │
│  问题三：无消息分片                                               │
│  ┌─────────────────────────────────────────────────────────┐   │
│  │ • 所有订阅者都能收到全部消息                              │   │
│  │ • 无法将消息分给不同消费者处理                            │   │
│  │ • 无法实现负载均衡                                       │   │
│  └─────────────────────────────────────────────────────────┘   │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

| 特性 | Pub/Sub | RabbitMQ | Kafka | Stream |
|------|---------|----------|-------|--------|
| 消息持久化 | ❌ 无 | ✅ 有 | ✅ 有 | ✅ 可选 |
| 消息确认 | ❌ 无 | ✅ ACK | ✅ ACK | ✅ XACK |
| 消息重试 | ❌ 无 | ✅ 支持 | ✅ 支持 | ✅ PEL 重发 |
| 消费者组 | ❌ 无 | ✅ 有 | ✅ 有 | ✅ Consumer Group |
| 消息回溯 | ❌ 不支持 | ✅ 支持 | ✅ 支持 | ✅ XRANGE |

> A: Pub/Sub 有三个致命局限：第一无消息持久化，消息即发即弃，订阅者不在线或网络抖动都会丢消息；第二无消费确认，发出去就不管了，消费失败也没法重试；第三无消费者组，多个订阅者都能收到全部消息，无法实现负载均衡。所以 Pub/Sub 只适合对可靠性要求不高的实时通知场景，比如在线用户的弹幕推送、管理后台实时刷新。

## 🔥 扩展追问 2：XREADGROUP 的 `>` 和 `0` 参数有什么区别？

**两种读取模式的对比**

```
┌─────────────────────────────────────────────────────────────────┐
│                    XREADGROUP 参数区别                           │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  XREADGROUP GROUP group consumer STREAMS stream ">"            │
│  └── ">" 表示只读取新消息，不读取 Pending 消息                 │
│                                                                 │
│  XREADGROUP GROUP group consumer STREAMS stream "0"            │
│  └── "0" 表示只读取 Pending 消息（之前获取但未确认的）         │
│                                                                 │
│  ┌─────────────────────────────────────────────────────────┐   │
│  │                                                         │   │
│  │  场景演示：                                              │   │
│  │                                                         │   │
│  │  Step1: 读取消息 msg1, msg2                            │   │
│  │          XREADGROUP ... STREAMS stream ">"             │   │
│  │          → 返回 [msg1, msg2]                           │   │
│  │          → msg1, msg2 进入 PEL（待确认）               │   │
│  │                                                         │   │
│  │  Step2: 处理 msg1 成功，ACK                            │   │
│  │          XACK stream group msg1                        │   │
│  │          → msg1 从 PEL 删除                            │   │
│  │          → msg1 处理完成                               │   │
│  │                                                         │   │
│  │  Step3: 处理 msg2 时消费者崩溃                          │   │
│  │                                                         │   │
│  │  Step4: 消费者重启，重新读取 Pending                    │   │
│  │          XREADGROUP ... STREAMS stream "0"             │   │
│  │          → 返回 [msg2]（之前的 Pending）              │   │
│  │          → msg2 重新被处理                             │   │
│  │                                                         │   │
│  │  Step5: 正常读取新消息                                  │   │
│  │          XREADGROUP ... STREAMS stream ">"             │   │
│  │          → 返回 [msg3]（新消息）                       │   │
│  │                                                         │   │
│  └─────────────────────────────────────────────────────────┘   │
│                                                                 │
│  实际代码：                                                      │
│  ┌─────────────────────────────────────────────────────────┐   │
│  │                                                         │   │
│  │  // 正常流程：读新消息                                  │   │
│  │  XREADGROUP GROUP g c STREAMS s ">"                    │   │
│  │                                                         │   │
│  │  // 消费者重启后：先读 Pending，再读新消息               │   │
│  │  XREADGROUP GROUP g c STREAMS s "0"  // Pending        │   │
│  │  XREADGROUP GROUP g c STREAMS s ">"  // 新消息         │   │
│  │                                                         │   │
│  └─────────────────────────────────────────────────────────┘   │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

> A: `>` 表示只读取新消息（New Entry），不会读取之前已经发给消费者但还没 ACK 的 Pending 消息；`0` 表示只读取 Pending 消息。正常流程用 `>` 读取新消息并处理，如果处理失败或消费者崩溃，下次启动时用 `0` 读取 Pending 消息重新处理，保证消息不丢失。实际代码通常是消费者启动时先 `0` 再 `>`。

## 🔥 扩展追问 3：Stream 和 Kafka 有什么异同？

**核心对比**

| 维度 | Redis Stream | Apache Kafka |
|------|-------------|-------------|
| **定位** | Redis 数据结构 | 分布式消息队列 |
| **持久化** | RDB/AOF + Stream 内部 | 日志文件系统 |
| **消息回溯** | ✅ XRANGE | ✅ 从 offset 回溯 |
| **消费者组** | ✅ Consumer Group | ✅ Consumer Group |
| **分区机制** | slot（最多 16384） | partition（可自定义） |
| **吞吐量** | ⭐⭐⭐⭐⭐ 单机高 | ⭐⭐⭐⭐⭐ 分布式高 |
| **消息堆积** | 有限（需 XTRIM） | 无限（按 retention） |
| **延迟** | ⭐⭐⭐⭐⭐ 极低 | ⭐⭐⭐ 中等 |
| **部署复杂度** | ⭐ 简单 | ⭐⭐⭐ 复杂 |

```
┌─────────────────────────────────────────────────────────────────┐
│                    选型决策树                                     │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  需要分布式部署、极高吞吐量？                                     │
│       │                                                         │
│       ├── YES ──▶ Kafka / RabbitMQ                             │
│       │                                                         │
│       └── NO ──▶ 单机或简单集群？                               │
│                      │                                          │
│                      ├── YES ──▶ Redis Stream ✅                │
│                      │                                          │
│                      └── NO ──▶ 需要严格顺序？                   │
│                                     │                           │
│                                     ├── YES ──▶ Kafka           │
│                                     └── NO ──▶ Stream / MQ      │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

> A: Redis Stream 和 Kafka 本质都是日志流，都能持久化、支持消费者组、支持消息回溯。区别是 Kafka 是专业的分布式消息队列，支持跨机器集群、无限消息堆积、超高吞吐量，适合大数据场景；Stream 是 Redis 的内置数据结构，部署简单、延迟极低，适合中小型项目的可靠消息场景，缺点是单机 Redis 的吞吐量不如 Kafka 分布式集群。如果项目已经用了 Redis，用 Stream 做可靠消息比引入 Kafka 更轻量。

## 🔥 扩展追问 4：Stream 如何防止消息堆积？

**三种控制方式**

```bash
# 方式1：MAXLEN - 保留最近 N 条
XTRIM stream MAXLEN 10000
# 删除超过 10000 条的所有旧消息

# 方式2：MAXLEN ~ - 近似裁剪（更高效）
XTRIM stream MAXLEN ~ 10000
# 近似裁剪，可能多删几条，但性能更好

# 方式3：MINID - 删除 ID 小于指定值的所有消息
XTRIM stream MINID 1527854169173-0
# 删除所有 ID 小于这个值的消息

# 方式4：XADD 时自动裁剪
XADD stream MAXLEN 1000 * field value
# 写入时自动保证不超过 1000 条
```

```java
// 自动清理实现
public class StreamAutoCleanup {
    
    private JedisPool jedisPool;
    private static final String STREAM_KEY = "notification:stream";
    private static final int MAX_LENGTH = 10000;
    
    /**
     * 定时清理（建议用定时任务）
     */
    @Scheduled(fixedRate = 60000)  // 每分钟
    public void cleanup() {
        try (Jedis jedis = jedisPool.getResource()) {
            // 近似裁剪，性能更好
            long trimmed = jedis.xtrim(STREAM_KEY, XTrimParams.Builder.maxLen(~MAX_LENGTH));
            
            if (trimmed > 0) {
                System.out.println("清理了 " + trimmed + " 条过期消息");
            }
        }
    }
    
    /**
     * 生产者写入时自动清理
     */
    public void addWithAutoCleanup(Jedis jedis, Map<String, String> fields) {
        // XADD + MAXLEN 原子操作
        StreamEntryID id = jedis.xadd(
            STREAM_KEY, 
            StreamEntryID.NEW_ENTRY, 
            fields,
            XAddParams.Builder.maxLen(MAX_LENGTH)  // 自动裁剪
        );
        
        System.out.println("消息已添加，ID: " + id);
    }
}
```

> A: Stream 有三种控制堆积的方式：MAXLEN 保留最近 N 条消息，近似裁剪（带 `~`）更高效；MINID 删除 ID 小于指定值的所有消息，适合按时间窗口清理；还可以在 XADD 时直接加 MAXLEN 参数，原子操作。实际生产建议用定时任务定期清理，保留最近 7 天或 10000 条的消息即可。

## 🔥 扩展追问 5：Stream 消费者组数量有限制吗？

**多个消费者组的实际用法**

```
┌─────────────────────────────────────────────────────────────────┐
│                    多个消费者组                                   │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  同一个 Stream 可以创建多个消费者组，互不影响：                   │
│                                                                 │
│  Stream ──▶ msg1, msg2, msg3, msg4, msg5                       │
│              │                                                   │
│              ├── Group A ──▶ consumer-1                        │
│              │                    （处理消息A）                   │
│              │                                                   │
│              ├── Group B ──▶ consumer-2                        │
│              │                    （处理消息B）                   │
│              │                                                   │
│              └── Group C ──▶ consumer-3                        │
│                               （处理消息C）                       │
│                                                                 │
│  场景：                                                          │
│  • Group A：发邮件通知                                           │
│  • Group B：发短信通知                                           │
│  • Group C：记录日志                                            │
│                                                                 │
│  ✅ 每条消息会被三个组分别处理一次                                │
│  ✅ 类似于 Kafka 的多个 consumer group                          │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

```bash
# 创建多个消费者组
XGROUP CREATE notification:stream email-workers $ MKSTREAM
XGROUP CREATE notification:stream sms-workers $ MKSTREAM
XGROUP CREATE notification:stream log-workers $ MKSTREAM

# 每个组独立消费同一份消息
XREADGROUP GROUP email-workers email-consumer-1 STREAMS notification:stream ">"
XREADGROUP GROUP sms-workers sms-consumer-1 STREAMS notification:stream ">"
XREADGROUP GROUP log-workers log-consumer-1 STREAMS notification:stream ">"
```

> A: 同一个 Stream 可以创建多个消费者组，每个组独立消费同一份消息，互不影响，类似于 Kafka 的多个 consumer group。实际场景比如：Group A 负责发邮件、Group B 负责发短信、Group C 负责记录日志，每条消息会被三个组分别处理一次。需要注意的是，每个组内的消息分配规则和单个组一致。

---

## 一句话总结（面试收尾用）

> **Pub/Sub 的核心局限是无消息持久化、无消费确认、无消费者组，消息即发即弃，只适合对可靠性要求不高的实时通知场景；Stream 通过消费者组和 Pending Entry List 实现可靠消息，消费者通过 XREADGROUP 读取消息后进入 PEL，处理成功后 XACK 从 PEL 删除，失败则通过 XPENDING 发现超时消息用 XCLAIM 重新分发；Stream 和 Kafka 本质都是日志流，都能持久化、支持消费者组和消息回溯，但 Stream 更轻量适合中小型项目，Kafka 适合高吞吐分布式场景。**

---

**面试加分话术：**

> "我之前做过一个订单通知系统，一开始用的 Pub/Sub，结果线上出过两次事故：一次是消费者重启期间的消息全丢了，用户投诉订单发货了没收到通知；另一次是消息处理失败后没法重试，用户收到重复通知。后来调研了三种方案：Kafka 太重要单独部署、RabbitMQ 维护成本高、Stream 正好能解决我们的问题。我们用 Stream 重构了通知系统，每条消息都会进 PEL，消费者处理成功后 ACK，失败就留在 PEL 里。每分钟有个定时任务扫描 XPENDING，如果消息空闲超过 30 秒就 XCLAIM 给另一个消费者重试。重构后消息零丢失，用户也没再收到重复通知。还有个经验：Stream 的消息 ID 用时间戳-序列号，天然有序，如果需要严格顺序消费，可以在消息体内加序列号字段自己判断。"
