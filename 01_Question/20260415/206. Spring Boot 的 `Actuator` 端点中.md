### 1.3 基础配置

```yaml
# application.yml

spring:
  application:
    name: my-service

management:
  # 端点根路径（默认 /actuator）
  endpoints:
    web:
      base-path: /actuator
  
  # 所有端点的通用配置
  endpoint:
    health:
      # 显示详情模式：never / when_authorized / always
      show-details: when_authorized
      # 聚合所有健康指标的状态（一个 down 则 total down）
      show-components: when_authorized
  
  # 公开哪些端点（生产推荐）
  endpoints:
    web:
      exposure:
        include: health,info,metrics,prometheus
        exclude: env,beans,mappings,configprops,loggers,heapdump
  
  # 健康检查的探测配置（K8s / K8s Liveness/Readiness）
  health:
    livenessState:
      enabled: true
    readinessState:
      enabled: true
    # 自定义探针端点
    probes:
      enabled: true

  # 指标配置
  metrics:
    # 指标 tags，可用于区分环境
    tags:
      application: ${spring.application.name}
    # 导出到 Prometheus
    export:
      prometheus:
        enabled: true
```

## 🔥 扩展追问 1：如何自定义 Health Indicator？

> A: **实现 `HealthIndicator` 接口或继承 `AbstractHealthIndicator`。**

```java
// 方式1：实现 HealthIndicator 接口
@Component
public class MySQLHealthIndicator implements HealthIndicator {

    @Autowired
    private DataSource dataSource;

    @Override
    public Health health() {
        try (Connection conn = dataSource.getConnection()) {
            // 执行健康检查
            boolean valid = conn.isValid(2);
            if (valid) {
                return Health.up()
                    .withDetail("database", "MySQL")
                    .withDetail("url", conn.getMetaData().getURL())
                    .build();
            }
        } catch (Exception e) {
            return Health.down()
                .withDetail("error", e.getMessage())
                .withException(e)
                .build();
        }
        return Health.unknown().build();
    }
}
```

```java
// 方式2：继承 AbstractHealthIndicator（推荐，更简洁）
@Component
public class RedisHealthIndicator extends AbstractHealthIndicator {

    @Autowired
    private StringRedisTemplate redisTemplate;

    @Override
    protected void doHealthCheck(Health.Builder builder) throws Exception {
        String result = redisTemplate.getConnectionFactory()
            .getConnection()
            .ping();
        
        if ("PONG".equals(result)) {
            builder.up()
                .withDetail("version", 
                    redisTemplate.getConnectionFactory()
                        .getConnection()
                        .serverCommands()
                        .info("server")
                        .getValue("version"));
        } else {
            builder.down()
                .withDetail("error", "Redis ping failed");
        }
    }
}
```

```java
// 方式3：自定义业务健康检查（检查依赖的外部服务）
@Component
public class ThirdPartyAPIHealthIndicator extends AbstractHealthIndicator {

    @Autowired
    private RestTemplate restTemplate;

    @Autowired
    private ThirdPartyProperties properties;

    @Override
    protected void doHealthCheck(Health.Builder builder) throws Exception {
        String url = properties.getHealthCheckUrl();
        
        try {
            ResponseEntity<String> response = restTemplate.getForEntity(
                url, String.class);
            
            if (response.getStatusCode().is2xxSuccessful()) {
                builder.up()
                    .withDetail("thirdPartyService", "OK")
                    .withDetail("responseTime", 
                        response.getHeaders().getDate());
            } else {
                builder.down()
                    .withDetail("thirdPartyService", "DEGRADED")
                    .withDetail("status", response.getStatusCode());
            }
        } catch (Exception e) {
            builder.down()
                .withDetail("thirdPartyService", "UNAVAILABLE")
                .withException(e);
        }
    }
}
```

## 🔥 扩展追问 2：/actuator/prometheus 是怎么把 Micrometer 指标暴露给 Prometheus 的？

> A: **Micrometer 充当指标"适配层"，将各监控系统的 API 统一化，PrometheusMeterRegistry 负责将指标转为 Prometheus 格式。**

```

│                         Micrometer 指标采集架构                             │
│                                                                             │
│  ┌─────────────┐   ┌─────────────┐   ┌─────────────┐   ┌─────────────┐     │
│  │ Micrometer  │   │ Micrometer  │   │ Micrometer  │   │ Micrometer  │     │
│  │   Timer     │   │   Counter   │   │   Gauge    │   │   Summary   │     │
│  └──────┬──────┘   └──────┬──────┘   └──────┬──────┘   └──────┬──────┘     │
│         │                 │                 │                 │           │
│         └─────────────────┼─────────────────┼─────────────────┘           │
│                           │                 │                               │
│                           ▼                 ▼                               │
│              ┌────────────────────────────┐                                 │
│              │     MeterRegistry         │  ← 统一注册表                   │
│              │  (核心接口，所有指标的容器) │                                 │
│              └─────────────┬──────────────┘                                 │
│                            │                                                │
│         ┌──────────────────┼──────────────────┐                              │
│         │                  │                  │                              │
│         ▼                  ▼                  ▼                              │
│  ┌─────────────┐   ┌─────────────┐   ┌─────────────┐                          │

│  │ MeterReg.  │   │ MeterReg.  │   │  MeterReg.  │                          │
```

```java
// Micrometer 核心 API 示例

// 1. Counter - 累加计数器
Counter counter = meterRegistry.counter("http_requests_total", 
    "method", "GET", "status", "200");
counter.increment();

// 2. Timer - 记录耗时
Timer timer = meterRegistry.timer("http_server_requests_seconds",
    "uri", "/api/users");
timer.record(() -> {
    // 执行业务逻辑
});

// 3. Gauge - 动态数值（需要外部维护值）
AtomicInteger activeUsers = new AtomicInteger(0);
meterRegistry.gauge("active_users", activeUsers);
activeUsers.incrementAndGet();

// 4. Summary - 百分位统计
DistributionSummary summary = meterRegistry.summary("response_size",
    "endpoint", "/api/users");
summary.record(1024);
```

## 🔥 扩展追问 3：Actuator 的端点路径如何自定义？如何给端点添加自定义参数？

> A: **通过 `management.endpoints.web.path-mapping` 可以修改端点路径。**

```yaml
management:
  endpoints:
    web:
      base-path: /manage  # 修改根路径
  endpoint:
    health:
      # 修改 health 端点的路径
      path: healthz
    metrics:
      path: stats

# 访问变为：
# /manage/healthz
# /manage/stats
```

```java
// 自定义带参数的端点
@Component
@Endpoint(id = "customQuery")
public class CustomQueryEndpoint {

    @ReadOperation
    public String query(
            @Selector(defaultValue = "default") String type,
            @QueryParameter("startDate") String startDate,
            @QueryParameter("endDate") String endDate) {
        return String.format("Query type=%s, from=%s to=%s", type, startDate, endDate);
    }

    @WriteOperation
    public String execute(
            @Selector String taskId,
            @RequestBody Map<String, Object> params) {
        // 执行自定义任务
        return "Task started: " + taskId;
    }
}
```

## 5. 一句话总结

> **Actuator 是 Spring Boot 的"仪表盘"，`/health` 暴露服务存活性、`/metrics` 暴露 Micrometer 指标、`/env` 和 `/beans` 暴露配置和 Bean 信息；生产环境必须用独立端口 + 最小化暴露策略，`/actuator/prometheus` 通过 Micrometer 的 PrometheusMeterRegistry 将所有指标转为 Prometheus 格式供监控系统抓取。**

---

## 6. 面试加分话术

> "我们项目里 Actuator 玩得比较深。拿 Prometheus 监控举例：引入 `micrometer-registry-prometheus` 依赖后，所有自定义指标（业务埋点、接口耗时、分库分表路由次数）自动注册到 `PrometheusMeterRegistry`，通过 `/actuator/prometheus` 统一暴露。生产上我们把 Actuator 端口配成 9090 只监听 127.0.0.1，Nginx 层对监控网络放行 `/actuator/prometheus` 和 `/actuator/health/livenessState`、`/actuator/health/readinessState`，K8s 的 LivenessProbe 和 ReadinessProbe 直接用这两个探针做滚动更新判断。健康检查我们自定义了两个：一个是数据库连接池 ` DruidHealthIndicator`，超时 2 秒不返回就 down；一个是外部 SSO 服务 `ThirdPartySSOHealthIndicator`，连续 3 次失败才会影响健康状态，防止网络抖动误判。自定义指标这块，我们用 `@Timed` 注解统一记录接口耗时，百分位数 P50/P90/P99 全自动生成，Grafana 看板直接配置就行。"
