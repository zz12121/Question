### 蓝绿部署核心原理

```
┌─────────────────────────────────────────────────────────────────┐
│                    蓝绿部署架构图                                 │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│   ┌─────────────────────────────────────────────────────────┐   │
│   │                    Load Balancer / Ingress              │   │
│   │                    (Nginx / 云 LB / K8s Service)         │   │
│   └────────────────────────────┬────────────────────────────┘   │
│                                │                                   │
│                                │ 流量切换                          │
│                                ▼                                   │
│   ┌──────────────────┐  旧版本  │  新版本  ┌──────────────────┐   │
│   │                  │          │           │                  │   │
│   │   [Blue]        │          │           │   [Green]        │   │
│   │   v1.0.0        │  ◀── 流量 │           │   v1.1.0        │   │
│   │   生产环境       │          │           │   预热环境       │   │
│   │                  │          │           │                  │   │
│   │   Pod × 10      │          │           │   Pod × 10      │   │
│   │                  │          │           │                  │   │
│   └──────────────────┘          │           └──────────────────┘   │
│                                 │                                   │
│          ┌──────────────────────┴──────────────────────┐         │
│          │                                              │         │
│          │   数据库 / 共享存储（蓝绿共用）               │         │
│          │                                              │         │
│          └──────────────────────────────────────────────┘         │
│                                                                 │
│   流程:                                                          │
│   1. 部署 v1.1.0 到 Green 环境（预热）                           │
│   2. 验证 Green 环境正常                                        │
│   3. 切换流量：Blue(v1.0) ←→ Green(v1.1)                        │
│   4. 观察一段时间，确认无问题                                    │
│   5. 销毁 Blue 环境（v1.0）                                     │
│                                                                 │
│   优势: 流量切换秒级，回滚秒级                                    │
│   代价: 双倍资源                                                │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

```java
// ========== 蓝绿切换核心逻辑 ==========

@Service
public class BlueGreenDeployService {
    
    @Autowired
    private KubernetesClient kubernetesClient;
    @Autowired
    private NotificationService notificationService;
    
    private static final String NAMESPACE = "production";
    private static final String SERVICE_NAME = "order-service";
    private static final String CURRENT_SLOT_LABEL = "slot";
    
    /**
     * 执行蓝绿切换
     * 流程：Green 预热 → 切换流量 → 观察 → 销毁 Blue
     */
    @Transactional
    public DeployResult switchToGreen(String greenVersion) {
        log.info("开始蓝绿切换，目标版本: {}", greenVersion);
        
        // 1. 检查 Green 环境是否就绪
        checkGreenHealth(greenVersion);
        
        // 2. 记录切换前状态（用于回滚）
        saveSwitchSnapshot(greenVersion);
        
        // 3. 切换 Service Selector
        switchServiceSlot("green");
        
        // 4. 灰度观察
        startObservation(greenVersion);
        
        // 5. 通知相关人员
        notificationService.sendDeployNotification(
            "蓝绿切换完成", "Green v" + greenVersion + " 已接收流量");
        
        return DeployResult.success(greenVersion);
    }
    
    /**
     * 检查 Green 环境健康状态
     */
    private void checkGreenHealth(String version) {
        // 查询 Green deployment 状态
        Deployment green = kubernetesClient.apps().deployments()
            .inNamespace(NAMESPACE)
            .withName("order-service-green")
            .get();
        
        if (green == null) {
            throw new DeployException("Green 环境不存在");
        }
        
        // 检查 Pod 就绪数量
        int readyReplicas = green.getStatus().getReadyReplicas();
        int desiredReplicas = green.getSpec().getReplicas();
        
        if (readyReplicas < desiredReplicas) {
            throw new DeployException(
                String.format("Green 环境 Pod 未就绪: %d/%d", readyReplicas, desiredReplicas));
        }
        
        // 探活接口验证
        for (int i = 0; i < 10; i++) {
            try {
                ResponseEntity<String> response = restTemplate.getForEntity(
                    "http://order-service-green:8080/actuator/health/live",
                    String.class);
                
                if (response.getStatusCode().is2xxSuccessful()) {
                    log.info("Green 环境健康检查通过");
                    return;
                }
            } catch (Exception e) {
                log.warn("Green 健康检查失败，第 {} 次重试", i + 1);
                Thread.sleep(3000);
            }
        }
        
        throw new DeployException("Green 环境健康检查超时");
    }
    
    /**
     * 切换 Service 的 selector
     * 核心操作：slot: blue → slot: green
     */
    private void switchServiceSlot(String targetSlot) {
        log.info("切换流量到 {} 环境", targetSlot);
        
        Service service = kubernetesClient.services()
            .inNamespace(NAMESPACE)
            .withName(SERVICE_NAME)
            .get();
        
        // 补丁更新 selector
        Map<String, String> selector = new HashMap<>();
        selector.put("app", "order-service");
        selector.put(CURRENT_SLOT_LABEL, targetSlot);
        
        service.getSpec().setSelector(selector);
        
        kubernetesClient.services()
            .inNamespace(NAMESPACE)
            .withName(SERVICE_NAME)
            .patch(service);
        
        log.info("Service selector 已更新: slot={}", targetSlot);
    }
    
    /**
     * 灰度观察阶段
     */
    private void startObservation(String version) {
        // 观察 5 分钟，监控错误率
        int observationMinutes = 5;
        
        for (int i = 0; i < observationMinutes; i++) {
            try {
                Thread.sleep(60000);  // 每分钟检查一次
                
                Metrics metrics = fetchMetrics(version);
                
                if (metrics.getErrorRate() > 0.01) {  // 错误率 > 1%
                    log.error("Green 环境错误率异常: {}，自动回滚", metrics.getErrorRate());
                    rollback();
                    throw new DeployException("错误率过高，自动回滚");
                }
                
                log.info("第 {} 分钟观察: 错误率={}, p99延迟={}ms",
                    i + 1, metrics.getErrorRate(), metrics.getP99Latency());
                
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
                break;
            }
        }
        
        log.info("观察期结束，Green 环境正常");
    }
    
    /**
     * 销毁 Blue 环境（老版本）
     */
    public void destroyBlue() {
        Deployment blue = kubernetesClient.apps().deployments()
            .inNamespace(NAMESPACE)
            .withName("order-service-blue")
            .get();
        
        if (blue != null) {
            kubernetesClient.apps().deployments()
                .inNamespace(NAMESPACE)
                .withName("order-service-blue")
                .delete();
            
            log.info("Blue 环境已销毁");
        }
    }
    
    /**
     * 回滚到 Blue
     */
    public void rollback() {
        log.info("执行回滚，切回 Blue 环境");
        
        // 1. 切回 Blue
        switchServiceSlot("blue");
        
        // 2. 通知
        notificationService.sendDeployNotification(
            "蓝绿回滚", "已切回 Blue 环境");
    }
}
```

## 🔥 扩展追问 1：数据库 schema 变更如何兼容蓝绿？

> **A:** 数据库变更是蓝绿部署最大的难题，需要**向下兼容**：

```
┌─────────────────────────────────────────────────────────────────┐
│              蓝绿部署中的数据库兼容性策略                          │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│   核心原则：新旧代码必须同时兼容数据库 schema                     │
│                                                                 │
│   可行方案：                                                    │
│                                                                 │
│   方案1: -expand-contract（扩展-收缩）                         │
│   ┌─────────────────────────────────────────────────────────┐  │
│   │  Phase 1: Expand                                        │  │
│   │  - 新增字段（允许 NULL）                                 │  │
│   │  - 新增索引（允许重复）                                  │  │
│   │  - 新版本代码：可读写新字段，也能读旧字段                │  │
│   │  - 旧版本代码：只读旧字段，不受影响                      │  │
│   └─────────────────────────────────────────────────────────┘  │
│   ┌─────────────────────────────────────────────────────────┐  │
│   │  Phase 2: Migrate                                       │  │
│   │  - 数据迁移（老数据填充新字段默认值）                    │  │
│   │  - 全量数据回填                                          │  │
│   └─────────────────────────────────────────────────────────┘  │
│   ┌─────────────────────────────────────────────────────────┐  │
│   │  Phase 3: Contract                                       │  │
│   │  - 新版本切流后，旧代码不再运行                          │  │
│   │  - 删除废弃字段                                          │  │
│   │  - 删除旧索引                                            │  │
│   └─────────────────────────────────────────────────────────┘  │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

```sql
-- ========== 数据库变更脚本示例 ==========

-- Phase 1: Expand（新版本上线前执行）
-- 新增字段（允许 NULL，有默认值）
ALTER TABLE orders ADD COLUMN customer_type VARCHAR(20) DEFAULT 'NORMAL' COMMENT '客户类型';

-- 新增索引（不起名带 version，防止冲突）
CREATE INDEX idx_orders_customer_type ON orders(customer_type);

-- Phase 2: Migrate（新版本运行一段时间后执行）
-- 数据回填
UPDATE orders SET customer_type = 'VIP' WHERE total_amount > 10000;
UPDATE orders SET customer_type = 'NORMAL' WHERE customer_type IS NULL;

-- Phase 3: Contract（旧版本彻底下架后执行）
-- 删除旧字段（可选，保守做法是不删）
-- ALTER TABLE orders DROP COLUMN legacy_field;
```

```java
// ========== 代码兼容性实现 ==========

@Entity
@Table(name = "orders")
public class Order {
    
    // 旧字段（保留兼容）
    @Column(name = "status")
    private Integer status;
    
    // 新字段
    @Column(name = "customer_type")
    private String customerType;
    
    // 兼容读写：新旧字段二选一
    @Transient
    public String getCustomerType() {
        // 优先读新字段
        if (customerType != null) {
            return customerType;
        }
        // 兜底：根据金额推导客户类型
        if (this.totalAmount != null && this.totalAmount.compareTo(new BigDecimal("10000")) > 0) {
            return "VIP";
        }
        return "NORMAL";
    }
    
    // 写操作只写新字段
    public void setCustomerType(String customerType) {
        this.customerType = customerType;
    }
}
```

## 🔥 扩展追问 2：蓝绿和金丝雀有什么区别？

> **A:** 两者核心区别在于流量切换策略：

| 维度 | 蓝绿部署 (Blue-Green) | 金丝雀部署 (Canary) |
|------|----------------------|-------------------|
| **流量切换** | 全量切换（0% → 100%） | 渐进切换（1% → 5% → 20% → 100%） |
| **回滚速度** | 秒级（全量切回） | 慢（需要逐步切回） |
| **资源成本** | 双倍资源 | 少量额外资源 |
| **风险控制** | 切换后才发现问题 | 小流量验证，降低影响范围 |
| **适用场景** | 高可用要求高、小版本发布 | 大版本、机器学习、复杂功能 |
| **灰度能力** | 无（只有是/否） | 强（可按用户/地区/版本灰度） |

```
┌─────────────────────────────────────────────────────────────────┐
│              蓝绿 vs 金丝雀 vs 滚动更新对比                       │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│   蓝绿部署：                                                    │
│   ┌─────────────────────────────────────────────────────────┐  │
│   │  v1: ████████████████████  (100%)                      │  │
│   │  v2:                     ████████████████████  (100%)  │  │
│   │       ↑ 切换瞬间                                          │  │
│   └─────────────────────────────────────────────────────────┘  │
│                                                                 │
│   金丝雀部署：                                                  │
│   ┌─────────────────────────────────────────────────────────┐  │
│   │  v1: ████████████████████████████                        │  │
│   │  v2: ████                                              │  │
│   │       └─1%──→ 5%──→ 20%──→ 50%──→ 100%                 │  │
│   └─────────────────────────────────────────────────────────┘  │
│                                                                 │
│   滚动更新（Rolling Update）：                                   │
│   ┌─────────────────────────────────────────────────────────┐  │
│   │  Pod1: [v1][v2][v2][v1][v1][v2]  (逐步替换)            │  │
│   └─────────────────────────────────────────────────────────┘  │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

```yaml
# ========== K8s 金丝雀配置 ==========

# 使用 Argo Rollouts 实现金丝雀
apiVersion: argoproj.io/v1alpha1
kind: Rollout
metadata:
  name: order-service
spec:
  replicas: 10
  strategy:
    canary:
      steps:
      - setWeight: 5       # 1. 先放 5% 流量到新版本
      - pause: {duration: 10m}  # 2. 观察 10 分钟
      - setWeight: 20      # 3. 增加到 20%
      - pause: {duration: 10m}  # 4. 再观察
      - setWeight: 50      # 5. 50%
      - pause: {}
      - setWeight: 100     # 6. 全量
      
      canaryMetadata:       # 新版本标签
        labels:
          version: canary
      stableMetadata:       # 老版本标签
        labels:
          version: stable
      
      trafficRouting:
        nginx:
          stableIngress: order-service-stable
          additionalIngressAnnotations:
            canary-by-weight: "true"
      
      analysis:
        templates:
        - templateName: error-rate-analysis
        startingStep: 1
        args:
        - name: service-name
          value: order-service-canary

---
# 金丝雀分析模板（自动监控 + 自动回滚）
apiVersion: argoproj.io/v1alpha1
kind: AnalysisTemplate
metadata:
  name: error-rate-analysis
spec:
  args:
  - name: service-name
  metrics:
  - name: error-rate
    interval: 1m
    successCondition: result[0] < 0.01
    failureLimit: 3
    provider:
      prometheus:
        address: http://prometheus:9090
        query: |
          sum(rate(http_requests_total{
            service="{{args.service-name}}",
            status=~"5.."
          }[5m]))
          /
          sum(rate(http_requests_total{
            service="{{args.service-name}}"
          }[5m]))
```

## 🔥 扩展追问 3：蓝绿部署如何验证新版本正确？

> **A:** 切流前的完整验证流程：

```java
@Service
public class GreenEnvironmentValidator {
    
    @Autowired
    private RestTemplate restTemplate;
    
    /**
     * 完整验证流程
     */
    public ValidationResult validateGreen(String greenUrl) {
        ValidationResult result = new ValidationResult();
        
        // 1. 健康检查
        result.add(check("health", healthCheck(greenUrl)));
        
        // 2. 接口功能验证
        result.add(check("create_order", testCreateOrder(greenUrl)));
        result.add(check("query_order", testQueryOrder(greenUrl)));
        result.add(check("cancel_order", testCancelOrder(greenUrl)));
        
        // 3. 冒烟测试（关键业务路径）
        result.add(check("smoke_test", smokeTest(greenUrl)));
        
        // 4. 数据一致性验证（Blue vs Green）
        result.add(check("data_consistency", verifyDataConsistency(greenUrl)));
        
        // 5. 性能基准测试
        result.add(check("performance", performanceTest(greenUrl)));
        
        return result;
    }
    
    /**
     * 冒烟测试：核心链路走一遍
     */
    private boolean smokeTest(String greenUrl) {
        try {
            // 用户登录
            String token = login(greenUrl, "13800138000", "password123");
            
            // 下单
            String orderId = createOrder(greenUrl, token, 1001L);
            
            // 支付
            payOrder(greenUrl, token, orderId, "ALIPAY");
            
            // 查询订单状态
            Order order = queryOrder(greenUrl, token, orderId);
            
            // 断言订单状态
            return "PAID".equals(order.getStatus());
            
        } catch (Exception e) {
            log.error("冒烟测试失败", e);
            return false;
        }
    }
    
    /**
     * 数据一致性：对比 Blue 和 Green 的查询结果
     */
    private boolean verifyDataConsistency(String greenUrl) {
        String blueUrl = "http://order-service-blue:8080";
        
        // 用同一订单 ID 查询 Blue 和 Green
        String testOrderId = "ORDER_" + System.currentTimeMillis();
        
        Order blueOrder = queryOrder(blueUrl, token, testOrderId);
        Order greenOrder = queryOrder(greenUrl, token, testOrderId);
        
        // 核心字段必须一致
        if (!blueOrder.equalsExcludingFields(greenOrder, "updateTime")) {
            log.error("数据不一致! Blue={}, Green={}", blueOrder, greenOrder);
            return false;
        }
        
        return true;
    }
    
    /**
     * 性能测试
     */
    private boolean performanceTest(String greenUrl) {
        int concurrentUsers = 50;
        int requestsPerUser = 100;
        
        long start = System.currentTimeMillis();
        // 压测...
        long duration = System.currentTimeMillis() - start;
        
        double avgLatency = duration / (double) (concurrentUsers * requestsPerUser);
        
        // p99 延迟 < 200ms
        return avgLatency < 200;
    }
}
```

## 🔥 扩展追问 4：蓝绿部署的回滚方案如何设计？

> **A:** 回滚分两种：**观察期快速回滚**和**事后回滚**：

```java
@Service
public class BlueGreenRollbackService {
    
    /**
     * 一键回滚（观察期使用）
     */
    public void rollbackToBlue() {
        log.info("执行紧急回滚，切回 Blue 环境");
        
        // 1. 切换 Service
        Service service = kubernetesClient.services()
            .inNamespace(NAMESPACE)
            .withName(SERVICE_NAME)
            .get();
        
        Map<String, String> selector = service.getSpec().getSelector();
        selector.put("slot", "blue");  // 切回 blue
        
        kubernetesClient.services()
            .inNamespace(NAMESPACE)
            .withName(SERVICE_NAME)
            .patch(service);
        
        // 2. 立即告警
        alertService.sendCritical("蓝绿回滚", "已切回 Blue 环境，请确认");
        
        // 3. 记录回滚事件
        recordRollbackEvent("green", "blue", "manual_rollback");
    }
    
    /**
     * 事后回滚（Green 已销毁情况）
     * 需要重新部署 Blue
     */
    public void rollbackBlueFromImage() {
        log.info("重新部署 Blue 环境");
        
        // 1. 查询 Blue 历史的镜像版本
        String blueImage = getBlueImageFromHistory();
        
        // 2. 重新创建 Blue Deployment
        Deployment blueDeployment = buildBlueDeployment(blueImage);
        kubernetesClient.apps().deployments()
            .inNamespace(NAMESPACE)
            .create(blueDeployment);
        
        // 3. 等待 Blue 就绪
        waitForDeploymentReady("order-service-blue");
        
        // 4. 切换流量
        switchServiceSlot("blue");
    }
    
    /**
     * 从历史记录获取 Blue 的镜像版本
     */
    private String getBlueImageFromHistory() {
```
