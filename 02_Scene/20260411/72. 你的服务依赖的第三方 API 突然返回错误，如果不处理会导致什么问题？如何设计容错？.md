```java
// 1. 定义第三方 API 调用的统一封装
@Service
public class ThirdPartyApiClient {
    
    @Autowired
    private RestTemplate restTemplate;
    @Autowired
    private PaymentFeignClient paymentFeign;
    
    // ========== 熔断器状态枚举 ==========
    public enum CircuitState {
        CLOSED,     // 正常，流量通行
        OPEN,       // 熔断，拒绝所有请求
        HALF_OPEN   // 半开，放行一个请求试探
    }
    
    // ========== 熔断器核心逻辑 ==========
    @Component
    public static class CircuitBreaker {
        
        private volatile CircuitState state = CircuitState.CLOSED;
        private AtomicLong failureCount = new AtomicLong(0);
        private AtomicLong successCount = new AtomicLong(0);
        private volatile long lastFailureTime = 0;
        
        // 配置
        private static final int FAILURE_THRESHOLD = 5;        // 失败5次熔断
        private static final int SUCCESS_THRESHOLD = 3;        // 成功后3次才关闭
        private static final long OPEN_DURATION_MS = 30000;     // 熔断30秒
        private static final long HALF_OPEN_REQUEST_COUNT = 1; // 半开只放行1个
        
        /**
         * 判断是否允许请求通过
         */
        public boolean allowRequest() {
            switch (state) {
                case CLOSED:
                    return true;
                case OPEN:
                    // 熔断超时，进入半开状态
                    if (System.currentTimeMillis() - lastFailureTime > OPEN_DURATION_MS) {
                        state = CircuitState.HALF_OPEN;
                        log.info("CircuitBreaker 进入半开状态");
                        return true;
                    }
                    return false;
                case HALF_OPEN:
                    return true;
            }
            return true;
        }
        
        /**
         * 记录调用成功
         */
        public void recordSuccess() {
            if (state == CircuitState.HALF_OPEN) {
                if (successCount.incrementAndGet() >= SUCCESS_THRESHOLD) {
                    state = CircuitState.CLOSED;
                    failureCount.set(0);
                    successCount.set(0);
                    log.info("CircuitBreaker 恢复正常");
                }
            } else {
                failureCount.set(0);  // 成功重置失败计数
            }
        }
        
        /**
         * 记录调用失败
         */
        public void recordFailure() {
            lastFailureTime = System.currentTimeMillis();
            
            if (state == CircuitState.HALF_OPEN) {
                // 半开状态下失败，立即重新熔断
                state = CircuitState.OPEN;
                log.warn("CircuitBreaker 半开状态失败，重新熔断");
            } else {
                if (failureCount.incrementAndGet() >= FAILURE_THRESHOLD) {
                    state = CircuitState.OPEN;
                    log.error("CircuitBreaker 触发熔断! 失败次数={}", FAILURE_THRESHOLD);
                }
            }
        }
        
        public CircuitState getState() {
            return state;
        }
    }
    
    // ========== 支付 API 调用示例 ==========
    @Autowired
    private CircuitBreaker paymentCircuitBreaker;
    
    private static final int TIMEOUT_MS = 3000;  // 超时3秒
    private static final int MAX_RETRIES = 2;
    
    /**
     * 带完整容错逻辑的支付调用
     */
    public PaymentResult invokePayment(PaymentRequest request) {
        // 1. 熔断器检查
        if (!paymentCircuitBreaker.allowRequest()) {
            log.warn("支付API熔断中，返回降级结果");
            return PaymentResult.degraded("支付服务暂时不可用，请稍后重试");
        }
        
        // 2. 带超时控制的调用
        long startTime = System.currentTimeMillis();
        try {
            PaymentResult result = executeWithTimeout(
                () -> paymentFeign.createPayment(request),
                TIMEOUT_MS
            );
            
            // 3. 成功记录
            paymentCircuitBreaker.recordSuccess();
            return result;
            
        } catch (TimeoutException e) {
            // 4. 超时处理
            paymentCircuitBreaker.recordFailure();
            log.error("支付API超时: {}ms", TIMEOUT_MS);
            return PaymentResult.timeout();
            
        } catch (Exception e) {
            // 5. 异常处理 + 记录失败
            paymentCircuitBreaker.recordFailure();
            log.error("支付API调用异常", e);
            return PaymentResult.error(e.getMessage());
        }
    }
    
    /**
     * 带重试的调用（仅对部分异常重试）
     */
    public <T> T invokeWithRetry(Supplier<T> supplier, int maxRetries) {
        Exception lastException = null;
        
        for (int i = 0; i <= maxRetries; i++) {
            try {
                return supplier.get();
            } catch (Exception e) {
                lastException = e;
                
                // 只有网络超时、连接拒绝才重试，业务异常不重试
                if (!isRetryableException(e)) {
                    throw e;
                }
                
                if (i < maxRetries) {
                    // 指数退避: 1s, 2s, 4s
                    long delay = (long) Math.pow(2, i) * 1000;
                    log.warn("调用失败，{}ms 后重试第{}次", delay, i + 1);
                    Thread.sleep(delay);
                }
            }
        }
        
        throw new RuntimeException("重试次数耗尽", lastException);
    }
    
    private boolean isRetryableException(Exception e) {
        return e instanceof TimeoutException 
            || e instanceof ConnectException
            || e instanceof SocketTimeoutException
            || (e.getCause() instanceof ConnectException);
    }
    
    /**
     * 带超时的执行
     */
    private <T> T executeWithTimeout(Supplier<T> supplier, long timeoutMs) throws TimeoutException {
        return CompletableFuture.supplyAsync(supplier)
            .timeout(timeoutMs, TimeUnit.MILLISECONDS)
            .exceptionally(ex -> {
                throw new RuntimeException(ex);
            })
            .join();
    }
}
```

## 🔥 扩展追问 1：限流和熔断有什么区别？什么时候用哪个？

> **A:** 限流保护自己，熔断保护下游。两者互补：

| 维度 | 限流 (RateLimiter) | 熔断 (CircuitBreaker) |
|------|-------------------|---------------------|
| **目的** | 保护自己不被冲垮 | 保护下游不被压死 |
| **触发条件** | 请求速率超过阈值 | 下游错误率超过阈值 |
| **处理方式** | 拒绝多余请求（返回429） | 拒绝所有请求，快速失败 |
| **恢复方式** | 自动放行 | 试探性放行（半开） |
| **作用位置** | 网关层/入口层 | 服务调用层 |
| **例子** | 限制 QPS ≤ 1000 | 5秒内失败>50%则熔断 |

```java
// 实际组合使用：限流 + 熔断
@Service
public class ProtectedApiClient {
    
    // 限流：保护自己，每秒最多 500 次调用
    private RateLimiter rateLimiter = RateLimiter.create(500);
    
    // 熔断：保护下游
    private CircuitBreaker circuitBreaker = new CircuitBreaker();
    
    public String callThirdPartyApi(String param) {
        // 1. 限流检查
        if (!rateLimiter.tryAcquire()) {
            log.warn("触发限流，拒绝请求");
            throw new RateLimitException("请求过于频繁，请稍后重试");
        }
        
        // 2. 熔断检查
        if (!circuitBreaker.allowRequest()) {
            log.warn("第三方API熔断中");
            return fallback();  // 走降级
        }
        
        // 3. 执行调用
        try {
            String result = thirdPartyHttpClient.post(param);
            circuitBreaker.recordSuccess();
            return result;
        } catch (Exception e) {
            circuitBreaker.recordFailure();
            return fallback();
        }
    }
}
```

## 🔥 扩展追问 2：重试次数怎么定？无限重试会怎样？

> **A:** 无限重试会放大故障，导致**重试风暴**（Retry Storm）：

```
┌─────────────────────────────────────────────────────────────────┐
│                    重试风暴示意图                                 │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  第三方 API 故障（100%失败）                                     │
│        │                                                        │
│        ▼                                                        │
│  10000 用户请求 ──▶ 全部失败                                     │
│        │                                                        │
│        ▼                                                        │
│  每个请求重试 3 次 ──▶ 30000 次重试请求                          │
│        │                                                        │
│        ▼                                                        │
│  第三方 API 雪上加霜 ──▶ 彻底宕机                                │
│                                                                 │
│  ⚠️ 后果：                                                      │
│  - 第三方彻底恢复时间更长                                         │
│  - 本服务线程池耗尽                                              │
│  - 网络带宽被打满                                                 │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

```java
// 重试策略配置
@Service
public class RetryStrategy {
    
    /**
     * 根据不同场景配置不同重试策略
     */
    public RetryPolicy selectPolicy(ApiType apiType) {
        switch (apiType) {
            case PAYMENT:
                // 支付：谨慎重试，最多1次（资金相关）
                return RetryPolicy.builder()
                    .maxRetries(1)
                    .backoffMs(500)
                    .retryableExceptions(new Class[]{TimeoutException.class})
                    .build();
                    
            case SMS:
                // 短信：允许多次重试（可延迟发送）
                return RetryPolicy.builder()
                    .maxRetries(3)
                    .backoffMs(1000)
                    .retryableExceptions(new Class[]{
                        TimeoutException.class, 
                        ServiceUnavailableException.class
                    })
                    .build();
                    
            case RISK_CHECK:
                // 风控：快速失败，不重试（走降级更划算）
                return RetryPolicy.builder()
                    .maxRetries(0)  // 不重试
                    .build();
                    
            default:
                return RetryPolicy.builder()
                    .maxRetries(2)
                    .backoffMs(500)
                    .build();
        }
    }
    
    /**
     * 退避策略：指数退避 + 抖动
     */
    public long calculateBackoffMs(int retryCount) {
        // 基础延迟：1s, 2s, 4s, 8s
        long baseDelay = (long) Math.pow(2, retryCount) * 1000;
        // 加随机抖动 ±25%，防止惊群效应
        long jitter = (long) (baseDelay * 0.25 * Math.random());
        return baseDelay + jitter;
    }
}
```

## 🔥 扩展追问 3：降级策略具体怎么设计？

> **A:** 降级策略分三个层次：

| 层次 | 策略 | 场景 |
|------|------|------|
| **业务降级** | 返回默认值/缓存值 | 风控、推荐、人脸识别 |
| **功能降级** | 关闭非核心功能 | 弹窗、积分、游戏化功能 |
| **兜底降级** | 返回友好提示 | 所有外部依赖不可用 |

```java
@Service
public class DegradationStrategy {
    
    // 一级：读缓存
    public UserProfile getUserProfileFallback(Long userId) {
        // 尝试从 Redis 缓存获取
        String cached = redisTemplate.opsForValue().get("user:profile:" + userId);
        if (cached != null) {
            log.info("风控降级，从缓存获取用户画像: {}", userId);
            return JSON.parseObject(cached, UserProfile.class);
        }
        
        // 二级：返回空画像（让业务侧决定怎么处理）
        log.warn("风控降级，返回空画像: userId={}", userId);
        return UserProfile.empty(userId);
    }
    
    // 二级：返回保守默认值
    public RiskLevel riskCheckFallback(Long userId, String orderId) {
        // 风控不可用时，走保守策略
        // 不拦截，但记录告警，人工后续处理
        alertService.send("风控降级，待人工审核: userId={}, orderId={}", userId, orderId);
        return RiskLevel.HIGH;  // 标记为高风险，后续人工跟进
    }
    
    // 三级：关闭非核心功能
    @Autowired
    private FeatureToggle featureToggle;
    
    public void handleThirdPartyUnavailable(String feature) {
        if (featureToggle.isCoreFeature(feature)) {
            // 核心功能降级，抛出异常让上层处理
            throw new ServiceUnavailableException(feature + " 暂时不可用");
        } else {
            // 非核心功能，直接关闭
            featureToggle.disable(feature);
            log.info("非核心功能已关闭: {}", feature);
        }
    }
}
```

## 🔥 扩展追问 4：Sentinel/Hystrix 怎么处理？

> **A:** 生产环境推荐用成熟的熔断框架：

```java
// ========== Sentinel 熔断规则 ==========
/*
# 熔断策略配置（推送到配置中心）
[
  {
    "resource": "paymentApi",
    "count": 10,
    "grade": 1,                    // 1=慢调用比例, 0=失败比例
    "slowRatioThreshold": 0.6,     // 60% 慢调用触发熔断
    "minRequestAmount": 5,         // 最小请求数
    "statIntervalMs": 1000,        // 统计时间窗口 1s
    "timeWindow": 10               // 熔断持续时间 10s
  }
]
*/

// 代码层面
@SentinelResource(value = "paymentApi",
                  blockHandler = "paymentBlockHandler",
                  fallback = "paymentFallback")
public PaymentResult callPayment(PaymentRequest request) {
    return paymentFeign.createPayment(request);
}

// 熔断时处理
public PaymentResult paymentBlockHandler(PaymentRequest request, BlockException e) {
    log.warn("Sentinel 熔断触发: {}", e.getClass().getSimpleName());
    return PaymentResult.degraded("支付服务繁忙");
}

// 异常时降级
public PaymentResult paymentFallback(PaymentRequest request, Throwable e) {
    log.error("支付调用异常", e);
    return PaymentResult.degraded("支付失败，请稍后重试");
}

// ========== Resilience4j 熔断器 ==========
/*
<dependency>
    <groupId>io.github.resilience4j</groupId>
    <artifactId>resilience4j-spring-boot2</artifactId>
</dependency>
*/

@Configuration
public class Resilience4jConfig {
    
    @Bean
    public CircuitBreakerRegistry circuitBreakerRegistry() {
        CircuitBreakerConfig config = CircuitBreakerConfig.custom()
            .failureRateThreshold(50)           // 50% 失败率触发熔断
            .slowCallRateThreshold(80)         // 80% 慢调用触发熔断
            .slowCallDurationThreshold(Duration.ofSeconds(3))  // 3s 算慢调用
            .waitDurationInOpenState(Duration.ofSeconds(30))  // 30s 后进入半开
            .permittedNumberOfCallsInHalfOpenState(3)         // 半开允许 3 个请求
            .slidingWindowType(SlidingWindowType.COUNT_BASED)  // 基于计数
            .slidingWindowSize(10)                            // 滑动窗口 10 个请求
            .minimumNumberOfCalls(5)                          // 最小 5 个请求才计算
            .build();
            
        return CircuitBreakerRegistry.of(config);
    }
}

@Service
public class PaymentServiceWithResilience4j {
    
    private final CircuitBreaker circuitBreaker;
    private final Retry retry;
    
    public PaymentServiceWithResilience4j(CircuitBreakerRegistry registry) {
        circuitBreaker = registry.circuitBreaker("paymentApi");
        retry = Retry.ofDefaults("paymentApi");
    }
    
    public PaymentResult callPayment(PaymentRequest request) {
        // 组合使用：重试 + 熔断 + 超时
        Supplier<PaymentResult> decoratedSupplier = CircuitBreaker
            .decorateSupplier(circuitBreaker, () -> paymentFeign.createPayment(request));
        
        decoratedSupplier = Retry
            .decorateSupplier(retry, decoratedSupplier);
            
        decoratedSupplier = TimeLimiter
            .decorateSupplier(timeLimiter, Duration.ofSeconds(3), decoratedSupplier);
            
        return decoratedSupplier.get();
    }
}
```

## 🔥 扩展追问 5：如何监控第三方 API 健康状态？

> **A:** 监控是容错的最后一道防线：

```java
@Service
public class ThirdPartyApiMonitor {
    
    // 健康度指标
    private AtomicLong totalCalls = new AtomicLong(0);
    private AtomicLong failedCalls = new AtomicLong(0);
    private ConcurrentHashMap<String, Long> latencyHistogram = new ConcurrentHashMap<>();
    
    // 记录调用指标
    public void recordCall(String apiName, long latencyMs, boolean success) {
        totalCalls.incrementAndGet();
        if (!success) {
            failedCalls.incrementAndGet();
        }
        
        // 记录延迟分布
        String bucket = latencyBucket(latencyMs);
        latencyHistogram.merge(apiName + ":" + bucket, 1L, Long::sum);
        
        // 实时告警
        if (!success && isConsecutiveFailure(apiName)) {
            alertService.send("第三方API连续失败告警: {}", apiName);
        }
    }
    
    // 监控指标暴露（Prometheus 格式）
    @GetMapping("/metrics/third-party")
    public String metrics() {
        StringBuilder sb = new StringBuilder();
        sb.append("# HELP third_party_api_total_total\n");
        sb.append("# TYPE third_party_api_total_total gauge\n");
        
        for (String api : getMonitoredApis()) {
            long total = totalCalls.get();
            long failed = failedCalls.get();
            double failRate = total > 0 ? (double) failed / total : 0;
            
            sb.append(String.format(
                "third_party_api_requests_total{api=\"%s\"} %d\n", api, total));
            sb.append(String.format(
                "third_party_api_failures_total{api=\"%s\"} %d\n", api, failed));
            sb.append(String.format(
                "third_party_api_fail_rate{api=\"%s\"} %.4f\n", api, failRate));
        }
        
        return sb.toString();
    }
    
    // 定期巡检：主动探测第三方 API
    @Scheduled(fixedDelay = 30000)
    public void healthCheck() {
        for (ThirdPartyApi api : allApis) {
            long start = System.currentTimeMillis();
            try {
                api.healthCheck();  // 发一个简单请求探测
                long latency = System.currentTimeMillis() - start;
                recordCall(api.getName(), latency, true);
            } catch (Exception e) {
                recordCall(api.getName(), 0, false);
                log.error("第三方API健康检查失败: {}", api.getName());
            }
        }
    }
}
```

> **总结：第三方 API 容错 → 熔断防雪崩 + 限流防冲垮 + 降级保核心 + 监控早发现**
>
> **核心是：把第三方 API 故障当常态来设计，而不是当异常。每一层都要有兜底，快速失败比傻等更优。**

> "我之前踩过一个坑：第三方支付接口超时 30 秒，我们没做超时控制，结果 Tomcat 200 个线程全部阻塞在支付调用上，用户完全进不了系统。后来加了 3 秒超时 + 熔断器，熔断后直接返回'支付服务繁忙'，用户至少能正常浏览商品，不卡在那儿。这个改动让我意识到：**超时是最容易被忽略的容错手段**，比熔断、降级都基础。另外还有个经验：所有第三方调用都要加 traceId，日志里要能串起来，方便排查是哪个第三方拖累了整体响应。"
