## 三、核心服务实现

### 1. 活动名额扣减（并发安全）

```java
@Service
public class RegistrationService {
    
    @Autowired private ActivityRepository activityRepo;
    @Autowired private RegistrationRepository registrationRepo;
    @Autowired private WaitlistService waitlistService;
    @Autowired private DistributedLock lock;
    
    /**
     * 报名核心流程
     */
    @Transactional
    public RegistrationResult register(Long userId, RegisterRequest request) {
        String activityId = request.getActivityId();
        
        // 1. 校验活动状态
        Activity activity = activityRepo.findByActivityId(activityId);
        validateActivity(activity);
        
        // 2. 校验报名时间窗口
        validateSignupTime(activity);
        
        // 3. 校验是否重复报名
        if (registrationRepo.existsByActivityIdAndUserId(activityId, userId)) {
            return RegistrationResult.fail("您已报名，请勿重复提交");
        }
        
        // 4. 校验名额（乐观锁 + 重试）
        return executeWithQuotaLock(activity, userId, request);
    }
    
    private RegistrationResult executeWithQuotaLock(
            Activity activity, Long userId, RegisterRequest request) {
        
        String lockKey = "register:quota:" + activity.getActivityId();
        String lockValue = UUID.randomUUID().toString();
        
        try {
            // 获取分布式锁（防止同一时刻多人抢最后名额）
            boolean locked = lock.tryLock(lockKey, lockValue, 5, TimeUnit.SECONDS);
            if (!locked) {
                return RegistrationResult.fail("系统繁忙，请稍后重试");
            }
            
            // 重新查询活动（获取最新名额）
            activity = activityRepo.findByActivityId(activity.getActivityId());
            
            // 5. 名额判断
            int remainQuota = activity.getQuota() - activity.getRegisteredCount();
            
            if (remainQuota <= 0) {
                // 名额已满，进入候补队列
                return handleWaitlist(activity, userId, request);
            }
            
            // 6. 创建报名记录（状态=待审核）
            Registration registration = Registration.builder()
                .registrationId(generateRegistrationId())
                .activityId(activity.getActivityId())
                .userId(userId)
                .userName(request.getUserName())
                .userPhone(request.getUserPhone())
                .formData(request.getFormData())
                .status(RegistrationStatus.PENDING)
                .build();
            
            registrationRepo.save(registration);
            
            // 7. 原子扣减名额（乐观锁）
            int updated = activityRepo.decrementQuota(activity.getId(), 
                activity.getRegisteredCount());
            
            if (updated == 0) {
                // 并发冲突，回滚
                registrationRepo.delete(registration);
                return RegistrationResult.fail("名额已被抢光，请稍后重试");
            }
            
            // 8. 触发审核流程
            auditService.submitForAudit(registration);
            
            return RegistrationResult.success(registration.getRegistrationId(), 
                activity.getRequireAudit() == 1 ? "报名成功，待审核" : "报名成功");
            
        } finally {
            lock.unlock(lockKey, lockValue);
        }
    }
    
    private void validateActivity(Activity activity) {
        if (activity == null) {
            throw new BusinessException("活动不存在");
        }
        if (activity.getStatus() != ActivityStatus.SIGNUP_OPEN) {
            throw new BusinessException("活动未开放报名");
        }
    }
    
    private void validateSignupTime(Activity activity) {
        LocalDateTime now = LocalDateTime.now();
        if (now.isBefore(activity.getSignupStartTime())) {
            throw new BusinessException("报名未开始");
        }
        if (now.isAfter(activity.getSignupEndTime())) {
            throw new BusinessException("报名已结束");
        }
    }
}
```

### 2. 审核服务

```java
@Service
public class AuditService {
    
    @Autowired private AuditRuleRepository ruleRepo;
    @Autowired private RegistrationRepository registrationRepo;
    @Autowired private NotificationService notificationService;
    
    public void submitForAudit(Registration registration) {
        Activity activity = registration.getActivity();
        
        // 免审场景：直接通过
        if (activity.getRequireAudit() == 0) {
            approveRegistration(registration, "系统自动审核通过");
            return;
        }
        
        // 执行审核规则
        AuditDecision decision = evaluateRules(registration);
        
        switch (decision.getAction()) {
            case AUTO_PASS -> approveRegistration(registration, decision.getReason());
            case AUTO_REJECT -> rejectRegistration(registration, decision.getReason());
            case MANUAL_REVIEW -> sendToManualQueue(registration);
        }
    }
    
    private AuditDecision evaluateRules(Registration registration) {
        List<AuditRule> rules = ruleRepo.findByEnabledOrderByPriorityDesc();
        
        for (AuditRule rule : rules) {
            boolean matched = evaluateRule(rule, registration);
            
            if (matched) {
                return AuditDecision.builder()
                    .action(rule.getAction())
                    .reason(rule.getRuleName() + "触发")
                    .ruleId(rule.getId())
                    .build();
            }
        }
        
        // 默认通过
        return AuditDecision.builder()
            .action(AuditAction.AUTO_PASS)
            .reason("符合报名条件")
            .build();
    }
    
    private boolean evaluateRule(AuditRule rule, Registration registration) {
        RuleConfig config = JSON.parseObject(rule.getRuleConfig(), RuleConfig.class);
        
        return switch (rule.getRuleType()) {
            case RuleType.WHITELIST -> 
                config.getUserIds().contains(registration.getUserId());
            
            case RuleType.BLACKLIST -> 
                config.getUserIds().contains(registration.getUserId()) ||
                config.getPhonePrefixes().stream()
                    .anyMatch(p -> registration.getUserPhone().startsWith(p));
            
            case RuleType.INTEGRAL_THRESHOLD -> 
                userService.getUserIntegral(registration.getUserId()) >= config.getMinIntegral();
            
            case RuleType.LEVEL_THRESHOLD -> 
                userService.getUserLevel(registration.getUserId()) >= config.getMinLevel();
            
            default -> false;
        };
    }
    
    @Transactional
    public void approveRegistration(Registration registration, String remark) {
        registration.setStatus(RegistrationStatus.APPROVED);
        registration.setAuditRemark(remark);
        registration.setAuditedAt(LocalDateTime.now());
        registrationRepo.save(registration);
        
        // 通知用户
        notificationService.send(Notification.builder()
            .userId(registration.getUserId())
            .type(NotificationType.REGISTRATION_APPROVED)
            .title("报名审核通过")
            .content("您报名的活动已审核通过，请按时参加")
            .build());
    }
}
```

### 3. 候补队列管理

```java
@Service
public class WaitlistService {
    
    @Autowired private WaitlistRepository waitlistRepo;
    @Autowired private ActivityRepository activityRepo;
    @Autowired private RegistrationService registrationService;
    @Autowired private NotificationService notificationService;
    
    /**
     * 名额释放时通知候补
     */
    public void notifyNextInWaitlist(String activityId) {
        Activity activity = activityRepo.findByActivityId(activityId);
        
        // 查询候补队列首位
        Waitlist waitlist = waitlistRepo
            .findFirstByActivityIdAndStatusOrderByPositionAsc(
                activityId, WaitlistStatus.WAITING);
        
        if (waitlist == null) {
            return; // 没有候补
        }
        
        // 更新为已通知状态
        waitlist.setStatus(WaitlistStatus.NOTIFIED);
        waitlist.setNotifiedAt(LocalDateTime.now());
        waitlist.setExpireAt(LocalDateTime.now().plusHours(24)); // 24小时内确认
        waitlistRepo.save(waitlist);
        
        // 发送通知
        notificationService.send(Notification.builder()
            .userId(waitlist.getUserId())
            .type(NotificationType.WAITLIST_NOTIFIED)
            .title("有名额空出，请尽快确认")
            .content("您排队的位置已到，请在24小时内确认报名")
            .build());
    }
    
    /**
     * 候补用户确认报名
     */
    @Transactional
    public RegistrationResult confirmFromWaitlist(Long userId, String activityId) {
        Waitlist waitlist = waitlistRepo
            .findByActivityIdAndUserId(activityId, userId);
        
        if (waitlist == null) {
            return RegistrationResult.fail("您不在候补名单中");
        }
        
        if (waitlist.getStatus() != WaitlistStatus.NOTIFIED) {
            return RegistrationResult.fail("候补状态已过期，请重新排队");
        }
        
        if (LocalDateTime.now().isAfter(waitlist.getExpireAt())) {
            // 超时，标记过期，通知下一位
            waitlist.setStatus(WaitlistStatus.EXPIRED);
            waitlistRepo.save(waitlist);
            notifyNextInWaitlist(activityId);
            return RegistrationResult.fail("确认超时，名额已释放给下一位");
        }
        
        // 转为正式报名
        return registrationService.registerDirectly(userId, activityId, 
            waitlist.getRegistrationId());
    }
}
```

```
├─────────────────────────────────────────────────────────────────────────┤
│                                                                          │
│  ┌─────────────┐     ┌─────────────┐     ┌─────────────┐               │
│  │   前端限流   │────▶│  令牌桶限流   │────▶│  队列缓冲    │               │
│  │ (按钮置灰)  │     │  (每S 500)   │     │  (异步处理)  │               │

│                                                  │                      │
│                                                  ▼                      │

│  │                     名额保护 - 三层防线                          │   │
│  │                                                                   │   │
│  │  L1: Redis 计数器（预扣减，请求拦截 90%）                         │   │
│  │      SETNX quota:{activityId} {quota}                            │   │
│  │                                                                   │   │
│  │  L2: 分布式锁（互斥，保证原子性）                                   │   │
│  │      Redisson Lock                                               │   │
│  │                                                                   │   │
│  │  L3: 数据库乐观锁（兜底，最终一致性）                               │   │
│  │      UPDATE activity SET registered_count = ?                   │   │
│  │      WHERE id = ? AND registered_count < quota                  │   │

│  └─────────────────────────────────────────────────────────────────┘   │
│                                                                          │
└─────────────────────────────────────────────────────────────────────────┘
```

```java
@Service
public class HighConcurrencyRegistrationService {
    
    @Autowired private RedisTemplate<String, Object> redisTemplate;
    
    /**
     * Redis 预扣减（第一道防线）
     */
    public boolean tryPreDecrementQuota(String activityId) {
        String quotaKey = "activity:quota:remain:" + activityId;
        Long remain = redisTemplate.opsForValue().decrement(quotaKey);
        return remain != null && remain >= 0;
    }
    
    /**
     * 回滚 Redis 预扣减
     */
    public void rollbackPreDecrement(String activityId) {
        String quotaKey = "activity:quota:remain:" + activityId;
        redisTemplate.opsForValue().increment(quotaKey);
    }
    
    /**
     * 初始化活动名额到 Redis
     */
    public void initQuotaToRedis(Activity activity) {
        String quotaKey = "activity:quota:remain:" + activity.getActivityId();
        int remainQuota = activity.getQuota() - activity.getRegisteredCount();
        redisTemplate.opsForValue().set(quotaKey, remainQuota);
    }
}
```

## 🔥 扩展追问 1：如何防止黄牛党重复报名？

> **A:** 多层防控体系：

| 层级 | 手段 | 说明 |
|-----|------|------|
| 业务层 | 唯一报名约束 | `(activity_id, user_id)` 唯一索引 |
| 网关层 | 行为检测 | 同 IP 多账号、机器行为识别 |
| 规则层 | 黑名单 | 手机号/IP/设备指纹命中黑名单 |
| 核验层 | 短信/人脸 | 报名后二次验证（秒杀场景）|

```java
// 黑名单规则示例
@Component
public class AntiScalperRules {
    
    public AuditDecision evaluateAntiScalper(Registration registration) {
        // 1. 手机号段黑名单
        if (phoneBlacklistService.isBlocked(registration.getUserPhone())) {
            return AuditDecision.reject("手机号在黑名单中");
        }
        
        // 2. 同一 IP 多账号检测
        String ip = getClientIp(registration);
        int todaySignups = registrationRepo.countByIpAndToday(ip);
        if (todaySignups > 3) {
            return AuditDecision.manualReview("同一IP报名次数异常");
        }
        
        // 3. 设备指纹去重
        String deviceId = registration.getDeviceId();
        if (registrationRepo.existsByActivityIdAndDeviceId(
                registration.getActivityId(), deviceId)) {
            return AuditDecision.reject("同一设备重复报名");
        }
        
        return AuditDecision.pass();
    }
}
```

## 🔥 扩展追问 2：报名费支付场景如何设计？

> **A:** 经典先报名后支付模式：

```
报名成功 ─────▶ 待支付 ─────▶ 支付成功 ─────▶ 完成报名
                  │                │
                  │ (超时30分钟)    │ (失败)
                  ▼                ▼
              自动取消 ─────▶ 释放名额 ─────▶ 候补通知
```

```java
@Service
public class PaymentService {
    
    @Transactional
    public void handlePaymentCallback(PaymentCallback callback) {
        Registration registration = registrationRepo
            .findByRegistrationId(callback.getOutTradeNo());
        
        if (callback.isSuccess()) {
            // 支付成功
            registration.setPayStatus(PayStatus.PAID);
            registration.setPaidAt(LocalDateTime.now());
            registration.setStatus(RegistrationStatus.APPROVED);
            registrationRepo.save(registration);
        } else {
            // 支付失败
            handlePaymentFailed(registration);
        }
    }
    
    // 定时任务：超时未支付释放名额
    @Scheduled(cron = "0 */5 * * * ?")
    public void releaseUnpaidSlots() {
        LocalDateTime expireTime = LocalDateTime.now().minusMinutes(30);
        
        List<Registration> unpaidList = registrationRepo
            .findByStatusAndCreatedAtBefore(
                RegistrationStatus.UNPAID, expireTime);
        
        for (Registration reg : unpaidList) {
            cancelRegistration(reg.getRegistrationId(), "支付超时自动取消");
            waitlistService.notifyNextInWaitlist(reg.getActivityId());
        }
    }
}
```

```java
// 审核规则配置（管理员后台配置）
@Configuration
public class AuditRuleEngine {
    
    public AuditDecision evaluate(Registration registration) {
        List<AuditRule> rules = ruleRepo.findAllEnabled();
        
        // 按优先级排序，取第一个匹配的
        for (AuditRule rule : sortByPriority(rules)) {
            if (!rule.matches(registration)) {
                continue;
            }
            
            return AuditDecision.builder()
                .action(rule.getAction())
                .reason(rule.getMatchDescription())
```

## 🔥 扩展追问 4：如何设计活动的"签到"功能？

> **A:** 支持多方式签到：

| 签到方式 | 实现 | 适用场景 |
|---------|------|---------|
| 二维码签到 | 扫码 + 人脸/手机号核验 | 大型会议 |
| 定位签到 | GPS 范围判断 | 户外活动 |
| 口令签到 | 输入正确口令 | 直播场景 |
| 刷卡签到 | RFID/NFC | 内部活动 |

```java
@Service
public class SigninService {
    
    /**
     * 二维码签到
     */
    public SigninResult signinByQRCode(Long userId, String activityId, 
            String qrCode, String verifyCode) {
        
        // 1. 验证二维码有效性
        QRTicket ticket = qrCodeService.verify(qrCode, activityId);
        if (ticket == null) {
            return SigninResult.fail("二维码无效");
        }
        
        // 2. 二次验证（手机后四位 / 人脸）
        if (!ticket.getVerifyCode().equals(verifyCode)) {
            return SigninResult.fail("验证码错误");
        }
        
        // 3. 执行签到
        return executeSignin(userId, activityId, ticket);
    }
    
    @Transactional
    public SigninResult executeSignin(Long userId, String activityId, 
            QRTicket ticket) {
        
        Registration registration = registrationRepo
            .findByActivityIdAndUserId(activityId, userId);
        
        if (registration.getStatus() != RegistrationStatus.APPROVED) {
            return SigninResult.fail("报名未通过或已取消");
        }
        
        registration.setStatus(RegistrationStatus.SIGNED_IN);
        registration.setSignedAt(LocalDateTime.now());
        registration.setSigninMethod("QR_CODE");
        registrationRepo.save(registration);
        
        return SigninResult.success("签到成功");
    }
}
```

## 一句话总结

> **活动报名系统的核心是：时间窗口校验 + 名额原子扣减（Redis预扣→分布式锁→乐观锁）+ 规则引擎审核 + 候补队列通知，通过分层防护应对高并发抢名额场景。**
