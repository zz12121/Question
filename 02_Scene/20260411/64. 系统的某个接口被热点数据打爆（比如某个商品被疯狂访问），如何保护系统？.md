### 2.1 多级缓存设计

```java
// ========== 1. 三级缓存实现 ==========
@Service
public class ProductCacheService {
    
    // L1: 本地缓存（Caffeine），访问速度最快
    private final Cache<Long, Product> localCache = Caffeine.newBuilder()
        .maximumSize(1000)                    // 最多缓存 1000 个商品
        .expireAfterWrite(Duration.ofSeconds(10))  // 10秒过期
        .recordStats()                        // 开启统计
        .build();
    
    @Autowired
    private RedisTemplate<String, Product> redisTemplate;
    
    @Autowired
    private ProductMapper productMapper;
    
    /**
     * 三级缓存读取
     * L1(本地) → L2(Redis) → L3(DB)
     */
    public Product getProduct(Long productId) {
        // L1: 本地缓存（纳秒级）
        Product product = localCache.getIfPresent(productId);
        if (product != null) {
            return product;
        }
        
        // L2: Redis 缓存（微秒级）
        String redisKey = "product:" + productId;
        product = redisTemplate.opsForValue().get(redisKey);
        if (product != null) {
            // 回填 L1
            localCache.put(productId, product);
            return product;
        }
        
        // L3: 数据库（毫秒级）
        product = productMapper.selectById(productId);
        if (product != null) {
            // 写回 L2 和 L1
            redisTemplate.opsForValue().set(redisKey, product, Duration.ofMinutes(5));
            localCache.put(productId, product);
        }
        
        return product;
    }
    
    /**
     * 更新缓存：双删策略
     */
    @Transactional
    public void updateProduct(Product product) {
        // 1. 先删 Redis
        redisTemplate.delete("product:" + product.getId());
        
        // 2. 更新 DB
        productMapper.updateById(product);
        
        // 3. 延迟删 Redis（等主从同步完成）
        CompletableFuture.delayedExecutor(500, TimeUnit.MILLISECONDS)
            .execute(() -> redisTemplate.delete("product:" + product.getId()));
        
        // 4. 删除本地缓存（下次请求会从 L2/L3 重新加载）
        localCache.invalidate(product.getId());
    }
}
```

### 2.2 热点数据检测与自动隔离

```java
// ========== 1. 热点数据检测器 ==========
@Service
public class HotspotDetector {
    
    // 滑动窗口：统计最近 5 秒的访问频率
    private LoadingCache<Long, LongAdder> accessCounter = Caffeine.newBuilder()
        .maximumSize(10000)
        .expireAfterWrite(Duration.ofSeconds(5))
        .build(key -> new LongAdder());
    
    // 热点阈值：5 秒内访问超过 100 次
    private static final int HOT_THRESHOLD = 100;
    
    // 热点数据缓存：Key=商品ID, Value=热点数据
    private final Cache<Long, Object> hotspotCache = Caffeine.newBuilder()
        .maximumSize(100)
        .expireAfterWrite(Duration.ofSeconds(30))
        .build();
    
    /**
     * 记录访问，自动识别热点
     */
    public void recordAccess(Long productId) {
        accessCounter.get(productId).increment();
        
        // 检测是否为热点
        if (accessCounter.get(productId).sum() > HOT_THRESHOLD) {
            markAsHotspot(productId);
        }
    }
    
    /**
     * 标记为热点，加入热点缓存
     */
    private void markAsHotspot(Long productId) {
        // 异步加载热点数据到本地缓存
        CompletableFuture.runAsync(() -> {
            Object data = loadFromRedisOrDB(productId);
            hotspotCache.put(productId, data);
            log.info("检测到热点数据，productId={}", productId);
        });
    }
    
    /**
     * 获取热点数据
     */
    public Object getHotspotData(Long productId) {
        return hotspotCache.getIfPresent(productId);
    }
}

// ========== 2. 商品查询增强 ==========
@Service
public class ProductService {
    
    @Autowired
    private HotspotDetector hotspotDetector;
    
    @Autowired
    private ProductCacheService cacheService;
    
    public Product getProduct(Long productId) {
        // 记录访问（异步，不阻塞主流程）
        hotspotDetector.recordAccess(productId);
        
        // 热点数据走独立缓存
        Object hotspotData = hotspotDetector.getHotspotData(productId);
        if (hotspotData != null) {
            return (Product) hotspotData;
        }
        
        // 普通走三级缓存
        return cacheService.getProduct(productId);
    }
}
```

```java
@Target(ElementType.METHOD)
@Retention(RetentionPolicy.RUNTIME)
public @interface RateLimit {
    int value() default 100;           // 每秒允许的请求数
    int timeoutMs() default 0;          // 0 = 直接拒绝
}

@Aspect
@Component
public class RateLimitAspect {
    
    private ConcurrentHashMap<String, RateLimiter> limiters = new ConcurrentHashMap<>();
    
    @Around("@annotation(rateLimit)")
    public Object around(ProceedingJoinPoint pjp, RateLimit rateLimit) throws Throwable {
        String key = getMethodKey(pjp);
        
        // 获取或创建限流器
        RateLimiter limiter = limiters.computeIfAbsent(key, 
            k -> RateLimiter.create(rateLimit.value()));
        
        // 尝试获取令牌
        if (!limiter.tryAcquire(rateLimit.timeoutMs(), TimeUnit.MILLISECONDS)) {
            throw new BizException("请求过于频繁，请稍后再试");
        }
        
        return pjp.proceed();
    }
}
```

### 2.4 本地队列 + 异步处理削峰

```java
// ========== 热点数据访问计数（异步批量写入）==========
@Service
public class HotspotAccessService {
    
    // 本地队列，暂存访问记录
    private BlockingQueue<AccessLog> localQueue = new LinkedBlockingQueue<>(10000);
    
    @PostConstruct
    public void init() {
        // 启动批量消费线程
        CompletableFuture.runAsync(this::batchConsumer);
    }
    
    /**
     * 记录访问（快速写入本地队列）
     */
    public void recordAccess(Long productId, Long userId) {
        AccessLog log = new AccessLog(productId, userId, System.currentTimeMillis());
        
        if (!localQueue.offer(log)) {
            // 队列满了，降级处理：直接返回，不阻塞
            metrics.increment("access.queue.full");
        }
    }
    
    /**
     * 批量消费队列，批量写入 DB
     */
    private void batchConsumer() {
        List<AccessLog> batch = new ArrayList<>(100);
        
        while (true) {
            try {
                AccessLog log = localQueue.poll(1, TimeUnit.SECONDS);
                if (log != null) {
                    batch.add(log);
                }
                
                // 凑够 100 条或超时 1 秒，批量写入
                if (batch.size() >= 100 || (batch.size() > 0 && batch.size() < 100)) {
                    batchInsert(batch);
                    batch.clear();
                }
            } catch (Exception e) {
                log.error("批量消费失败", e);
            }
        }
    }
}
```

## 🔥 扩展追问 1：缓存和 DB 数据不一致怎么办？

> **A:** 缓存问题的核心是"删除而非更新"，推荐 Cache Aside + 延迟双删。

```java
// ========== Cache Aside 模式 ==========
/**
 * 读：Cache → DB → 写回 Cache
 * 写：先更新 DB，再删除 Cache（不是更新 Cache！）
 */
@Service
public class ProductServiceWithCacheAside {
    
    /**
     * 读
     */
    public Product getProduct(Long id) {
        // 1. 先查缓存
        Product cached = redisTemplate.opsForValue().get("product:" + id);
        if (cached != null) {
            return cached;
        }
        
        // 2. 缓存没有，查 DB
        Product product = productMapper.selectById(id);
        
        // 3. 写回缓存
        if (product != null) {
            redisTemplate.opsForValue().set("product:" + id, product, Duration.ofMinutes(5));
        }
        
        return product;
    }
    
    /**
     * 写（关键：删除而非更新）
     */
    @Transactional
    public void updateProduct(Product product) {
        // 1. 先更新 DB
        productMapper.updateById(product);
        
        // 2. 再删除缓存（不是更新！）
        redisTemplate.delete("product:" + product.getId());
        
        // 3. 延迟双删（解决主从延迟问题）
        CompletableFuture.delayedExecutor(500, TimeUnit.MILLISECONDS)
            .execute(() -> redisTemplate.delete("product:" + product.getId()));
    }
}
```

## 🔥 扩展追问 2：热点 key 导致 Redis 也扛不住怎么办？

> **A:** Redis 扛不住热点的原因是指令都打到同一个分片。可以做热点 key 拆分或本地缓存兜底。

```java
// ========== 方案一：热点 key 打散到多个 key ==========
@Service
public class DistributedCacheService {
    
    /**
     * 热点 key 自动打散
     * 100 个节点，每秒请求打散到不同的 key
     */
    public String getHotspotKey(String key) {
        // 1. 检测是否为热点
        String hotspotKey = "hotspot:" + key;
        if (redisTemplate.hasKey(hotspotKey)) {
            // 2. 热点打散：加随机后缀
            int suffix = ThreadLocalRandom.current().nextInt(100);
            return key + ":shard:" + suffix;
        }
        return key;
    }
    
    /**
     * 热点数据预热到多个分片
     */
    public void preheatHotspot(Long productId) {
        Product product = productMapper.selectById(productId);
        
        // 预热到多个分片
        for (int i = 0; i < 100; i++) {
            redisTemplate.opsForValue().set(
                "product:" + productId + ":shard:" + i, 
                product, 
                Duration.ofMinutes(10)
            );
        }
        
        // 标记为热点
        redisTemplate.opsForValue().set("hotspot:product:" + productId, "1");
    }
}
```

## 🔥 扩展追问 3：如何识别热点 key？

> **A:** Redis 本身有 Monitor 命令，但生产慎用。可以借助 RedisCluster 或客户端埋点。

```java
// ========== 1. Redis 客户端埋点 ==========
@Component
public class RedisHotspotTracker {
    
    // 滑动窗口统计访问频率
    private LoadingCache<String, LongAdder> accessStats = Caffeine.newBuilder()
        .maximumSize(100000)
        .expireAfterWrite(Duration.ofSeconds(60))
        .build(k -> new LongAdder());

     * 在 Redis 操作时自动统计
     */
    public <T> T execute(RedisCallback<T> callback) {

        T result = callback.execute();

        }

    }
```

## 🔥 扩展追问 4：如何设计熔断降级策略？

> **A:** 熔断的核心是"快速失败 + 自动恢复"，防止级联故障。

```java
// ========== Sentinel 熔断规则 ==========
@PostConstruct
public void initDegradeRules() {
    List<DegradeRule> rules = new ArrayList<>();
    
    DegradeRule rule = new DegradeRule("getProduct")
        .setGrade(CircuitBreakerStrategy.ERROR_RATIO.getType())  // 按错误比例
        .setCount(0.5)           // 错误比例超过 50%
        .setMinRequestAmount(100) // 最小请求数 100
        .setStatIntervalMs(1000) // 1 秒内统计
        .setSlowRatioThreshold(2.0)  // RT 超过 2 秒算慢调用
        .setTimeWindow(10);       // 熔断 10 秒后进入半开状态
    
    rules.add(rule);
    DegradeRuleManager.loadRules(rules);
}

// ========== 自定义熔断器 ==========
@Service
public class CustomCircuitBreaker {
    
    private final Map<String, CircuitState> circuits = new ConcurrentHashMap<>();
    
    private static final int FAIL_THRESHOLD = 50;        // 失败次数阈值
    private static final int SLOW_THRESHOLD_MS = 1000;     // 慢调用阈值
    private static final Duration HALF_OPEN_DURATION = Duration.ofSeconds(30);
    
    public <T> T call(String key, Supplier<T> supplier) {
        CircuitState state = circuits.get(key);
        
        // 熔断中
        if (state == CircuitState.OPEN) {
            throw new CircuitBreakerOpenException(key);
        }
        
        try {
            T result = supplier.get();
            
            // 成功：重置计数器
            circuits.remove(key);
            return result;
            
        } catch (Exception e) {
            // 失败：增加计数器
            CircuitCounter counter = circuits.computeIfAbsent(key, k -> new CircuitCounter());
            counter.failCount.incrementAndGet();
            
            if (counter.failCount.get() > FAIL_THRESHOLD) {
                // 触发熔断
                circuits.put(key, new CircuitState(CircuitState.OPEN, 
                    Instant.now().plus(HALF_OPEN_DURATION)));
            }
            
            throw e;
        }
    }
}
```

## 🔥 扩展追问 5：如何预热热点数据？

> **A:** 热点数据预热是防止"缓存击穿"的关键，要提前加载。

```java
// ========== 1. 定时预热 ==========
@Service
public class HotspotPreheatService {
    
    @Autowired
    private ProductMapper productMapper;
    
    @Autowired
    private ProductCacheService cacheService;
    
    @Autowired
    private HotspotDetector hotspotDetector;
    
    /**
     * 定时预热热点数据
     * 每天凌晨 3 点执行
     */
    @Scheduled(cron = "0 0 3 * * ?")
    public void preheatHotspotData() {
        // 1. 查询热点商品列表（根据历史访问数据）
        List<Long> hotspotProductIds = productMapper.selectTopHotProducts(100);
        
        for (Long productId : hotspotProductIds) {
            try {
                // 2. 预热到各级缓存
                Product product = productMapper.selectById(productId);
                
                // 写 Redis
                redisTemplate.opsForValue().set(
                    "product:" + productId, 
                    product, 
                    Duration.ofMinutes(10)
                );
                
                // 写本地缓存
                cacheService.getLocalCache().put(productId, product);
                
                // 标记热点
                hotspotDetector.markAsHotspot(productId);
                
            } catch (Exception e) {
                log.error("预热失败 productId={}", productId, e);
            }
        }
        
        log.info("热点数据预热完成，共 {} 个商品", hotspotProductIds.size());
    }
}

// ========== 2. 热点活动提前预热 ==========
@Service
public class FlashSalePreheatService {
    
    /**
     * 秒杀活动开始前 5 分钟预热
     */
    @Scheduled(fixedDelay = 300000)  // 每 5 分钟检查一次
    public void preheatFlashSale() {
        List<FlashSaleActivity> activities = 
            flashSaleMapper.selectStartingSoon(LocalDateTime.now().plusMinutes(5));
        
        for (FlashSaleActivity activity : activities) {
            List<Long> productIds = productMapper.selectIdsByActivity(activity.getId());
            
            for (Long productId : productIds) {
                Product product = productMapper.selectById(productId);
                
                // 预热到 Redis 集群的所有分片
                for (int i = 0; i < 100; i++) {
                    redisTemplate.opsForValue().set(
                        "flash:product:" + productId + ":shard:" + i,
                        product,
                        Duration.ofHours(2)
                    );
                }
                
                log.info("秒杀商品预热完成: {}", productId);
            }
        }
    }
}
```

## 一句话总结

> **热点数据保护的核心是"多层缓存 + 限流熔断 + 热点隔离"：L1 本地缓存扛住 90% 热数据、热点 key 打散到多分片、Sentinel 按参数限流和熔断兜底、热点检测后预热数据到本地缓存，延迟双删保证缓存一致性，防止击穿/穿透/雪崩。**

> "之前双十一零点峰值时，某爆款商品详情页 10 万 QPS 直接把 Redis 打挂了，所有请求穿透到 DB，数据库连接池瞬间耗尽，整个服务雪崩。后来我做了几个优化：第一是加本地缓存（Caffeine），热点数据直接在 JVM 内存里返回，不走 Redis，本地缓存命中率能到 70-80%，Redis 压力降了 90%；第二是热点 key 自动识别，用滑动窗口统计访问频率，超过阈值自动把数据预热到 100 个本地缓存分片；第三是加了 Sentinel 限流，单机每秒超过 2000 QPS 直接返回兜底页，不让请求打到缓存层。现在这套架构抗住了双十一峰值，也没什么运维压力。"
