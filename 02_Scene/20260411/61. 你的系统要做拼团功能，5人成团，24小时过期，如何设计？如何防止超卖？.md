### 2.2 核心服务实现

```java
// ========== 1. 拼团服务 ==========
@Service
public class GroupPurchaseService {
    
    @Autowired
    private GroupPurchaseMapper groupMapper;
    @Autowired
    private OrderService orderService;
    @Autowired
    private StockService stockService;
    
    /**
     * 创建拼团（团长开团）
     */
    @Transactional
    public GroupPurchaseDTO createGroup(Long userId, Long skuId, Long activityId) {
        // 1. 校验活动状态和库存
        PromotionActivity activity = getActivity(activityId);
        if (!activity.isActive() || activity.isExpired()) {
            throw new BizException("活动未开始或已结束");
        }
        
        // 2. 扣减活动库存（乐观锁）
        int rows = stockMapper.decrementStockWithLock(activityId, skuId, 1);
        if (rows == 0) {
            throw new BizException("库存不足");
        }
        
        // 3. 创建拼团
        PromotionGroup group = new PromotionGroup();
        group.setGroupNo(generateGroupNo());
        group.setActivityId(activityId);
        group.setSkuId(skuId);
        group.setHeadUserId(userId);
        group.setCurrentCount(1);
        group.setTargetCount(5);
        group.setStatus(GroupStatus.OPEN);
        group.setExpireAt(LocalDateTime.now().plusHours(24));
        groupMapper.insert(group);
        
        // 4. 创建团长成员记录
        GroupMember headMember = new GroupMember();
        headMember.setGroupId(group.getId());
        headMember.setUserId(userId);
        headMember.setRole(MemberRole.HEAD);
        headMember.setStatus(MemberStatus.JOINED);  // 团长直接算加入
        groupMemberMapper.insert(headMember);
        
        // 5. 创建待支付订单
        Order order = orderService.createGroupOrder(userId, skuId, group.getGroupNo());
        
        return buildGroupDTO(group, order);
    }
    
    /**
     * 参与拼团（团员参团）
     */
    @Transactional
    public GroupPurchaseDTO joinGroup(Long userId, String groupNo) {
        // 1. 查询拼团
        PromotionGroup group = groupMapper.selectByGroupNo(groupNo);
        if (group == null || group.getStatus() != GroupStatus.OPEN) {
            throw new BizException("拼团不存在或已结束");
        }
        
        // 2. 检查是否过期
        if (LocalDateTime.now().isAfter(group.getExpireAt())) {
            throw new BizException("拼团已过期");
        }
        
        // 3. 检查是否已参加过
        if (groupMemberMapper.existsByUserAndGroup(userId, group.getId())) {
            throw new BizException("您已在此拼团中");
        }
        
        // 4. 扣减活动库存（乐观锁）
        int rows = stockMapper.decrementStockWithLock(group.getActivityId(), group.getSkuId(), 1);
        if (rows == 0) {
            throw new BizException("库存不足");
        }
        
        // 5. 加入拼团
        groupMemberMapper.insert(new GroupMember(userId, group.getId(), MemberRole.MEMBER));
        
        // 6. 更新拼团人数
        int newCount = groupMapper.incrementCount(group.getId());
        
        // 7. 创建待支付订单
        Order order = orderService.createGroupOrder(userId, group.getSkuId(), groupNo);
        
        // 8. 检查是否成团
        if (newCount >= group.getTargetCount()) {
            completeGroup(group.getId());
        }
        
        return buildGroupDTO(group, order);
    }
    
    /**
     * 扣减库存（乐观锁实现）
     */
    // Mapper XML
    // <update id="decrementStockWithLock">
    //   UPDATE promotion_activity_stock
    //   SET lock_stock = lock_stock + #{count},
    //       version = version + 1
    //   WHERE activity_id = #{activityId}
    //     AND sku_id = #{skuId}
    //     AND lock_stock + #{count} + sold_stock <= total_stock
    //     AND version = #{version}
    // </update>
    
    /**
     * 订单支付成功后回调
     */
    @Transactional
    public void onOrderPaid(String orderNo) {
        Order order = orderService.getByOrderNo(orderNo);
        
        // 更新成员状态
        groupMemberMapper.updateStatusByOrderNo(orderNo, MemberStatus.PAID);
        
        // 真正扣减 SKU 库存（如果之前只是锁定）
        stockService.confirmDeduct(order.getSkuId(), 1);
        
        // 检查是否成团
        PromotionGroup group = groupMapper.selectByGroupNo(order.getGroupNo());
        if (group.getCurrentCount() >= group.getTargetCount()) {
            completeGroup(group.getId());
        }
    }
    
    /**
     * 成团成功
     */
    private void completeGroup(Long groupId) {
        PromotionGroup group = groupMapper.selectById(groupId);
        
        // 只有 OPEN 状态才能转 SUCCESS
        int rows = groupMapper.updateStatusIfMatch(groupId, GroupStatus.OPEN, GroupStatus.SUCCESS);
        if (rows == 0) {
            return;  // 已经被其他线程处理
        }
        
        // 通知订单系统确认所有订单
        List<String> orderNos = groupMemberMapper.selectOrderNosByGroupId(groupId);
        for (String orderNo : orderNos) {
            orderService.confirmOrder(orderNo);
        }
    }
}
```

### 2.3 定时任务处理过期

```java
// ========== 拼团过期检查定时任务 ==========
@Component
public class GroupExpireTask {
    
    @Autowired
    private GroupPurchaseMapper groupMapper;
    @Autowired
    private OrderService orderService;
    @Autowired
    private StockService stockService;
    
    /**
     * 每分钟检查一次过期拼团
     */
    @Scheduled(cron = "0 * * * * ?")
    public void processExpiredGroups() {
        // 查出所有 OPEN 且已过期的拼团
        List<PromotionGroup> expiredGroups = groupMapper.selectExpiredOpenGroups();
        
        for (PromotionGroup group : expiredGroups) {
            processExpire(group);
        }
    }
    
    @Transactional
    public void processExpire(PromotionGroup group) {
        // CAS 更新状态，防止重复处理
        int rows = groupMapper.updateStatusIfMatch(
            group.getId(), 
            GroupStatus.OPEN, 
            GroupStatus.EXPIRED
        );
        if (rows == 0) {
            return;  // 已被其他节点处理
        }
        
        // 查出所有已支付但未成团的订单
        List<Order> pendingOrders = orderMapper.selectPaidByGroupNo(group.getGroupNo());
        
        for (Order order : pendingOrders) {
            // 退款
            refundService.refund(order.getOrderNo(), "拼团超时解散");
            // 归还活动库存
            stockMapper.returnStock(group.getActivityId(), group.getSkuId(), 1);
            // 更新订单状态
            orderService.markRefunded(order.getOrderNo());
        }
    }
}
```

## 🔥 扩展追问 2：如何保证最终一致性？支付成功但拼团失败了怎么办？

> **A:** 两种方案：TCC 补偿 或 消息队列最终一致。

```java
// 方案一：TCC 补偿模式
public class GroupTccService {
    
    // Try: 锁定库存
    public boolean tryLock(Long activityId, Long skuId, int count) {
        return stockService.tryLockStock(activityId, skuId, count);
    }
    
    // Confirm: 确认扣减（只有拼团成功才调用）
    public boolean confirm(Long activityId, Long skuId, int count) {
        return stockService.confirmDeduct(skuId, count);
    }
    
    // Cancel: 回滚库存（拼团失败/过期）
    public boolean cancel(Long activityId, Long skuId, int count) {
        return stockService.releaseStock(activityId, skuId, count);
    }
}
```

## 🔥 扩展追问 5：团长取消订单后拼团还能继续吗？

> **A:** 看业务规则，常见两种模式。

| 模式 | 团长取消后 | 说明 |
|------|-----------|------|
| 团必须存在 | 团解散 | 团不能没有团长，其他人的订单全部退款 |
| 团可转让 | 转让给下一人 | 新团员补位成为团长 |

```java
// 方案一：团必须存在（简单）
public void cancelOrder(String orderNo) {
    Order order = orderMapper.selectByOrderNo(orderNo);
    
    if (order.isHead()) {
        // 团长取消 → 整个团解散
        List<Order> allOrders = orderMapper.selectByGroupNo(order.getGroupNo());
        for (Order o : allOrders) {
            refundService.refund(o.getOrderNo(), "团长取消订单，拼团解散");
        }
        groupMapper.updateStatus(groupId, GroupStatus.DISMISSED);
    } else {
        // 团员取消 → 只退自己的，团继续
        refundService.refund(orderNo, "用户取消");
        groupMapper.decrementCount(groupId);
        groupMemberMapper.delete(order.getUserId(), groupId);
    }
}

// 方案二：团可转让（复杂）
public void cancelOrder(String orderNo) {
    if (order.isHead()) {
        // 找一个已支付的团员升级为团长
        Long newHeadId = groupMemberMapper.findNextHead(groupId);
        if (newHeadId != null) {
            groupMemberMapper.promoteToHead(newHeadId, groupId);
        } else {
            // 没其他人了，解散
            dismissGroup(groupId);
        }
    }

}
```

## 一句话总结

> **拼团核心是"库存锁定 + 状态机 + 定时兜底"：用户下单时用乐观锁扣减活动库存创建待支付订单，支付成功后确认并检查是否成团，超时由定时任务补偿处理退款和库存归还，防超卖关键是让库存扣减和订单状态绑定而不是直接减 SKU 库存。**

---

**面试加分话术：**

> "之前做拼团项目时出过超卖问题：一开始用 Redis 扣库存，但 Redis 挂了导致库存数据丢失，大量超卖订单涌进来。后来改成 DB 乐观锁 `UPDATE SET stock = stock - 1 WHERE stock > 0`，每次扣库存都 CAS 更新，库存扣到 0 后后续请求直接失败。但这样性能差，我们又加了 Redis 缓存层做预扣，Redis 扣成功了才查 DB，Redis 和 DB 每分钟对账，Redis 挂了直接降级到纯 DB 模式。另外团长取消订单的处理很关键，必须看是团长的订单还是团员的，如果是团长取消原则上整个团要解散，其他人的钱要原路退回。"
