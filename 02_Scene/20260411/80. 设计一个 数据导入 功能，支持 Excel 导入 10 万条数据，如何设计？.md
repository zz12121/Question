## 三、大文件分片上传（前端）

```javascript
// 前端分片上传核心逻辑
class FileUploader {
    constructor(file, chunkSize = 5 * 1024 * 1024) { // 5MB分片
        this.file = file;
        this.chunkSize = chunkSize;
        this.chunks = Math.ceil(file.size / chunkSize);
        this.uploadedChunks = new Set();
    }
    
    async upload() {
        // 1. 询问服务器已上传的分片（断点续传）
        const uploaded = await this.checkUploadedChunks();
        this.uploadedChunks = new Set(uploaded);
        
        // 2. 并发上传未上传的分片（控制并发数）
        const tasks = [];
        for (let i = 0; i < this.chunks; i++) {
            if (!this.uploadedChunks.has(i)) {
                tasks.push(this.uploadChunk(i));
            }
        }
        
        await Promise.all(this.limitConcurrency(tasks, 3));
        
        // 3. 通知服务端合并分片
        await this.mergeChunks();
    }
    
    async uploadChunk(chunkIndex) {
        const start = chunkIndex * this.chunkSize;
        const end = Math.min(start + this.chunkSize, this.file.size);
        const chunk = this.file.slice(start, end);
        
        const formData = new FormData();
        formData.append('chunk', chunk);
        formData.append('index', chunkIndex);
        formData.append('total', this.chunks);
        formData.append('fileHash', await this.getFileHash());
        
        await axios.post('/api/import/upload-chunk', formData);
        
        // 更新进度
        this.uploadedChunks.add(chunkIndex);
        this.onProgress(this.uploadedChunks.size / this.chunks);
    }
    
    // 限制并发数
    *limitConcurrency(tasks, limit) {
        for (let i = 0; i < tasks.length; i += limit) {
            yield Promise.all(tasks.slice(i, i + limit));
        }
    }
}
```

```java
@RestController
@RequestMapping("/api/import")
public class ImportController {
    
    @Autowired private FileStorageService fileStorage;
    @Autowired private ImportTaskService taskService;
    
    // 1. 分片上传
    @PostMapping("/upload-chunk")
    public Response uploadChunk(
            @RequestParam("chunk") MultipartFile chunk,
            @RequestParam("index") int index,
            @RequestParam("total") int total,
            @RequestParam("fileHash") String fileHash) {
        
        // 保存到临时分片目录
        String chunkPath = fileStorage.getChunkPath(fileHash, index);
        fileStorage.saveChunk(chunkPath, chunk);
        
        // 检查是否所有分片都上传完成
        if (fileStorage.isChunksComplete(fileHash, total)) {
            // 触发合并
            taskService.mergeChunksAsync(fileHash, total);
        }
        
        return Response.success();
    }
    
    // 2. 创建导入任务（前端轮询查询）
    @PostMapping("/create-task")
    public Response createTask(@RequestBody CreateTaskRequest request) {
        // 校验文件是否存在

            .fileName(request.getFileName())
            .filePath(filePath)
            .fileHash(request.getFileHash())
            .userId(getCurrentUserId())
            .build());
        
        return Response.success(task.getTaskId());
    }
```

## 五、Excel 解析与批量导入

```java
@Service
public class ExcelImportService {
    
    // 批次大小配置
    private static final int BATCH_SIZE = 1000;
    private static final int MAX_RETRY = 3;
    
    @Autowired private ImportTaskRepository taskRepo;
    @Autowired private ImportBatchRepository batchRepo;
    @Autowired private ImportErrorRepository errorRepo;
    
    /**
     * 解析 Excel 并生成导入批次
     */
    public void parseAndDispatch(String taskId) {
        ImportTask task = taskRepo.findByTaskId(taskId);
        
        try {
            // 1. 使用 EasyExcel 流式读取（避免 OOM）
            long totalRows = countRows(task.getFilePath());
            task.setTotalRows((int) totalRows);
            task.setStatus(TaskStatus.PROCESSING);
            taskRepo.save(task);
            
            // 2. 按批次提交到 MQ
            int batchCount = (int) Math.ceil(totalRows * 1.0 / BATCH_SIZE);
            for (int i = 0; i < batchCount; i++) {
                int startRow = i * BATCH_SIZE + 2; // 跳过表头
                int endRow = Math.min(startRow + BATCH_SIZE - 1, (int) totalRows);
                
                ImportBatch batch = ImportBatch.builder()
                    .taskId(taskId)
                    .batchNo(i + 1)
                    .startRow(startRow)
                    .endRow(endRow)
                    .status(BatchStatus.PENDING)
                    .build();
                batchRepo.save(batch);
                
                // 发送到 MQ
                importBatchProducer.send(ImportBatchMessage.builder()
                    .taskId(taskId)
                    .batchId(batch.getId())
                    .startRow(startRow)
                    .endRow(endRow)
                    .build());
            }
            
        } catch (Exception e) {
            task.setStatus(TaskStatus.FAILED);
            task.setErrorMsg(e.getMessage());
            taskRepo.save(task);
        }
    }
    
    /**
     * 处理单个批次（消费者）
     */
    @KafkaListener(topics = "import-batches")
    @Transactional
    public void processBatch(ImportBatchMessage msg) {
        ImportBatch batch = batchRepo.findById(msg.getBatchId());
        
        try {
            // 1. 流式读取 Excel 指定行
            List<YourDataEntity> rows = readExcelRows(
                batch.getTask().getFilePath(), 
                batch.getStartRow(), 
                batch.getEndRow()
            );
            
            // 2. 数据校验
            ValidationResult validation = validateRows(rows);
            if (validation.hasErrors()) {
                // 保存错误明细
                saveErrorRecords(batch, validation.getErrors());
                rows = validation.getValidRows(); // 只保留有效数据
            }
            
            // 3. 批量插入（MySQL）
            if (!rows.isEmpty()) {
                yourDataRepository.saveAll(rows);
            }
            
            // 4. 更新批次状态
            batch.setStatus(validation.hasErrors() 
                ? BatchStatus.PARTIAL_FAIL : BatchStatus.SUCCESS);
            batch.setFailCount(validation.getErrorCount());
            batchRepo.save(batch);
            
            // 5. 更新任务汇总
            updateTaskSummary(batch.getTaskId());
            
        } catch (Exception e) {
            batch.setStatus(BatchStatus.FAILED);
            batch.setErrorMsg(e.getMessage());
            batch.setRetryCount(batch.getRetryCount() + 1);
            batchRepo.save(batch);
            
            // 重试机制
            if (batch.getRetryCount() < MAX_RETRY) {
                importBatchProducer.send(msg); // 重新入队
            }
        }
    }
    
    /**
     * EasyExcel 流式读取指定行
     */
    private List<YourDataEntity> readExcelRows(String filePath, 
            int startRow, int endRow) {
        List<YourDataEntity> rows = new ArrayList<>();
        
        EasyExcel.read(filePath, YourDataEntity.class, 
            new AnalysisEventListener<YourDataEntity>() {
                
                private int currentRow = 0;
                
                @Override
                public void invoke(YourDataEntity data, 
                        AnalysisContext context) {
                    currentRow++;
                    if (currentRow >= startRow && currentRow <= endRow) {
                        rows.add(data);
                    }
                }
                
                @Override
                public void doAfterAllAnalysed(AnalysisContext context) {
                    // 读取完成
                }
            }).sheet().doRead();
        
        return rows;
    }
}
```

## 六、数据校验逻辑

```java
@Component
public class DataValidator {
    
    @Autowired private YourDataRepository repository;
    
    public ValidationResult validateRows(List<YourDataEntity> rows) {
        ValidationResult result = new ValidationResult();
        
        // 按行校验
        for (int i = 0; i < rows.size(); i++) {
            YourDataEntity row = rows.get(i);
            List<String> errors = new ArrayList<>();
            
            // 1. 必填校验
            if (row.getName() == null || row.getName().isBlank()) {
                errors.add("姓名为必填项");
            }
            
            // 2. 格式校验
            if (row.getPhone() != null && !isValidPhone(row.getPhone())) {
                errors.add("手机号格式错误");
            }
            
            if (row.getEmail() != null && !isValidEmail(row.getEmail())) {
                errors.add("邮箱格式错误");
            }
            
            // 3. 业务校验（唯一性）
            if (repository.existsByPhone(row.getPhone())) {
                errors.add("手机号已存在");
            }
            
            // 4. 关联校验
            if (!categoryRepository.existsById(row.getCategoryId())) {
                errors.add("categoryId 不存在");
            }
            
            if (!errors.isEmpty()) {
                result.addError(row.getRowNo(), row, errors);
            }
        }
        
        // 去除重复（Excel内部重复）
        Set<String> uniqueKeys = new HashSet<>();
        rows.removeIf(row -> {
            String key = row.getPhone() + "_" + row.getIdCard();
            return !uniqueKeys.add(key);
        });
        if (uniqueKeys.size() < rows.size()) {
            result.addGlobalError("Excel中存在重复数据");
        }
        
        return result;
    }
}
```

## 🔥 扩展追问 1：如何处理 10 万行数据导入导致 OOM？

> **A:** 核心是**流式处理 + 内存隔离**：

```
┌────────────────────────────────────────────────────────────────┐
│                    内存保护策略                                  │
├────────────────────────────────────────────────────────────────┤
│                                                                │
│  1️⃣ 流式读取（不用 POI Workbook）                                │
│     ├── EasyExcel / Alibaba FastExcel（SAX 模式）               │
│     ├── 每次只读一行，不加载整表到内存                           │
│     └── 实测：100万行 Excel，内存 < 200MB                       │
│                                                                │
│  2️⃣ 分批提交 MQ                                                │
│     ├── 每批 1000 行，写入后立即释放内存                         │
│     └── 消费者也按批次处理，不累积数据                           │
│                                                                │
│  3️⃣ JVM 参数调优                                               │
│     └── -Xms512m -Xmx1g -XX:+UseG1GC                           │
│                                                                │
│  4️⃣ 文件存储分离                                               │
│     ├── 大文件存 OSS/S3，不占用应用内存                         │
│     └── 应用只读文件路径和行号范围                               │
│                                                                │
└────────────────────────────────────────────────────────────────┘
```

> **POI vs EasyExcel 内存对比**：
> - POI (默认模式)：100行 Excel → ~50MB 内存
> - EasyExcel (SAX模式)：100万行 Excel → ~100MB 内存

## 🔥 扩展追问 2：导入中途失败如何实现断点续传？

```java
// 方案1：任务级别的断点（记录已处理批次）
@Service
public class BatchResumeService {
    
    public void resumeFromLastBatch(String taskId) {
        ImportTask task = taskRepo.findByTaskId(taskId);
        
        // 查询未完成/失败的批次
        List<ImportBatch> pendingBatches = batchRepo
            .findByTaskIdAndStatusIn(taskId, 
                Arrays.asList(BatchStatus.PENDING, BatchStatus.FAILED));
        
        for (ImportBatch batch : pendingBatches) {
            importBatchProducer.send(ImportBatchMessage.builder()
                .taskId(taskId)
                .batchId(batch.getId())
                .startRow(batch.getStartRow())
                .endRow(batch.getEndRow())
                .retry(true)  // 标记重试
                .build());
        }
    }
}
```

## 🔥 扩展追问 3：如何保证导入事务一致性？

> **A:** 核心是**小事务 + 最终一致性**：

```java
@Service
public class TransactionalBatchImport {
    
    // 每批次独立事务，失败只影响当前批次
    @Transactional(isolation = Isolation.READ_COMMITTED)
    public BatchResult importBatch(List<YourDataEntity> rows) {
        BatchResult result = new BatchResult();
        
        // 1. 数据校验
        ValidationResult validation = validate(rows);
        rows = validation.getValidRows();
        
        // 2. 批量插入（MySQL）
        try {
            yourDataRepository.saveAll(rows);
            result.setSuccessCount(rows.size());
        } catch (DuplicateKeyException e) {
            // 唯一键冲突，单条重试（跳过冲突）
            result = handleDuplicateKey(rows);
        }
        
        // 3. 保存错误
        errorRepo.saveAll(validation.getErrors());
        
        return result;
    }
    
    // 全局一致性保证（可选，额外开销大）
    @GlobalTransactional
    public void importWithGlobalTx(String taskId, List<YourDataEntity> rows) {
        // 场景：导入的数据必须与现有数据强一致
        // 代价：性能下降 30-50%，谨慎使用
        importBatch(rows);
    }
}
```

> **不建议**：整个导入使用一个大事务，因为 10 万条数据会：
> - 锁住大量行，导致并发插入阻塞
> - 事务日志膨胀（Undo/Redo Log）
> - 回滚极慢

## 🔥 扩展追问 4：如何防止用户重复导入同一文件？

> **A:** 文件 hash + 任务状态检查：

```java
@Service
public class DuplicateImportDetector {
    
    public void checkDuplicate(Long userId, String fileHash, String fileName) {
        // 1. 检查是否存在相同文件的进行中任务
        ImportTask existing = taskRepo.findByUserAndFileHash(userId, fileHash)
            .filter(t -> t.getStatus() == TaskStatus.PROCESSING)
            .orElse(null);
        
        if (existing != null) {
            throw new BusinessException("该文件正在导入中，请勿重复提交");
        }
        
        // 2. 检查历史是否成功导入过
        long count = taskRepo.countByUserAndFileHashAndSuccess(userId, fileHash);
        if (count > 0) {
            throw new BusinessException("该文件已成功导入过");
        }
    }
}
```
