### 常见死锁场景代码

```java
// ========== 场景1：synchronized 嵌套导致的 AB-BA 死锁 ==========

public class OrderService {
    
    // 对象锁
    private final Object lockA = new Object();
    private final Object lockB = new Object();
    
    // 方法1：先锁A再锁B
    public void createOrder(Long orderId) {
        synchronized (lockA) {           // 持有 lockA
            try {
                Thread.sleep(100);       // 模拟业务处理
            } catch (InterruptedException e) {}
            
            // ⚠️ 在持有一个锁时调用另一个方法，该方法也会加锁
            updateInventory(orderId);    // 里面会锁 lockB → 可能死锁
        }
    }
    
    // 方法2：先锁B再锁A（不同线程调用顺序相反）
    public void cancelOrder(Long orderId) {
        synchronized (lockB) {           // 持有 lockB
            try {
                Thread.sleep(100);
            } catch (InterruptedException e) {}
            
            // ⚠️ 如果线程1执行 createOrder 时，线程2执行 cancelOrder
            // 线程1等 lockB，线程2等 lockA → 死锁
            refreshOrder(orderId);       // 里面会锁 lockA
        }
    }
    
    private void updateInventory(Long orderId) {
        synchronized (lockB) {           // 锁 B
            inventoryMapper.decrease(orderId);
        }
    }
    
    private void refreshOrder(Long orderId) {
        synchronized (lockA) {           // 锁 A
            orderMapper.refresh(orderId);
        }
    }
}


// ========== 场景2：ReentrantLock 与 synchronized 混用 ==========

public class PaymentService {
    
    private final ReentrantLock lockA = new ReentrantLock();
    private final ReentrantLock lockB = new ReentrantLock();
    
    // ⚠️ 使用 lock() + unlock()，不遵守 try-finally 规范
    public void deductBalance(Long userId, BigDecimal amount) {
        lockA.lock();                    // 持 lockA
        
        if (checkRisk(userId)) {          // 调用内部方法
            lockB.lock();                // 持 lockB
            accountMapper.deduct(userId, amount);
            lockB.unlock();
        }
        
        // ⚠️ 如果 checkRisk 抛异常，lockA 永远不释放 → 死锁
        lockA.unlock();
    }
    
    public void refund(Long userId, BigDecimal amount) {
        lockB.lock();                    // 持 lockB
        
        // ⚠️ 与上方对称调用
        refreshAccount(userId);          // 里会调 deductBalance
        lockB.unlock();
    }
}


// ========== 场景3：数据库行锁死锁 ==========

@Service
public class TransferService {
    
    @Autowired
    private JdbcTemplate jdbcTemplate;
    
    /**
     * 转账：双向锁顺序不一致导致死锁
     */
    @Transactional
    public void transfer(Long fromUserId, Long toUserId, BigDecimal amount) {
        // ⚠️ A用户转B：先锁A再锁B
        // UPDATE user_balance SET amount = amount - 100 WHERE id = 1;  (锁A)
        jdbcTemplate.update(
            "UPDATE user_balance SET amount = amount - ? WHERE id = ?",
            amount, fromUserId);
        
        // 模拟网络延迟（增大死锁窗口）
        try { Thread.sleep(100); } catch (Exception e) {}
        
        // UPDATE user_balance SET amount = amount + 100 WHERE id = 2;  (锁B)
        jdbcTemplate.update(
            "UPDATE user_balance SET amount = amount + ? WHERE id = ?",
            amount, toUserId);
        
        // 问题：如果同时发生 A→B 和 B→A：
        // T1: 锁 user:1，等 user:2
        // T2: 锁 user:2，等 user:1
        // → MySQL InnoDB 行锁死锁检测 → T2 被回滚
    }
}
```

### 死锁代码修复

```java
// ========== 修复1：固定加锁顺序 ==========

public class OrderServiceFixed {
    
    // 所有需要多锁的地方，按 hash 顺序加锁
    public void createOrder(Long orderId, Long skuId) {
        // 固定顺序：hash 小的先锁
        Long first = orderId < skuId ? orderId : skuId;
        Long second = orderId < skuId ? skuId : orderId;
        
        if (orderId < skuId) {
            synchronized (getLock(first)) {
                synchronized (getLock(second)) {
                    doCreateOrder(orderId, skuId);
                }
            }
        } else {
            synchronized (getLock(first)) {
                synchronized (getLock(second)) {
                    doCreateOrder(orderId, skuId);
                }
            }
        }
    }
    
    // 统一获取锁对象（按 ID hash 到固定对象）
    private Object getLock(Long id) {
        return lockPool.get(id % lockPool.size());
    }
    
    // 或者更简单的方案：单对象锁（牺牲并发，但彻底避免死锁）
    private final Object globalLock = new Object();
    
    public void createOrderSimple(Long orderId) {
        synchronized (globalLock) {
            doCreateOrder(orderId);
        }
    }
}


// ========== 修复2：ReentrantLock tryLock + 超时 ==========

public class PaymentServiceFixed {
    
    private final ReentrantLock lockA = new ReentrantLock();
    private final ReentrantLock lockB = new ReentrantLock();
    
    private static final long TRY_LOCK_TIMEOUT_MS = 3000;
    
    public boolean transfer(Long fromUserId, Long toUserId, BigDecimal amount) {
        // 尝试获取第一个锁
        if (!lockA.tryLock(TRY_LOCK_TIMEOUT_MS, TimeUnit.MILLISECONDS)) {
            log.warn("获取锁A超时: userId={}", fromUserId);
            return false;
        }
        
        try {
            // 尝试获取第二个锁
            if (!lockB.tryLock(TRY_LOCK_TIMEOUT_MS, TimeUnit.MILLISECONDS)) {
                log.warn("获取锁B超时: userId={}", fromUserId);
                // 这里不会死锁，但需要决定：回滚还是重试
                return false;
            }
            
            try {
                // 执行业务
                doTransfer(fromUserId, toUserId, amount);
                return true;
            } finally {
                lockB.unlock();
            }
        } finally {
            lockA.unlock();
        }
    }
}


// ========== 修复3：数据库行锁按 ID 顺序 ==========

@Service
public class TransferServiceFixed {
    
    @Transactional
    public void transfer(Long fromUserId, Long toUserId, BigDecimal amount) {
        // ⚠️ 关键：始终先锁 ID 小的，再锁 ID 大的
        Long firstId = fromUserId < toUserId ? fromUserId : toUserId;
        Long secondId = fromUserId < toUserId ? toUserId : fromUserId;
        
        // 模拟业务逻辑，实际按 ID 顺序加锁
        if (fromUserId < toUserId) {
            jdbcTemplate.update(
                "UPDATE user_balance SET amount = amount - ? WHERE id = ?",
                amount, fromUserId);
            jdbcTemplate.update(
                "UPDATE user_balance SET amount = amount + ? WHERE id = ?",
                amount, toUserId);
        } else {
            jdbcTemplate.update(
                "UPDATE user_balance SET amount = amount + ? WHERE id = ?",
                amount, toUserId);
            jdbcTemplate.update(
                "UPDATE user_balance SET amount = amount - ? WHERE id = ?",
                amount, fromUserId);
        }
        // 这样 A→B 和 B→A 都会先锁 1 再锁 2，不会死锁
    }
}
```

## 🔥 扩展追问 1：jstack 显示死锁但应用没挂，怎么优雅重启？

> **A:** 不能直接 kill -9（会丢数据），要分步处理：

```bash
# ========== 紧急处理流程 ==========

# 1. 先导出完整线程栈和堆（保留现场）
jstack <java_pid> > /tmp/thread_dump_$(date +%s).log
jmap -dump:format=b,file=/tmp/heap_$(date +%s).hprof <java_pid>

# 2. 先摘流量（不要直接重启）
# K8s: kubectl scale deployment xxx --replicas=0
# Nginx: upstream 摘掉这台机器

# 3. 等待现有请求处理完（最多等 30s）
sleep 30

# 4. 如果线程还没释放（死锁无法自动解除），发 SIGTERM 让 JVM 优雅关闭
kill -15 <java_pid>

# 5. JVM 收到 SIGTERM，会等所有请求处理完，但死锁线程不会结束
# 所以可能需要再等一段时间，如果还不退出发 SIGKILL
kill -9 <java_pid>   # 最后手段

# 6. 重启应用
# K8s: kubectl scale deployment xxx --replicas=3
```

```java
// JVM 优雅关闭钩子（添加业务需要的清理逻辑）
public class ShutdownHandler {
    
    @PreDestroy
    public void onShutdown() {
        log.info("收到关闭信号，开始清理...");
        
        // 1. 停止接收新请求
        try {
            Thread.sleep(5000);  // 等 5s 处理存量
        } catch (InterruptedException e) {}
        
        // 2. 强制 interrupt 死锁线程（自救，但可能丢数据）
        Thread.getAllStackTraces().keySet().forEach(thread -> {
            if (thread.getName().contains("pool")) {
                thread.interrupt();  // 死锁线程如果等 LockSupport.park，会被唤醒
            }
        });
        
        // 3. 关闭数据库连接（释放行锁）
        // HikariCP 会自动关闭
    }
}
```

## 🔥 扩展追问 3：活锁（LiveLock）和死锁有什么区别？

> **A:** 死锁是"不动"，活锁是"动但没进展"：

```
┌─────────────────────────────────────────────────────────────────┐
│              死锁 vs 活锁 vs 饥饿                                │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│   死锁 (Deadlock)                                               │
│   ┌─────────────────────────────────────────────────────────┐   │
│   │ 线程1: 持有A ──▶ 等B ──▶ (永远等待)                    │   │
│   │ 线程2: 持有B ──▶ 等A ──▶ (永远等待)                    │   │
│   │  → 两个线程都不动，CPU 空闲                             │   │
│   └─────────────────────────────────────────────────────────┘   │
│                                                                 │
│   活锁 (LiveLock)                                               │
│   ┌─────────────────────────────────────────────────────────┐   │
│   │ 线程1: 尝试获取A ──▶ A被占用 ──▶ 退让                   │   │
│   │ 线程2: 尝试获取B ──▶ B被占用 ──▶ 退让                   │   │
│   │ (重复尝试但永远碰不上)                                   │   │
│   │ → 两个线程都在动，CPU 飙升，但业务没进展                  │   │
│   └─────────────────────────────────────────────────────────┘   │
│                                                                 │
│   饥饿 (Starvation)                                             │
│   ┌─────────────────────────────────────────────────────────┐   │
│   │ 线程1: 高优先级，一直占用资源                           │   │
│   │ 线程2: 低优先级，永远得不到资源                         │   │
│   │ → 低优先级线程饿死                                       │   │
│   └─────────────────────────────────────────────────────────┘   │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

```java
// 活锁示例：两个线程互相让路，但永远碰不上
public class LiveLockExample {
    
    public static void main(String[] args) {
        TransferTask task1 = new TransferTask("A->B", () -> true);
        TransferTask task2 = new TransferTask("B->A", () -> true);
        
        new Thread(task1).start();
        new Thread(task2).start();
    }
    
    static class TransferTask implements Runnable {
        private String name;
        private Supplier<Boolean> conflictCheck;
        
        public void run() {
            while (true) {
                // 尝试转账
                boolean success = tryTransfer();
                
                if (success) {
                    System.out.println(name + " 成功!");
                    break;
                } else {
                    // ⚠️ 活锁：遇到冲突就退让重试
                    // 双方同时退让 → 永远碰不上
                    System.out.println(name + " 冲突，短暂等待后重试...");
                    try { Thread.sleep(10); } catch (Exception e) {}
                }
            }
        }
        
        private boolean tryTransfer() {
            // 模拟检查
            return conflictCheck.get();
        }
    }
}

// 活锁修复：随机退避 + 有限重试
public class LiveLockFix {
    
    private final Random random = new Random();
    private static final int MAX_RETRY = 100;
    
    public void transferWithBackoff() {
        for (int i = 0; i < MAX_RETRY; i++) {
            if (tryTransfer()) {
                return;
            }
            
            // ⚠️ 关键：随机退避，打破同步
            long delay = random.nextInt(100);  // 0-100ms 随机
            try {
                Thread.sleep(delay);
            } catch (Exception e) {}
        }
        
        throw new BusinessException("转账冲突次数过多，请稍后重试");
    }
}
```

## 🔥 扩展追问 4：InnoDB 行锁死锁怎么排查？

> **A:** MySQL 有死锁检测机制，但排查日志很关键：

```sql
-- ========== 查看最近死锁日志 ==========

SHOW ENGINE INNODB STATUS;

-- 输出中的死锁信息：
/*
LATEST DETECTED DEADLOCK
------------------------
2024-04-14 10:30:00 7f8a5c01a000

Transaction 1:
    UPDATE user_balance SET amount = amount - 100 WHERE id = 1
    UPDATE user_balance SET amount = amount + 100 WHERE id = 2

Transaction 2:
    UPDATE user_balance SET amount = amount + 100 WHERE id = 2
    UPDATE user_balance SET amount = amount - 100 WHERE id = 1

    --- BREAKING THIS TRANSACTION (回滚 Transaction 2)
*/
```

```sql
-- ========== 查看锁等待信息 ==========

SELECT 
    r.trx_id,
    r.trx_mysql_thread_id,
    r.trx_state,
    r.trx_started,
    r.trx_rows_locked,
    r.trx_query,
    l.lock_id,
    l.lock_mode,
    l.lock_type,
    l.lock_table
FROM information_schema.INNODB_TRX r
JOIN information_schema.INNODB_LOCKS l 
    ON r.trx_id = l.lock_trx_id
WHERE r.trx_state = 'LOCK WAIT';
```

```java
// 业务层减少 InnoDB 行锁持有时间
@Service
public class OptimizedTransferService {
    
    @Transactional(isolation = Isolation.READ_COMMITTED)
    public void transfer(Long fromUserId, Long toUserId, BigDecimal amount) {
        // ⚠️ 关键：减少锁持有时间
        
        // 1. 先校验（不加锁）
        if (!verifyBalance(fromUserId, amount)) {
            throw new BusinessException("余额不足");
        }
        
        // 2. 乐观锁更新（不加表锁）
        int affected = jdbcTemplate.update(
            "UPDATE user_balance SET version = version + 1, " +
            "amount = amount - ? WHERE id = ? AND version = ?",
            amount, fromUserId, currentVersion);
        
        if (affected == 0) {
            throw new BusinessException("并发更新，请重试");
        }
        
        // 3. 更新目标（单独事务，不合并）
        jdbcTemplate.update(
            "UPDATE user_balance SET amount = amount + ? WHERE id = ?",
            amount, toUserId);
        
        // 4. 发送消息（异步，不在事务内）
        eventPublisher.publish(new TransferEvent(fromUserId, toUserId, amount));
    }
}
```

>
> **核心是：死锁四条件（互斥、占有等待、不可抢占、循环等待）破坏任一即可。线上先保留现场再重启，代码层面固定加锁顺序最有效。**
