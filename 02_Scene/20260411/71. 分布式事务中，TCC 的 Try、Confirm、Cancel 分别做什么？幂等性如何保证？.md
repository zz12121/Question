### 订单支付 + 库存扣减 + 积分发放 实例

```java
// 1. TCC 接口定义
@LocalTCC
public interface OrderTccService {
    
    /**
     * Try: 预留资源（冻结库存/冻结余额）
     * @param context 事务上下文，传递调用参数
     */
    @TwoPhaseBusinessAction(name = "prepareOrder", 
                            commitMethod = "confirm", 
                            rollbackMethod = "cancel")
    boolean tryPrepareOrder(BusinessActionContext context,
                            @BusinessActionContextParameter(paramName = "orderId") Long orderId,
                            @BusinessActionContextParameter(paramName = "userId") Long userId,
                            @BusinessActionContextParameter(paramName = "skuId") Long skuId,
                            @BusinessActionContextParameter(paramName = "count") Integer count,
                            @BusinessActionContextParameter(paramName = "points") Integer points);
    
    /**
     * Confirm: 确认执行（真正扣减/发放）
     */
    boolean confirm(BusinessActionContext context);
    
    /**
     * Cancel: 回滚（释放冻结）
     */
    boolean cancel(BusinessActionContext context);
}

// 2. 库存服务 TCC 实现
@Service
public class InventoryTccServiceImpl implements OrderTccService {
    
    @Autowired
    private InventoryMapper inventoryMapper;
    @Autowired
    private TccLogMapper tccLogMapper;
    
    @Override
    @Transactional
    public boolean tryPrepareOrder(BusinessActionContext context,
                                   Long orderId, Long userId, Long skuId,
                                   Integer count, Integer points) {
        log.info("=== Try 阶段: 预留库存, orderId={}, skuId={}, count={}", 
                 orderId, skuId, count);
        
        // 检查库存是否足够
        Inventory inventory = inventoryMapper.selectBySkuId(skuId);
        if (inventory.getStock() < count) {
            throw new BusinessException("库存不足");
        }
        
        // ⚠️ 核心1: 冻结库存（不减可用库存，加 frozen_stock）
        int frozen = inventoryMapper.freezeStock(skuId, count);
        if (frozen == 0) {
            throw new BusinessException("冻结库存失败，并发冲突");
        }
        
        // ⚠️ 核心2: 记录 TCC 日志（幂等保证）
        TccLog log = TccLog.builder()
            .xid(context.getXid())
            .actionId(context.getActionId())
            .status(TccStatus.TRYING)
            .orderId(orderId)
            .actionType("FREEZE_STOCK")
            .gmtCreate(new Date())
            .build();
        tccLogMapper.insert(log);
        
        // ⚠️ 核心3: 记录可补偿操作（用于 Cancel 时知道要恢复什么）
        TccCompensation compensation = TccCompensation.builder()
            .xid(context.getXid())
            .actionId(context.getActionId())
            .compensationType("UNFREEZE_STOCK")
            .skuId(skuId)
            .frozenCount(count)
            .build();
        compensationMapper.insert(compensation);
        
        return true;
    }
    
    @Override
    public boolean confirm(BusinessActionContext context) {
        log.info("=== Confirm 阶段: 确认扣减库存, xid={}", context.getXid());
        
        // ⚠️ 幂等检查：Confirm 可能被重复调用
        TccLog existingLog = tccLogMapper.selectByActionId(context.getActionId());
        if (existingLog != null && existingLog.getStatus() == TccStatus.CONFIRMED) {
            log.info("Confirm 已执行过，跳过, actionId={}", context.getActionId());
            return true;
        }
        
        // 根据 actionId 查找对应的补偿记录
        TccCompensation comp = compensationMapper.selectByActionId(context.getActionId());
        if (comp == null) {
            log.warn("未找到补偿记录，可能 Try 失败后直接 Confirm，忽略");
            return true;
        }
        
        // 真正扣减可用库存
        inventoryMapper.decreaseStock(comp.getSkuId(), comp.getFrozenCount());
        
        // 删除冻结记录
        inventoryMapper.removeFrozenStock(comp.getSkuId(), comp.getFrozenCount());
        
        // 更新 TCC 日志状态
        tccLogMapper.updateStatusByActionId(context.getActionId(), TccStatus.CONFIRMED);
        
        return true;
    }
    
    @Override
    public boolean cancel(BusinessActionContext context) {
        log.info("=== Cancel 阶段: 回滚冻结库存, xid={}", context.getXid());
        
        // ⚠️ 幂等检查
        TccLog existingLog = tccLogMapper.selectByActionId(context.getActionId());
        if (existingLog != null && existingLog.getStatus() == TccStatus.CANCELED) {
            log.info("Cancel 已执行过，跳过, actionId={}", context.getActionId());
            return true;
        }
        
        // 回滚：恢复冻结的库存
        TccCompensation comp = compensationMapper.selectByActionId(context.getActionId());
        if (comp == null) {
            log.warn("未找到补偿记录，可能 Try 未执行或已 Confirm，忽略");
            return true;
        }
        
        inventoryMapper.unfreezeStock(comp.getSkuId(), comp.getFrozenCount());
        
        // 更新 TCC 日志状态
        tccLogMapper.updateStatusByActionId(context.getActionId(), TccStatus.CANCELED);
        
        return true;
    }
}

// 3. 余额服务 TCC 实现（类似结构）
@Service
public class AccountTccServiceImpl implements OrderTccService {
    
    @Override
    public boolean tryPrepareOrder(BusinessActionContext context,
                                   Long orderId, Long userId, Long skuId,
                                   Integer count, Integer points) {
        log.info("=== Try 阶段: 冻结余额, userId={}, amount={}", userId, points);
        
        // 检查并冻结用户余额
        int frozen = accountMapper.freezeBalance(userId, points);
        if (frozen == 0) {
            throw new BusinessException("余额不足");
        }
        
        // 记录 TCC 日志
        saveTccLog(context, orderId, userId, points, TccStatus.TRYING);
        
        return true;
    }
    
    @Override
    public boolean confirm(BusinessActionContext context) {
        log.info("=== Confirm 阶段: 确认扣减余额");
        
        // 幂等检查
        if (isAlreadyConfirmed(context.getActionId())) {
            return true;
        }
        
        // 真正扣减余额
        TccCompensation comp = getCompensation(context.getActionId());
        accountMapper.decreaseBalance(comp.getUserId(), comp.getFrozenAmount());
        
        // 更新日志
        updateStatus(context.getActionId(), TccStatus.CONFIRMED);
        return true;
    }
    
    @Override
    public boolean cancel(BusinessActionContext context) {
        log.info("=== Cancel 阶段: 回滚冻结余额");
        
        // 幂等检查
        if (isAlreadyCanceled(context.getActionId())) {
            return true;
        }
        
        // 恢复冻结余额
        TccCompensation comp = getCompensation(context.getActionId());
        accountMapper.unfreezeBalance(comp.getUserId(), comp.getFrozenAmount());
        
        // 更新日志
        updateStatus(context.getActionId(), TccStatus.CANCELED);
        return true;
    }
}
```

```sql
-- TCC 操作日志表（幂等保证核心）
CREATE TABLE tcc_operation_log (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    xid VARCHAR(64) NOT NULL COMMENT '全局事务ID',
    action_id VARCHAR(64) NOT NULL COMMENT '分支action唯一ID',

    status TINYINT NOT NULL COMMENT '0-Trying, 1-Confirmed, 2-Canceled',

    UNIQUE KEY uk_xid_action (xid, action_id),

    gmt_create DATETIME NOT NULL,
    gmt_modified DATETIME,
    UNIQUE KEY uk_xid_action (xid, action_id),
    KEY idx_rollback_status (rollback_status)
) ENGINE=InnoDB COMMENT='TCC补偿记录';
```

## 🔥 扩展追问 1：TCC 和 Seata AT 模式有什么区别？

> **A:** 核心区别在于：Seata AT 是**无侵入 + 依赖全局锁**，TCC 是**纯业务补偿 + 无全局锁**。

| 维度 | TCC | Seata AT |
|------|-----|----------|
| **代码侵入** | 高（需写 Try/Confirm/Cancel） | 低（自动解析 SQL 生成回滚语句） |
| **性能损耗** | 低（无全局锁竞争） | 中等（行级锁，RC 隔离级别） |
| **一致性模型** | 最终一致 | 强一致（全局锁） |
| **资源预留** | ✅ 需要 Try 预留 | ❌ 不需要 |
| **跨语言支持** | ✅ 纯接口 RPC 调用 | ❌ 依赖 TC 事务协调器 |
| **故障恢复** | 补偿记录本地存储 | Undo Log 自动回滚 |
| **适用场景** | 冻结资源、第三方支付 | 同库跨表、跨库 |

```
┌─────────────────────────────────────────────────────────────────┐
│         Seata AT 模式 vs TCC 模式对比图                          │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│   Seata AT 模式（两阶段 + 自动补偿）                             │
│                                                                 │
│   Phase 1 (Prepare):                                            │
│   ┌─────────────────────────────────────────────────────────┐  │
│   │ TM: 全局事务开启                                        │  │
│   │                                                         │  │
│   │ RM: 执行业务 SQL，生成 Undo Log (记录修改前后镜像)        │  │
│   │      申请行级锁（本地锁，非全局锁）                        │  │
│   │      注册分支事务到 TC                                    │  │
│   └─────────────────────────────────────────────────────────┘  │
│                                                                 │
│   Phase 2 (Commit/Rollback):                                     │
│   ┌─────────────────────────────────────────────────────────┐  │
│   │ TC: 通知所有 RM 提交/回滚                               │  │
│   │                                                         │  │
│   │ 提交: 异步批量删除 Undo Log                             │  │
│   │ 回滚: 读取 Undo Log，执行反向 SQL                        │  │
│   │      DELETE → INSERT（镜像还原）                        │  │
│   │      UPDATE → UPDATE（镜像还原）                        │  │
│   └─────────────────────────────────────────────────────────┘  │
│                                                                 │
│   TCC 模式（三阶段 + 手动补偿）                                  │
│                                                                 │
│   Phase 1 (Try):                                                │
│   ┌─────────────────────────────────────────────────────────┐  │
│   │ TM: 全局事务开启                                        │  │
│   │                                                         │  │
│   │ RM: 冻结库存 / 冻结余额 / 预留资源                       │  │
│   │      不影响正常可用库存                                   │  │
│   └─────────────────────────────────────────────────────────┘  │
│                                                                 │
│   Phase 2 (Confirm):                                            │
│   ┌─────────────────────────────────────────────────────────┐  │
│   │ RM: 执行真正扣减 / 发放                                  │  │
│   │      释放冻结资源                                        │  │
│   └─────────────────────────────────────────────────────────┘  │
│                                                                 │
│   Phase 3 (Cancel - 仅失败时):                                   │
│   ┌─────────────────────────────────────────────────────────┐  │
│   │ RM: 回滚冻结资源                                        │  │
│   │      恢复库存 / 恢复余额                                  │  │
│   └─────────────────────────────────────────────────────────┘  │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

**实战选择建议：**
- **余额类资金操作** → TCC（必须显式冻结，防止超扣）
- **库存扣减** → TCC（需要冻结，防止超卖）
- **普通订单创建** → Seata AT（无资源预留，TCM 自动处理）
- **跨语言微服务** → TCC（Seata AT 仅支持 Java）

## 🔥 扩展追问 2：TCC 空回滚怎么解决？

> **A:** 空回滚是指 Try 根本没执行或执行失败，但 Cancel 被调用了。原因是：**TM 发送 Cancel 请求时，RM 网络超时/抖动导致 Confirm/Cancel 超时，TC 认为失败又重试 Cancel**。

```java
// 空回滚防御策略：在 Cancel 中检查 Try 是否真正执行过
@Override
public boolean cancel(BusinessActionContext context) {
    log.info("=== Cancel 阶段: 检查 Try 是否执行过, xid={}", context.getXid());
    
    // 1. 检查 TCC 日志是否存在
    TccLog tryLog = tccLogMapper.selectByXidAndActionId(
        context.getXid(), context.getActionId());
    
    if (tryLog == null) {
        // Try 根本没执行 → 空回滚，记录告警但不抛异常
        log.warn("⚠️ 空回滚检测! xid={}, actionId={}，Try 未执行，直接返回成功", 
            context.getXid(), context.getActionId());
        
        // 记录空回滚日志（便于排查）
        TccLog emptyRollbackLog = TccLog.builder()
            .xid(context.getXid())
            .actionId(context.getActionId())
            .status(TccStatus.EMPTY_ROLLBACK)  // 特殊状态
            .gmtCreate(new Date())
            .build();
        tccLogMapper.insert(emptyRollbackLog);
        
        // ⚠️ 关键：空回滚不能抛异常，否则 TC 会不断重试
        return true;
    }
    
    // 2. Try 已执行，正常回滚
    if (tryLog.getStatus() == TccStatus.CANCELED) {
        log.info("Cancel 已执行过，跳过幂等, actionId={}", context.getActionId());
        return true;
    }
    
    // 执行业务回滚逻辑...
    return doRollback(context);
}
```

## 🔥 扩展追问 3：TCC 悬挂问题怎么解决？

> **A:** 悬挂是指：**Cancel 比 Try 先执行了**（网络原因导致 Cancel 消息先到达），或者 Try 失败但 Cancel 被执行了。

```
┌─────────────────────────────────────────────────────────────────┐
│                    TCC 悬挂问题场景                               │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  正常流程:                                                      │
│  Try ──▶ Confirm                                               │
│                                                                 │
│  悬挂流程:                                                      │
│  Try 发送失败（网络超时）──▶ TC 超时 ──▶ 触发 Cancel            │
│                                    │                            │
│                                    ▼                            │
│                               Cancel 执行                        │
│                               但 Try 根本没成功                   │
│                                    │                            │
│                                    ▼                            │
│                               数据状态异常                       │
│                               （凭空扣减了冻结？）               │
│                                                                 │
│  另一种情况:                                                    │
│  Try 超时（但实际执行成功了）──▶ 重试 Try ──▶ Try 执行两次        │
│                                          │                       │
│                                          ▼                       │
│                                     Confirm 执行                 │
│                                     但 Cancel 可能也在执行       │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

```java
// 悬挂防御策略：Confirm 和 Cancel 都要检查 Try 是否真正成功
@Override
public boolean confirm(BusinessActionContext context) {
    log.info("=== Confirm 阶段: 防悬挂检查, xid={}", context.getXid());
    
    TccLog tryLog = tccLogMapper.selectByXidAndActionId(
        context.getXid(), context.getActionId());
    
    // 情况1: Try 根本没执行 → Confirm 无意义，直接返回成功
    if (tryLog == null) {
        log.warn("⚠️ 悬挂检测: Confirm 但 Try 未执行，跳过");
        return true;
    }
    
    // 情况2: Try 已取消 → 不应再 Confirm
    if (tryLog.getStatus() == TccStatus.CANCELED) {
        log.error("❌ 异常: Confirm 但 Try 已 Cancel，疑似悬挂，数据不一致!");
        // 记录告警，人工介入
        alertAndPause(context.getXid());
        throw new IllegalStateException("事务状态异常，Confirm 被悬挂");
    }
    
    // 正常 Confirm 逻辑
    return doConfirm(context);
}

@Override
public boolean cancel(BusinessActionContext context) {
    TccLog tryLog = tccLogMapper.selectByXidAndActionId(
        context.getXid(), context.getActionId());
    
    // Try 未执行或 Try 已 Confirm → Cancel 无意义
    if (tryLog == null || tryLog.getStatus() == TccStatus.CONFIRMED) {
        log.warn("⚠️ 悬挂检测: Cancel 但 Try 未执行或已 Confirm，跳过");
        return true;
    }
    
    if (tryLog.getStatus() == TccStatus.CANCELED) {
        return true;  // 幂等
    }
    
    return doRollback(context);
}
```

## 🔥 扩展追问 5：TCC 和 Saga 模式怎么选？

> **A:** TCC 适合**短事务 + 资源预留**，Saga 适合**长流程 + 无资源预留**：

| 维度 | TCC | Saga |
|------|-----|------|
| **资源预留** | ✅ Try 预留 | ❌ 无预留，直接执行 |
| **回滚方式** | 补偿（冻结→恢复） | 逆操作（正补偿） |
| **子事务数量** | 建议 ≤ 5 | 无限制（支持长链路） |
| **适用场景** | 库存、余额、优惠券 | 订单→支付→物流→评价 |
| **故障恢复** | TCC Log | Saga Log |
| **隔离性** | 可做到资源隔离 | 最终一致，隔离性较弱 |

```
Saga 模式示意（无资源预留，长链路）：

订单创建 ──▶ 库存扣减 ──▶ 支付扣款 ──▶ 发货通知 ──▶ 积分发放
    │            │            │            │            │
    ▼            ▼            ▼            ▼            ▼
  创建成功    扣减成功     扣款成功     发送成功     发放成功
    │            │            │            │            │
    │◀───────────│◀────────────│◀────────────│◀───────────│
    │            │            │            │            │
    ▼            ▼            ▼            ▼            ▼
  取消订单   库存恢复    余额恢复    取消发货    扣减积分

每个步骤都有对应的"撤销步骤"，失败时逆序执行
```

> **总结：TCC 三阶段 → Try 预留/冻结 → Confirm 确认 → Cancel 回滚**
>
> **幂等保证核心：TCC 日志表记录状态 + 每次操作前查状态 + 三种异常防御（空回滚/悬挂/重复调用）。**
