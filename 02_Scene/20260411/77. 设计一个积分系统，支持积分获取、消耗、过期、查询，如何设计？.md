```java
// ========== 1. 积分服务门面 ==========

@Service
public class PointService {
    
    @Autowired
    private PointEarnService earnService;
    @Autowired
    private PointSpendService spendService;
    @Autowired
    private PointQueryService queryService;
    
    /**
     * 获取积分
     * bizType: ORDER/SIGN/GIFT
     * bizId: 关联业务ID（订单号等）
     */
    @Transactional(rollbackFor = Exception.class)
    public PointResult earnPoints(Long userId, String bizType, String bizId, 
                                  Integer points, Map<String, Object> extData) {
        if (points == null || points <= 0) {
            return PointResult.fail("积分必须大于0");
        }
        
        // 校验规则
        PointRule rule = getPointRule(bizType);
        if (rule == null || rule.getStatus() == 0) {
            return PointResult.fail("该业务类型不支持积分获取");
        }
        
        // 计算实际积分
        int actualPoints = calculatePoints(points, rule);
        
        // 执行获取
        return earnService.earn(userId, bizType, bizId, actualPoints, rule);
    }
    
    /**
     * 消费积分（订单抵扣/兑换）
     */
    @Transactional(rollbackFor = Exception.class)
    public PointResult spendPoints(Long userId, String bizType, String bizId,
                                   Integer points, String remark) {
        if (points == null || points <= 0) {
            return PointResult.fail("积分必须大于0");
        }
        
        return spendService.spend(userId, bizType, bizId, points, remark);
    }
    
    /**
     * 查询积分余额
     */
    public PointAccount getAccount(Long userId) {
        return queryService.getAccount(userId);
    }
    
    /**
     * 查询积分明细
     */
    public Page<PointTransaction> getTransactions(Long userId, Integer page, 
                                                  Integer size, Integer type) {
        return queryService.getTransactions(userId, page, size, type);
    }
}

// ========== 2. 积分获取服务 ==========

@Service
public class PointEarnService {
    
    @Autowired
    private PointAccountMapper accountMapper;
    @Autowired
    private PointTransactionMapper transactionMapper;
    @Autowired
    private PointRuleMapper ruleMapper;
    
    /**
     * 积分获取（乐观锁保证一致性）
     */
    public PointResult earn(Long userId, String bizType, String bizId,
                           Integer points, PointRule rule) {
        
        // 1. 幂等检查：同一业务ID不重复发放
        if (StringUtils.isNotBlank(bizId)) {
            PointTransaction existTx = transactionMapper
                .selectByBizTypeAndBizId(bizType, bizId);
            if (existTx != null) {
                log.info("积分已发放，幂等跳过: bizType={}, bizId={}", bizType, bizId);
                return PointResult.success("重复发放");
            }
        }
        
        // 2. 获取或创建账户
        PointAccount account = getOrCreateAccount(userId);
        
        // 3. 乐观锁更新余额
        int updated = accountMapper.increaseBalance(account.getId(), 
            points, account.getVersion());
        
        if (updated == 0) {
            // 并发冲突，重试
            return retryEarn(userId, bizType, bizId, points, rule, 3);
        }
        
        // 4. 记录流水
        PointTransaction tx = PointTransaction.builder()
            .userId(userId)
            .accountId(account.getId())
            .type(PointType.EARN.getCode())
            .changeAmount(points)
            .balanceBefore(account.getBalance())
            .balanceAfter(account.getBalance() + points)
            .bizType(bizType)
            .bizId(bizId)
            .pointSource(PointSource.fromBizType(bizType).getCode())
            .expiredAt(calculateExpiredAt(rule))
            .status(PointStatus.VALID.getCode())
            .description(buildDescription(bizType, points))
            .build();
        
        transactionMapper.insert(tx);
        
        // 5. 更新账户统计
        accountMapper.updateStatsOnEarn(account.getId(), points);
        
        log.info("积分获取成功: userId={}, points={}, bizType={}, bizId={}", 
            userId, points, bizType, bizId);
        
        return PointResult.success(points);
    }
    
    private PointAccount getOrCreateAccount(Long userId) {
        PointAccount account = accountMapper.selectByUserId(userId);
        
        if (account == null) {
            account = PointAccount.builder()
                .userId(userId)
                .balance(0)
                .frozenBalance(0)
                .totalEarned(0)
                .totalSpent(0)
                .totalExpired(0)
                .version(0)
                .build();
            
            accountMapper.insert(account);
        }
        
        return account;
    }
    
    private LocalDateTime calculateExpiredAt(PointRule rule) {
        // 根据规则计算过期时间
        int validityDays = rule.getValidityDays();
        return LocalDateTime.now().plusDays(validityDays);
    }
    
    private int retryEarn(Long userId, String bizType, String bizId,
                          Integer points, PointRule rule, int remainRetries) {
        if (remainRetries <= 0) {
            throw new BusinessException("积分发放失败，请稍后重试");
        }
        
        PointAccount account = accountMapper.selectByUserId(userId);
        int updated = accountMapper.increaseBalance(account.getId(), 
            points, account.getVersion());
        
        if (updated == 0) {
            return retryEarn(userId, bizType, bizId, points, rule, remainRetries - 1);
        }
        
        return 0;
    }
}

// ========== 3. 积分消费服务 ==========

@Service
public class PointSpendService {
    
    @Autowired
    private PointAccountMapper accountMapper;
    @Autowired
    private PointTransactionMapper transactionMapper;
    
    /**
     * 积分消费（先冻后扣 + FIFO 过期策略）
     */
    @Transactional(rollbackFor = Exception.class)
    public PointResult spend(Long userId, String bizType, String bizId,
                             Integer points, String remark) {
        
        // 1. 幂等检查
        PointTransaction existTx = transactionMapper
            .selectByBizTypeAndBizId(bizType, bizId);
        if (existTx != null) {
            return PointResult.success("重复消费");
        }
        
        // 2. 冻结积分（FIFO：先冻即将过期的）
        int frozenPoints = freezePoints(userId, points);
        if (frozenPoints < points) {
            return PointResult.fail("可用积分不足");
        }
        
        // 3. 记录消费流水
        PointAccount account = accountMapper.selectByUserId(userId);
        
        PointTransaction tx = PointTransaction.builder()
            .userId(userId)
            .accountId(account.getId())
            .type(PointType.SPEND.getCode())
            .changeAmount(-points)
            .balanceBefore(account.getBalance())
            .balanceAfter(account.getBalance() - points)
            .bizType(bizType)
            .bizId(bizId)
            .status(PointStatus.VALID.getCode())
            .description(remark)
            .build();
        
        transactionMapper.insert(tx);
        
        // 4. 更新账户
        accountMapper.decreaseBalance(account.getId(), points);
        accountMapper.decreaseFrozen(account.getId(), frozenPoints);
        accountMapper.updateStatsOnSpend(account.getId(), points);
        
        log.info("积分消费成功: userId={}, points={}, bizType={}", 
            userId, points, bizType);
        
        return PointResult.success(points);
    }
    
    /**
     * 冻结积分（FIFO 策略：优先冻结即将过期的）
     */
    private int freezePoints(Long userId, Integer neededPoints) {
        int frozen = 0;
        
        // 查询即将过期的有效积分（按过期时间升序）
        List<PointTransaction> validPoints = transactionMapper
            .selectValidPointsByUserIdOrderByExpireAsc(userId);
        
        for (PointTransaction pt : validPoints) {
            if (frozen >= neededPoints) break;
            
            int canFreeze = Math.min(pt.getChangeAmount(), neededPoints - frozen);
            
            // 冻结这条流水
            transactionMapper.updateFrozenStatus(pt.getId(), canFreeze);
            frozen += canFreeze;
        }
        
        return frozen;
    }
}

// ========== 4. 积分过期服务 ==========

@Service
public class PointExpireService {
    
    @Autowired
    private PointTransactionMapper txMapper;
    @Autowired
    private PointAccountMapper accountMapper;
    @Autowired
    private PointExpireRecordMapper expireRecordMapper;
    
    /**
     * 定时过期任务（每天凌晨执行）
     */
    @Scheduled(cron = "0 0 2 * * ?")  // 每天凌晨2点
    public void expirePoints() {
        log.info("开始执行积分过期任务");
        
        int totalExpired = 0;
        int batchSize = 1000;
        long lastId = 0;
        
        while (true) {
            // 分批查询即将过期的积分
            List<PointTransaction> expiredPoints = txMapper
                .selectExpiringPoints(LocalDateTime.now(), lastId, batchSize);
            
            if (expiredPoints.isEmpty()) {
                break;
            }
            
            for (PointTransaction pt : expiredPoints) {
                try {
                    expireSinglePoint(pt);
                    totalExpired += pt.getChangeAmount();
                    lastId = pt.getId();
                } catch (Exception e) {
                    log.error("单条积分过期失败: id={}", pt.getId(), e);
                }
            }
            
            log.info("已处理 {} 条过期积分", totalExpired);
        }
        
        log.info("积分过期任务完成，共过期 {} 积分", totalExpired);
    }
    
    @Transactional
    public void expireSinglePoint(PointTransaction pt) {
        if (pt.getStatus() != PointStatus.VALID.getCode()) {
            return;  // 已使用或已过期，跳过
        }
        
        // 1. 更新流水状态
        txMapper.updateStatus(pt.getId(), PointStatus.EXPIRED.getCode());
        
        // 2. 扣减账户余额
        PointAccount account = accountMapper.selectByUserId(pt.getUserId());
        accountMapper.decreaseBalance(account.getId(), pt.getChangeAmount());
        accountMapper.updateStatsOnExpire(account.getId(), pt.getChangeAmount());
        
        // 3. 记录过期记录
        PointExpireRecord record = PointExpireRecord.builder()
            .userId(pt.getUserId())
            .transactionId(pt.getId())
            .expirePoints(pt.getChangeAmount())
            .status(0)  // 待处理
            .gmtExpired(pt.getExpiredAt())
            .build();
        expireRecordMapper.insert(record);
        
        log.info("积分过期: userId={}, points={}, expiredAt={}", 
            pt.getUserId(), pt.getChangeAmount(), pt.getExpiredAt());
    }
    
    /**
     * 过期前发送提醒（提前7天）
     */
    @Scheduled(cron = "0 0 10 * * ?")  // 每天上午10点
    public void sendExpireReminder() {
        LocalDateTime remindDate = LocalDateTime.now().plusDays(7);
        
        List<PointExpireStat> expiringPoints = txMapper
            .selectExpiringPointsGroupByUser(remindDate, 7);
        
        for (PointExpireStat stat : expiringPoints) {
            // 发送提醒消息
            sendSms(stat.getUserId(), 
                String.format("您有 %d 积分将于7天后过期，请尽快使用！", stat.getTotalPoints()));
        }
    }
}
```

## 🔥 扩展追问 1：积分高并发扣减如何保证不超扣？

> **A:** 核心是乐观锁 + 余额校验 + 幂等：

```java
// ========== 防超扣三保险 ==========

// 保险1：乐观锁
int updated = accountMapper.updateBalanceWithOptimisticLock(
    accountId, 
    -spendPoints,        // 扣减量
    currentBalance,      // 期望的当前余额（用于校验）
    version              // 当前版本号
);

// UPDATE user_points_account 
// SET balance = balance - #{spendPoints},
//     version = version + 1,
//     gmt_modified = NOW()
// WHERE id = #{accountId} 
//   AND balance >= #{spendPoints}           -- ⚠️ 余额校验
//   AND version = #{version}                -- ⚠️ 乐观锁

if (updated == 0) {
    throw new BusinessException("并发冲突，积分不足");
}

@RedisLock(key = "point:lock:user:{}", waitTime = 3, leaseTime = 10)
```

## 🔥 扩展追问 2：积分过期策略有哪些？FIFO 还是按比例？

> **A:** 常见三种策略：

```
┌─────────────────────────────────────────────────────────────────┐
│              积分过期策略对比                                     │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  策略1: FIFO（先到先过期）推荐                                  │
│  ┌─────────────────────────────────────────────────────────┐  │
│  │ 获取时间：2024-01-01 (100分, 过期2025-01-01)             │  │
│  │ 获取时间：2024-06-01 (200分, 过期2025-06-01)             │  │
│  │ 获取时间：2024-12-01 (50分,  过期2025-12-01)             │  │
│  │                                                          │  │
│  │ 消费时：先冻先过期的（2025-01-01那批先被使用）           │  │
│  │ ✅ 用户体验好，积分不会莫名其妙消失                       │  │
│  └─────────────────────────────────────────────────────────┘  │
│                                                                 │
│  策略2: LIFO（后到先过期）                                     │
│  │ 先冻最新获取的积分                                        │
│  │ ⚠️ 冷门库存永远不过期，不利于积分消耗                     │  │
│                                                                 │
│  策略3: 按比例过期                                             │
│  │ 每月固定比例积分过期（如每月末过期 10%）                  │
│  │ ⚠️ 计算复杂，用户感知差                                   │  │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

```java
// FIFO 过期实现
@Service
public class PointExpireStrategy {
    
    /**
     * 按 FIFO 策略选择要使用的积分
     * 返回：选中的流水ID列表
     */
    public List<Long> selectPointsByFIFO(Long userId, Integer neededPoints) {
        List<Long> selectedIds = new ArrayList<>();
        int selected = 0;
        
        // 按过期时间升序 + 创建时间升序
        List<PointTransaction> validPoints = transactionMapper
            .selectValidPointsOrderByExpireTimeAsc(userId);
        
        for (PointTransaction pt : validPoints) {
            if (selected >= neededPoints) break;
            
            int canUse = Math.min(pt.getRemainAmount(), neededPoints - selected);
            selectedIds.add(pt.getId());
            selected += canUse;
        }
        
        if (selected < neededPoints) {
            throw new BusinessException("可用积分不足");
        }
        
        return selectedIds;
    }
}
```

## 🔥 扩展追问 3：积分和钱如何换算？

> **A:** 积分本质是虚拟货币，有两种常见模式：

| 模式 | 说明 | 例子 |
|------|------|------|
| **1元=1积分** | 简单直观 | 消费100元得100积分 |
| **1元=N积分** | 灵活配置 | 消费100元得1000积分（1:10） |

```java
@Service
public class PointExchangeService {
    
    private static final int POINTS_PER_YUAN = 100;  // 1元 = 100积分
    
    /**
     * 消费金额转积分
     */
    public int yuanToPoints(BigDecimal amount) {
        return amount.multiply(new BigDecimal(POINTS_PER_YUAN)).intValue();
    }
    
    /**
     * 积分抵扣金额（上限）
     */
    public BigDecimal pointsToYuan(int points) {
        return new BigDecimal(points).divide(new BigDecimal(POINTS_PER_YUAN), 2, RoundingMode.FLOOR);
    }
    
    /**
     * 计算订单可抵扣金额
     * 规则：积分抵现上限为订单金额的10%
     */
    public PointDeductResult calculateDeduct(Long userId, BigDecimal orderAmount, Integer userPoints) {
        // 最大可抵扣金额 = 订单金额 × 10%
        BigDecimal maxDeduct = orderAmount.multiply(new BigDecimal("0.1"));
        
        // 用户积分可抵扣金额
        BigDecimal pointsValue = pointsToYuan(userPoints);
        
        // 取较小值
        BigDecimal actualDeduct = pointsValue.min(maxDeduct);
        
        // 实际使用积分
        int usePoints = Math.min(userPoints, actualDeduct.multiply(new BigDecimal(POINTS_PER_YUAN)).intValue());
        
        return PointDeductResult.builder()
            .usePoints(usePoints)
            .deductAmount(actualDeduct)
            .remainingPoints(userPoints - usePoints)
            .build();
    }
}
```

## 🔥 扩展追问 4：积分退款如何处理？

> **A:** 退款时需要**回退积分**或**扣除已使用积分**：

```java
@Service
public class PointRefundService {
    
    @Autowired
    private PointTransactionMapper txMapper;
    @Autowired
    private PointAccountMapper accountMapper;
    
    /**
     * 退款处理积分
     */
    @Transactional
    public PointResult handleRefund(Long userId, String orderId, Integer refundPoints) {
        // 1. 查询原订单的积分获取记录
        PointTransaction originalTx = txMapper
            .selectByBizTypeAndBizId("ORDER", orderId);
        
        if (originalTx == null) {
            // 原订单没有积分，跳过
            return PointResult.success("无需处理");
        }
        
        // 2. 检查积分是否已使用
        if (originalTx.getStatus() == PointStatus.SPENT.getCode()) {
            // 已使用：扣除已用积分 + 回退剩余积分
            Integer usedPoints = originalTx.getUsedPoints();
            Integer remainPoints = originalTx.getRemainPoints();
            
            // 扣除已用积分（从用户账户扣）
            deductUsedPoints(userId, usedPoints);
            
            // 回退剩余积分
            if (remainPoints > 0) {
                refundPoints = remainPoints;
            } else {
                refundPoints = 0;
            }
        }
        
        // 3. 如果有剩余积分未使用，直接退款
        if (refundPoints > 0) {
            PointTransaction refundTx = PointTransaction.builder()
                .userId(userId)
                .accountId(getAccountId(userId))
                .type(PointType.REFUND.getCode())
                .changeAmount(refundPoints)
                .bizType("ORDER_REFUND")
                .bizId(orderId)
                .status(PointStatus.REFUND.getCode())
                .description("订单退款，积分退回")
                .build();
            
            txMapper.insert(refundTx);
            accountMapper.increaseBalance(getAccountId(userId), refundPoints);
        }
        
        // 4. 标记原积分流水为已退款
        txMapper.updateStatus(originalTx.getId(), PointStatus.REFUND.getCode());
        
        return PointResult.success(refundPoints);
    }
    
    /**
     * 扣除已使用的积分（不够扣怎么办？）
     * 方案：不足部分从退款金额中扣除
     */
    private void deductUsedPoints(Long userId, Integer usedPoints) {
        PointAccount account = accountMapper.selectByUserId(userId);
        
        if (account.getBalance() >= usedPoints) {
            // 余额够，直接扣
            accountMapper.decreaseBalance(account.getId(), usedPoints);
        } else {
            // 余额不够，扣光余额，差额走现金补偿
            int actualDeduct = account.getBalance();
            accountMapper.decreaseBalance(account.getId(), actualDeduct);
            
            // 差额：通知运营手动处理
            log.warn("积分不足，需现金补偿: userId={}, usedPoints={}, actualDeduct={}", 
                userId, usedPoints, actualDeduct);
            
            alertService.send("用户积分不足需补偿", 
                String.format("userId=%d, 欠积分数=%d", userId, usedPoints - actualDeduct));
        }
    }
}
```

## 🔥 扩展追问 5：如何防止刷积分？

> **A:** 刷积分是积分系统最头疼的问题，需要多层防护：

```
┌─────────────────────────────────────────────────────────────────┐
│              防刷积分体系                                        │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│   第一层：规则限制                                               │
│   ┌─────────────────────────────────────────────────────────┐  │
│   │ - 每日获取上限（规则表配置）                              │  │
│   │ - 单笔获取上限                                           │  │
│   │ - 风控黑名单                                             │  │
│   └─────────────────────────────────────────────────────────┘  │
│                                                                 │
│   第二层：行为检测                                               │
│   ┌─────────────────────────────────────────────────────────┐  │
│   │ - 同一IP/设备短时间内大量请求                             │  │
│   │ - 短时间内积分变化异常                                    │  │
│   │ - 设备指纹 + IP 聚类分析                                  │  │
│   └─────────────────────────────────────────────────────────┘  │
│                                                                 │
│   第三层：MQ 异步 + 风控后置                                    │
│   ┌─────────────────────────────────────────────────────────┐  │
│   │ - 积分发放先写 MQ                                      │  │
│   │ - Consumer 异步执行发放 + 风控校验                     │  │
│   │ - 风控不通过则拒绝，告警                                │  │
│   └─────────────────────────────────────────────────────────┘  │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

```java
@Service
public class PointAntiBrushService {
    
    @Autowired
    private RiskControlService riskControl;
    
    /**
     * 积分发放前置检查
     */
    public CheckResult preCheck(Long userId, String bizType, Integer points,
                                String ip, String deviceId) {
        
        // 1. 黑名单检查
        if (riskControl.isBlackUser(userId)) {
            return CheckResult.reject("用户黑名单");
        }
        
        // 2. IP/设备频率检查
        if (riskControl.checkIpFrequency(ip, 60, 10)) {  // 1分钟10次
            return CheckResult.reject("IP频率异常");
        }
        
        // 3. 每日积分上限检查
        Integer dailyEarned = getDailyEarnedPoints(userId);
        Integer maxDaily = getMaxDailyPoints(bizType);
        
        if (dailyEarned + points > maxDaily) {
            return CheckResult.reject("超过每日积分上限");
        }
        
        // 4. 设备指纹检查（同一设备多账号）
        if (riskControl.checkDeviceMultiAccounts(deviceId, 3)) {
            return CheckResult.suspicious("设备多账号");
        }
        
        return CheckResult.pass();
    }
    
    /**
     * 异步风控后置检查（MQ 消费时）
     */
    public void asyncRiskCheck(PointEarnMessage message) {
        // 异步分析用户行为
        UserBehavior behavior = analyzeBehavior(message.getUserId());
        
        if (behavior.isSuspicious()) {
            // 扣回积分
            deductSuspiciousPoints(message.getUserId(), message.getPoints());
            
            // 告警
            alertService.sendRiskAlert(
                "疑似刷积分", 
                String.format("userId=%d, points=%d, reason=%s", 
                    message.getUserId(), message.getPoints(), behavior.getReason()));
        }
    }
}
```

>
> **核心是：流水记录一切变更、过期用定时任务扣减、获取和消费都走幂等校验、高并发用消息队列削峰。**

---

> "我之前设计过一套积分系统，踩过一个坑：用户积分过期时直接扣余额，没有记录过期流水。结果用户投诉说'我积分怎么平白无故少了'，查账根本查不出来。后来改了方案：过期扣减也生成一条 PointType.EXPIRE 的流水，用户能在明细里看到哪笔积分过期了，投诉率降了 90%。另一个经验是积分兑换比例的设计：当时定的 100积分=1元，结果用户薅羊毛薅得太狠，每单都把积分用光。后来加了'积分抵现上限10%'的限制，既让用户有获得感，又不至于被薅秃。"
