```java
// 租户解析拦截器（从Token/Header中提取）
@Component
public class TenantInterceptor implements HandlerInterceptor {
    
    @Override
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response, 
                             Object handler) throws Exception {
        // 1. 从JWT Token中解析租户信息
        String token = request.getHeader("Authorization");
        if (StringUtils.hasText(token)) {
            Claims claims = jwtUtil.parseToken(token);
            Long tenantId = claims.get("tenantId", Long.class);
            String tenantCode = claims.get("tenantCode", String.class);
            TenantContext.setTenantId(tenantId);
            TenantContext.setTenantCode(tenantCode);
        }
        
        // 2. 或从Header直接获取
        String tenantIdHeader = request.getHeader("X-Tenant-Id");
        if (StringUtils.hasText(tenantIdHeader)) {
            TenantContext.setTenantId(Long.parseLong(tenantIdHeader));
        }
        
        return true;
    }
    
    @Override
    public void afterCompletion(HttpServletRequest request, HttpServletResponse response,
                                 Object handler, Exception ex) throws Exception {
        TenantContext.clear();
    }
}
```

```java
@Configuration
public class MultiTenantDataSourceConfig {
    
    // 租户数据源缓存
    private final Map<Long, DataSource> tenantDataSourceCache = new ConcurrentHashMap<>();
    
    @Autowired
    private TenantDataSourceProperties tenantProperties;
    
    @Bean
    public DataSource dynamicDataSource() {
        DynamicDataSource dataSource = new DynamicDataSource();
        
        // 设置默认数据源（主库）
        dataSource.setDefaultTargetDataSource(primaryDataSource());
        
        // 设置目标数据源映射
        Map<Object, Object> targetDataSources = new HashMap<>();
        targetDataSources.put("primary", primaryDataSource());
        dataSource.setTargetDataSources(targetDataSources);
        
        return dataSource;
    }
    
    public DataSource getTenantDataSource(Long tenantId) {
        // 1. 先从缓存获取
        DataSource cached = tenantDataSourceCache.get(tenantId);
        if (cached != null) {
            return cached;
        }
        
        // 2. 缓存未命中，从配置加载
        Tenant tenant = tenantService.getById(tenantId);
        if (tenant == null || StringUtils.isBlank(tenant.getDbConfig())) {
            throw new TenantNotFoundException("租户不存在: " + tenantId);
        }
        
        // 3. 解析数据库配置
        TenantDbConfig dbConfig = JSON.parseObject(tenant.getDbConfig(), TenantDbConfig.class);
        
        // 4. 创建租户专属数据源
        DataSource tenantDs = createDataSource(dbConfig);
        
        // 5. 加入缓存
        tenantDataSourceCache.put(tenantId, tenantDs);
        
        return tenantDs;
    }
    
    private DataSource createDataSource(TenantDbConfig config) {
        HikariDataSource dataSource = new HikariDataSource();
        dataSource.setJdbcUrl(config.getJdbcUrl());
        dataSource.setUsername(config.getUsername());
        dataSource.setPassword(config.getPassword());
        dataSource.setDriverClassName("com.mysql.cj.jdbc.Driver");
        dataSource.setMaximumPoolSize(20);
        dataSource.setMinimumIdle(5);
        return dataSource;
    }
}

// 动态数据源切换
public class DynamicDataSource extends AbstractRoutingDataSource {
```

```java
@Configuration
public class SchemaRoutingDataSource extends AbstractDataSource {
    
    private DataSource primaryDataSource;
    
    public void setPrimaryDataSource(DataSource primaryDataSource) {
        this.primaryDataSource = primaryDataSource;
    }
    
    @Override
    public Connection getConnection() throws SQLException {
        Connection conn = primaryDataSource.getConnection();
        Long tenantId = TenantContext.getTenantId();
        
        if (tenantId != null) {
            // 设置Session变量指定Schema
            String schemaName = getSchemaName(tenantId);
            conn.createStatement().execute("USE " + schemaName);
        }
        
        return conn;
    }
    
    private String getSchemaName(Long tenantId) {
        Tenant tenant = tenantService.getById(tenantId);
        return tenant != null ? tenant.getSchemaName() : "public";
    }
}

    }
```

```java
// MyBatis Plus 租户插件
@Configuration
public class MybatisPlusConfig {
    
    @Bean
    public MybatisPlusInterceptor mybatisPlusInterceptor() {
        MybatisPlusInterceptor interceptor = new MybatisPlusInterceptor();
        
        // 1. 分页插件
        interceptor.addInnerInterceptor(new PaginationInnerInterceptor());
        
        // 2. 租户插件（自动注入tenant_id）
        interceptor.addInnerInterceptor(new TenantLineInnerInterceptor(
            new TenantLineHandler() {
                @Override
                public Expression getTenantId() {
                    // 从上下文获取当前租户ID
                    Long tenantId = TenantContext.getTenantId();
                    return new LongValue(tenantId != null ? tenantId : 0L);
                }
                
                @Override
                public String getTenantIdColumn() {
                    return "tenant_id";
                }
                
                @Override
                public boolean ignoreTable(String tableName) {
                    // 系统表不走租户隔离
                    return Arrays.asList("sys_user", "tenant", "tenant_package")
                        .contains(tableName);
                }
            }
        ));
        
        // 3. 数据权限插件
        interceptor.addInnerInterceptor(new DataPermissionInterceptor());
        
        return interceptor;
    }
}

// 全局租户过滤器（所有查询自动注入tenant_id条件）
@Component
@Slf4j
public class TenantQueryFilter {
    
    @Autowired
    private SqlSessionFactory sqlSessionFactory;
    
    @PostConstruct
    public void init() {
        // 配置全局SQL改写器
        sqlSessionFactory.getConfiguration().addInterceptor(new TenantSqlInterceptor());
    }
}

public class TenantSqlInterceptor implements Interceptor {
    
    @Override
    public Object intercept(Invocation invocation) throws Throwable {
        // 获取原始SQL
        BoundSql boundSql = invocation.getArgs()[0];
        String originalSql = boundSql.getSql();
        
        Long tenantId = TenantContext.getTenantId();
        if (tenantId != null) {
            // 改写SQL添加租户条件
            String tenantCondition = "tenant_id = " + tenantId;
            String rewrittenSql = addTenantCondition(originalSql, tenantCondition);
            
            // 替换SQL
            Field field = boundSql.getClass().getDeclaredField("sql");
            field.setAccessible(true);
            field.set(boundSql, rewrittenSql);
        }
        
        return invocation.proceed();
    }
    
    private String addTenantCondition(String sql, String condition) {
        // 简单实现：实际用AST解析
        if (sql.toLowerCase().contains("where")) {
            return sql + " AND " + condition;
        } else {
            return sql + " WHERE " + condition;
        }
    }
}
```

## 2. 扩展追问

### 2.1 🔥 扩展追问 1：三种方案如何选择？

> **A:**

**选型决策矩阵：**

| 考量维度 | 独立数据库 | 独立Schema | 共享Schema |
|---------|-----------|-----------|-----------|
| **数据隔离级别** | 物理隔离（最高） | 逻辑隔离（高） | 字段隔离（中等） |
| **成本** | 💰💰💰（每租户1套DB） | 💰💰（1套DB多Schema） | 💰（1套DB） |
| **租户规模上限** | 100-500个 | 500-2000个 | 无上限 |
| **单租户数据量** | 不限 | 中等（单DB容量） | 受限于单DB |
| **迁移/备份** | 独立备份，简单 | Schema级备份 | 需按租户筛选 |
| **跨租户查询** | ❌ 不支持 | ⚠️ 跨Schema查询复杂 | ✅ 支持（需特殊处理） |
| **升级维护** | 逐租户升级，成本高 | 逐租户或批量升级 | 统一升级，简单 |
| **故障影响** | 单租户故障不影响他 | 单租户故障不影响他 | 单租户可能拖垮全部 |
| **适用场景** | 金融、医疗（强隔离） | 中大型企业（中等隔离） | 通用SaaS（海量租户） |

**实际推荐：**

```
选型公式：

if (数据安全要求极高 && 预算充足) → 独立数据库
else if (租户数量 < 2000 && 中等隔离) → 独立Schema  
else → 共享Schema + 租户字段

混合策略（推荐）：
├── 核心业务表：共享Schema + tenant_id
├── 高敏感表（财务、隐私）：独立Schema
└── 超大客户：独立数据库
```

```sql
-- 创建策略：用户只能看自己的租户数据
CREATE POLICY tenant_isolation_policy ON orders
    USING (tenant_id = current_setting('app.current_tenant_id')::bigint);
```

```java
@Slf4j
public class TenantBackupService {

    private JdbcTemplate jdbcTemplate;
    
    /**
     * 导出指定租户全量数据
     */
    public void exportTenantData(Long tenantId, Path exportDir) {

        Path tenantDir = exportDir.resolve("tenant_" + tenantId);
        Files.createDirectories(tenantDir);
        
        // 2. 获取租户所有业务表
        List<String> businessTables = getBusinessTables();
        
        for (String table : businessTables) {
            // 3. 导出表数据（带租户条件）
            String exportFile = tenantDir.resolve(table + ".sql").toString();
            exportTableToFile(table, tenantId, exportFile);
        }
        
        // 4. 打包
        zipDirectory(tenantDir, exportDir.resolve("tenant_" + tenantId + ".zip"));
        
        log.info("租户[{}]数据导出完成: {}", tenantId, exportDir);
    }
    
    private void exportTableToFile(String table, Long tenantId, String file) {
        String sql = String.format(
            "SELECT * FROM %s WHERE tenant_id = %d INTO OUTFILE '%s' " +
            "FIELDS TERMINATED BY ',' ENCLOSED BY '\"' LINES TERMINATED BY '\\n'",
            table, tenantId, file
        );
        jdbcTemplate.execute(sql);
    }
    
    /**
     * 恢复租户数据
     */
    @Transactional
    public void restoreTenantData(Long tenantId, Path backupFile) {
        // 1. 清理旧数据
        deleteTenantData(tenantId);
        
        // 2. 解压备份文件
        Path tempDir = unzip(backupFile);
        
        // 3. 逐表恢复
        Path tenantDir = tempDir.resolve("tenant_" + tenantId);
        for (String table : getBusinessTables()) {
            Path sqlFile = tenantDir.resolve(table + ".sql");
            if (Files.exists(sqlFile)) {
                loadDataFromFile(table, sqlFile);
            }
        }
        
        log.info("租户[{}]数据恢复完成", tenantId);
    }
}
```

```java
@Service
@Slf4j
public class TenantUpgradeService {
    
    /**
     * 批量升级租户（分批执行，支持回滚）
     */
    public UpgradeResult batchUpgrade(Long batchSize, int sleepSeconds) {
        UpgradeResult result = new UpgradeResult();
        
        // 1. 获取待升级租户列表
        List<Long> pendingTenants = getPendingUpgradeTenants();
        result.setTotal(pendingTenants.size());
        
        int success = 0;
        int failed = 0;
        List<Long> failedTenants = new ArrayList<>();
        
        for (Long tenantId : pendingTenants) {
            try {
                // 设置租户上下文
                TenantContext.setTenantId(tenantId);
                
                // 执行升级
                upgradeTenant(tenantId);
                
                // 记录成功
                markUpgradeSuccess(tenantId);
                success++;
                
                // 休眠，控制升级速度
                if (sleepSeconds > 0) {
                    Thread.sleep(sleepSeconds * 1000L);
                }
                
            } catch (Exception e) {
                log.error("租户[{}]升级失败", tenantId, e);
                markUpgradeFailed(tenantId, e.getMessage());
                failedTenants.add(tenantId);
                failed++;
            } finally {
                TenantContext.clear();
            }
        }
        
        result.setSuccess(success);
        result.setFailed(failed);
        result.setFailedTenants(failedTenants);
        
        return result;
    }
    
    private void upgradeTenant(Long tenantId) {
        // 1. 执行租户特定的升级SQL
        executeUpgradeScript(tenantId);
        
        // 2. 清理租户缓存
        redisTemplate.delete("tenant:" + tenantId + ":*");
        
        // 3. 刷新元数据
        refreshTenantMetadata(tenantId);
    }
}
```

### 2.5 🔥 扩展追问 5：如何实现租户的配额管理和计费？

**配额管理核心逻辑：**

```java
// 租户配额检查注解
@Target(ElementType.METHOD)
@Retention(RetentionPolicy.RUNTIME)
public @interface TenantQuota {
    String resource() default "";  // 资源类型：user_count, storage, api_call
    long limit() default 0;       // 配额上限
}

// 配额检查拦截器
@Component
@Order(15)
public class QuotaCheckInterceptor implements MethodInterceptor {
    
    @Autowired
    private TenantQuotaService quotaService;
    
    @Override
    public Object invoke(MethodInvocation invocation) throws Throwable {
        Method method = invocation.getMethod();
        TenantQuota quota = method.getAnnotation(TenantQuota.class);
        
        if (quota != null) {
            Long tenantId = TenantContext.getTenantId();
            String resource = quota.resource();
            
            // 1. 检查套餐配额
            if (!quotaService.checkQuota(tenantId, resource, quota.limit())) {
                throw new QuotaExceededException(
                    "租户[{}]资源[{}]配额超限，当前限制:{}",
                    tenantId, resource, quota.limit()
                );
            }
        }
        
        return invocation.proceed();
    }
}

// 使用示例
@Service
public class UserService {
    
    @TenantQuota(resource = "user_count", limit = 100)
    public void createUser(User user) {
        // 业务逻辑
    }
}

// 配额服务
@Service
@Slf4j
public class TenantQuotaService {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    /**
     * 配额检查 + 计数
     */
    public boolean checkQuota(Long tenantId, String resource, long limit) {
        String key = String.format("quota:%d:%s", tenantId, resource);
        
        // 获取当前使用量
        Long currentCount = redisTemplate.opsForValue().increment(key);
        
        if (currentCount == null) {
            // 缓存未命中，查询数据库
            currentCount = queryResourceCount(tenantId, resource);
            redisTemplate.opsForValue().set(key, currentCount, 1, TimeUnit.HOURS);
            currentCount = redisTemplate.opsForValue().increment(key);
        }
        
        if (currentCount > limit) {
            log.warn("租户[{}]资源[{}]配额超限: {}/{}", tenantId, resource, currentCount, limit);
            return false;
        }
        
        return true;
    }
    
    /**
     * 获取租户套餐信息
     */
    public TenantPackage getTenantPackage(Long tenantId) {
        Tenant tenant = tenantMapper.selectById(tenantId);
        return packageMapper.selectByCode(tenant.getPackageCode());
    }
}
```

```java
@Service
@Slf4j
public class CrossTenantQueryManager {

        }
    }

     */

            // 执行导出
            exportService.export(request);
        } finally {
            TenantContext.clear();
        }
    }

// 租户上下文增强
public class TenantContext {
    
    private static final ThreadLocal<Long> TENANT_ID = new ThreadLocal<>();
    private static final ThreadLocal<Boolean> ADMIN_MODE = new ThreadLocal<>();
    
    public static String clearForAdmin() {
        String previous = String.valueOf(TENANT_ID.get());
        TENANT_ID.remove();
        ADMIN_MODE.set(true);
        return previous;
    }
    
    public static void restore(String previousTenantId) {
        if (previousTenantId != null && !"null".equals(previousTenantId)) {
            TENANT_ID.set(Long.parseLong(previousTenantId));
        }
        ADMIN_MODE.remove();
    }
    
    public static boolean isAdminMode() {
        return Boolean.TRUE.equals(ADMIN_MODE.get());
    }
}
```

## 3. 一句话总结

> **多租户数据隔离有三种主流方案：独立数据库（最高隔离、成本高）、独立Schema（中等隔离、成本中等）、共享Schema+tenant_id（灵活、成本低）。大多数SaaS产品采用方案三配合MyBatis Plus租户插件实现透明隔离，并通过配额管理实现租户级别的资源控制和计费。**

## 4. 面试加分话术

> "我之前负责过公司SaaS平台的多租户架构设计，当时考察了三种方案，最终选型是**共享Schema + tenant_id字段**的方案。
> 
> 选型依据是：当时平台上有几千个小微企业租户，数据量适中但租户数量多，独立数据库方案成本太高撑不住。
> 
> 核心技术实现是基于 **MyBatis Plus 的 TenantLineInnerInterceptor**，在SQL层面自动注入 `tenant_id` 条件，对业务代码完全透明。
> 
> 但这个方案有个坑必须注意：**跨租户查询和超级管理员场景**需要特殊处理，我们用 ThreadLocal 的 `ADMIN_MODE` 标记来区分，普通租户请求强制带tenant_id，管理员请求跳过过滤。
> 
> 另外数据安全方面我们做了多层保障：应用层拦截、数据库层用 PostgreSQL 的 RLS（行级安全策略）、审计日志记录所有跨租户操作。
> 
> 对于大客户或有合规要求的客户，我们提供了**混合部署模式**：核心数据（财务、隐私）走独立Schema，普通业务数据走共享Schema，兼顾成本和合规。"
