```java
// ========== 方案1：分布式锁防止任务重叠 ==========

@Service
public class DistributedScheduledTask {
    
    @Autowired
    private RedissonClient redissonClient;
    
    private static final String CLEANUP_LOCK_KEY = "scheduled:lock:cleanup";
    private static final long LOCK_LEASE_TIME = 30 * 60 * 1000;  // 30分钟（比最大执行时间略长）
    
    /**
     * 带分布式锁的定时任务
     * 核心思想：同一时刻只有一个节点能抢到锁，其他节点跳过
     */
    @Scheduled(cron = "0 */5 * * * ?")  // 每5分钟
    public void cleanupExpiredData() {
        RLock lock = redissonClient.getLock(CLEANUP_LOCK_KEY);
        
        // tryLock 非阻塞，抢不到就跳过
        boolean acquired = lock.tryLock();
        
        if (!acquired) {
            log.info("上一个清理任务还在执行，跳过本次");
            return;
        }
        
        try {
            log.info("获得清理锁，开始执行...");
            doCleanup();
        } finally {
            // ⚠️ finally 中释放锁（防止业务异常导致锁不释放）
            if (lock.isHeldByCurrentThread()) {
                lock.unlock();
            }
        }
    }
    
    private void doCleanup() {
        // 清理逻辑
    }
}


// ========== 方案2：任务分片 + 限流 ==========

@Service
public class ShardedCleanupTask {
    
    @Autowired
    private JdbcTemplate jdbcTemplate;
    @Autowired
    private RedissonClient redissonClient;
    
    // 分片数：100
    private static final int SHARD_COUNT = 100;
    // 每批处理量
    private static final int BATCH_SIZE = 1000;
    // 每批间隔（给数据库喘气时间）
    private static final long BATCH_DELAY_MS = 50;
    
    /**
     * 分片清理定时任务
     * 每次只处理一个分片，多次调度完成全部清理
     */
    @Scheduled(cron = "0 */5 * * * ?")
    public void cleanupExpiredData() {
        // 1. 获取当前应该处理哪个分片
        int currentShard = getCurrentShard();
        
        // 2. 清理当前分片（只处理100万分数据）
        cleanupShard(currentShard);
        
        // 3. 更新分片进度
        saveShardProgress(currentShard);
    }
    
    private void cleanupShard(int shard) {
        long startTime = System.currentTimeMillis();
        int totalDeleted = 0;
        
        while (true) {
            // 每次最多删除 1000 条
            String sql = """
                DELETE FROM order_log 
                WHERE status = 'COMPLETED' 
                  AND created_at < DATE_SUB(NOW(), INTERVAL 90 DAY)
                  AND (id % 100) = ?   -- 只删除当前分片
                LIMIT 1000
                """;
            
            int deleted = jdbcTemplate.update(sql, shard);
            totalDeleted += deleted;
            
            if (deleted == 0) {
                break;  // 当前分片清理完毕
            }
            
            // ⚠️ 每批之间短暂休眠，防止数据库被打满
            try {
                Thread.sleep(BATCH_DELAY_MS);
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
                break;
            }
            
            // ⚠️ 超时保护：执行超过10分钟强制退出
            if (System.currentTimeMillis() - startTime > 10 * 60 * 1000) {
                log.warn("分片{}清理超时，已处理{}条", shard, totalDeleted);
                break;
            }
        }
        
        log.info("分片{}清理完成，共删除{}条，耗时{}ms", 
            shard, totalDeleted, System.currentTimeMillis() - startTime);
    }
    
    /**
     * 轮询获取下一个未完成的分片
     */
    private int getCurrentShard() {
        String progressKey = "cleanup:shard:progress";
        RAtomicLong current = redissonClient.getAtomicLong(progressKey);
        
        long currentShard = current.incrementAndGet();
        if (currentShard >= SHARD_COUNT) {
            // 重置，从头开始
            current.set(0);
            return 0;
        }
        return (int) currentShard;
    }
}


// ========== 方案3：任务队列 + 异步处理 ==========

@Service
public class AsyncCleanupTask {
    
    @Autowired
    private TaskExecutor taskExecutor;  // Spring 内置线程池
    
    private static final int CONCURRENT_WORKERS = 3;  // 同时3个worker
    
    @Scheduled(cron = "0 */5 * * * ?")
    public void scheduleCleanup() {
        // 1. 计算需要清理的数据，拆成任务单元
        List<CleanupTask> tasks = buildCleanupTasks();
        
        // 2. 放入队列，异步并行处理
        for (CleanupTask task : tasks) {
            taskExecutor.execute(() -> processTask(task));
        }
    }
    
    private void processTask(CleanupTask task) {
        // 带信号量控制并发数
        Semaphore semaphore = new Semaphore(CONCURRENT_WORKERS);
        
        try {
            semaphore.acquire();
            doCleanup(task);
        } finally {
            semaphore.release();
        }
    }
    
    private List<CleanupTask> buildCleanupTasks() {
        // 按时间范围拆分成小任务
        // 例如：清理90天数据 → 拆成9个任务，每个任务清理10天
        List<CleanupTask> tasks = new ArrayList<>();
        LocalDate endDate = LocalDate.now().minusDays(90);
        LocalDate startDate = endDate.minusDays(10);
        
        for (int i = 0; i < 9; i++) {
            tasks.add(new CleanupTask(startDate, startDate.plusDays(10)));
            startDate = startDate.plusDays(10);
        }
        
        return tasks;
    }
}

 * Quartz 任务实现（自动 Misfire 处理）

    public void execute(JobExecutionContext context) throws JobExecutionException {

            cleanupService.cleanupFull();
        } else {
            // 正常增量清理
            cleanupService.cleanupIncremental();
        }
    }
```

## 🔥 扩展追问 1：分布式锁实现有哪些坑？

> **A:** 分布式锁有多个坑，核心是**锁粒度、锁超时、锁续期**：

```java
// ========== 常见坑1：锁超时比任务执行时间短 ==========

@Service
public class BadLockExample {
    
    public void badTask() {
        RLock lock = redissonClient.getLock("task:lock");
        
        // ⚠️ 坑1：锁超时 30 秒，但任务要跑 60 秒
        // 30 秒后锁自动释放，下一个节点进来 → 双重执行
        lock.tryLock(30, TimeUnit.SECONDS);
        
        doTask();  // 60 秒
        lock.unlock();
    }
}

// ✅ 修正：锁超时 = 任务最大执行时间 × 1.5
private static final long LOCK_LEASE_TIME = 90 * 60 * 1000;  // 90分钟


// ========== 常见坑2：finally 中 unlock 但锁已自动释放 ==========

@Service
public class BadLockExample2 {
    
    public void badTask() {
        RLock lock = redissonClient.getLock("task:lock");
        lock.tryLock();
        
        try {
            doTask();  // 业务逻辑抛异常
            if (condition) {
                return;  // ⚠️ 坑：提前 return，没释放锁
            }
        } finally {
            // 如果抛异常前已经 return，finally 会执行，但此时锁可能已被自动释放
            lock.unlock();
        }
    }
}

// ✅ 修正：加 isHeldByCurrentThread 判断
try {
    doTask();
} finally {
    if (lock.isHeldByCurrentThread()) {
        lock.unlock();
    }
}


// ========== 常见坑3：单机锁在多节点环境下无效 ==========

@Service
public class BadLockExample3 {
    
    // ⚠️ 坑：synchronized 是 JVM 锁，只对单个 JVM 有效
    // 多节点部署时，每个节点都有自己的锁 → 无意义
    private final Object localLock = new Object();
    
    public void badTask() {
        synchronized (localLock) {  // ❌ 无效
            doTask();
        }
    }
}

// ✅ 修正：必须用分布式锁（Redis/ZK/Etcd）
RLock lock = redissonClient.getLock("task:lock");
lock.tryLock();


// ========== 常见坑4：Redisson WatchDog 自动续期问题 ==========

@Service
public class RedissonWatchDogExample {
    
    // tryLock() 默认开启 WatchDog（每10秒续期）
    // ⚠️ 坑：如果没有主动 unlock()，锁会在 30 秒后自动释放
    // （WatchDog 默认 30 秒 TTL）
    
    public void task() {
        RLock lock = redissonClient.getLock("task:lock");
        
        // ✅ 推荐：显式指定 leaseTime（禁用 WatchDog）
        // 锁会自动在指定时间后释放，不续期
        boolean acquired = lock.tryLock(60, 60, TimeUnit.SECONDS);
        
        if (acquired) {
            try {
                doTask();
            } finally {
                lock.unlock();
            }
        }
    }
}
```

```java
xxl-job-admin 配置：
- 路由策略：第一个（保证只有一台执行）
- 任务超时时间：30分钟
- 失败重试次数：0（清理任务不重试）
- 阻塞处理策略：单机串行（不允许并发）

@XxlJob("dataCleanupJob")
public ReturnT<String> execute() {
    // xxl-job 保证：同一时刻只有一台机器执行
    // 如果机器挂了，xxl-job 自动在其他机器触发（Misfire）
    
    // ✅ 分页删除，防止锁表
    int totalDeleted = 0;
    while (true) {
        int deleted = XxlJobHelper.executeSql("""
            DELETE FROM order_log 
            WHERE created_at < DATE_SUB(NOW(), INTERVAL 90 DAY) 
            LIMIT 1000
            """);
        
        totalDeleted += deleted;
        if (deleted == 0) break;
        
        // ⚠️ 重要：每批休息 50ms，给数据库喘气
        Thread.sleep(50);
        
        // xxl-job 任务超时控制
        if (XxlJobHelper.getTaskTimeout() <= 0) break;
    }
    
    return ReturnT.SUCCESS;
}
```

## 🔥 扩展追问 3：分页删除和批量删除怎么选？

> **A:** 分页删除 vs 批量删除对比：

| 维度 | 分页 DELETE | 单次批量 DELETE |
|------|------------|----------------|
| **锁范围** | 分批获取锁，锁粒度小 | 一次性锁大量行 |
| **主从延迟** | 低（每批小事务） | 高（大事务同步慢） |
| **binlog** | 产生多个小事务 | 产生一个大事务 binlog |
| **回滚代价** | 较小 | 大（全回滚） |
| **执行时间** | 略长（多次交互） | 短（单次执行） |
| **推荐场景** | 生产环境首选 | 数据初始化/压测 |

```java
@Service
public class SafeDeleteStrategy {
    
    private static final int PAGE_SIZE = 1000;  // 每批 1000 条
    private static final long PAGE_DELAY_MS = 50;  // 批次间隔 50ms
    
    /**
     * 安全分页删除
     * 核心：LIMIT + SLEEP + 监控
     */
    public int pageDelete(String tableName, String whereClause, long maxDeleteCount) {
        int totalDeleted = 0;
        int deleted;
        
        do {
            // 1. 构建分页 SQL
            String sql = String.format("""
                DELETE FROM %s 
                WHERE %s 
                ORDER BY id 
                LIMIT %d
                """, tableName, whereClause, PAGE_SIZE);
            
            // 2. 执行删除
            deleted = jdbcTemplate.update(sql);
            totalDeleted += deleted;
            
            if (deleted == 0) break;
            
            // 3. 批次间隔（让主从复制追上）
            Thread.sleep(PAGE_DELAY_MS);
            
            // 4. 超时保护
            if (totalDeleted >= maxDeleteCount) {
                log.warn("达到最大删除数量{}，停止删除", maxDeleteCount);
                break;
            }
            
        } while (deleted == PAGE_SIZE);  // 有数据继续删
        
        return totalDeleted;
    }
    
    /**
     * 批量删除（谨慎使用）
     * 适用：确定数据量小、无主从延迟的场景
     */
    public int batchDelete(String tableName, String whereClause) {
        String sql = String.format("""
            DELETE FROM %s WHERE %s
            """, tableName, whereClause);
        
        // ⚠️ MySQL 大数据量 DELETE 会锁全表，加条件限制
        return jdbcTemplate.update(sql);
    }
}
```

## 🔥 扩展追问 4：如何监控定时任务的执行状态？

```java
// ========== 任务执行监控 ==========

@Service
public class TaskMonitor {
    
    // 任务执行指标（Prometheus）
    private final Counter taskExecutions = Counter.build()
        .name("scheduled_task_executions_total")
        .labelNames("task_name", "status")
        .register();
    
    private final Histogram taskDuration = Histogram.build()
        .name("scheduled_task_duration_seconds")
        .labelNames("task_name")
        .register();
    
    public void recordExecution(String taskName, boolean success, long durationMs) {
        taskExecutions.labels(taskName, success ? "success" : "failed").inc();
        taskDuration.labels(taskName).observe(durationMs / 1000.0);
        
        // 发送告警
        if (!success) {
            alertService.send("定时任务执行失败: {}", taskName);
        }
    }
    
    // 任务执行拦截器
    @Aspect
    @Component
    public class TaskMonitorAspect {
        
        @Around("@annotation(Scheduled) || @annotation(XxlJob || @annotation(ElasticJob))")
        public Object monitorTask(ProceedingJoinPoint point) throws Throwable {
            String taskName = point.getTarget().getClass().getSimpleName() + "." + 
                              point.getSignature().getName();
            long start = System.currentTimeMillis();
            
            try {
                Object result = point.proceed();
                recordExecution(taskName, true, System.currentTimeMillis() - start);
                return result;
            } catch (Exception e) {
                recordExecution(taskName, false, System.currentTimeMillis() - start);
                throw e;
            }
        }
    }
}
```

```java
// 多 Worker 协同清理
@Service
public class DistributedCleanupCoordinator {
    
    private static final int TOTAL_SHARDS = 1000;  // 总分片数
    private static final int SHARDS_PER_BATCH = 100;  // 每批分配100个分片
    
    // 1. 调度中心：拆分任务，推入队列
    @Scheduled(cron = "0 0 * * * ?")  // 每小时
    public void scheduleCleanupTasks() {
        for (int shard = 0; shard < TOTAL_SHARDS; shard++) {
            String taskKey = String.format("cleanup:task:%d", shard);
            
            // 幂等检查：任务已存在就不重复添加
            if (!redisTemplate.hasKey(taskKey)) {
                CleanupTask task = new CleanupTask(shard, shard);
                redisTemplate.opsForList().leftPush("cleanup:queue", task);
                redisTemplate.expire(taskKey, 2, TimeUnit.HOURS);
            }
        }
    }
    
    // 2. Worker 节点：抢任务执行
    @Scheduled(fixedDelay = 1000)
    public void workerConsumeTasks() {
        // 原子性抢任务（LPOP + LPUSH 实现）
        String taskJson = redisTemplate.opsForList().rightPop("cleanup:queue");
        if (taskJson == null) return;
        
        CleanupTask task = JSON.parseObject(taskJson, CleanupTask.class);
        
        try {
            executeCleanup(task);
            log.info("分片{}清理完成", task.getShardId());
        } catch (Exception e) {
            log.error("分片{}清理失败，重新放回队列", task.getShardId(), e);
            // 失败重试，最多3次
            if (task.getRetryCount() < 3) {
                task.incrementRetry();
                redisTemplate.opsForList().leftPush("cleanup:queue", task);
            } else {
                // 记录失败日志，告警人工处理
                alertService.send("分片{}清理失败，已重试3次", task.getShardId());
            }
        }
    }
}
```

> **总结：定时任务防重叠 → 分布式锁 + 分片 + 分页删除 + 监控告警**
>
> **核心是：同一时刻只有1个节点执行 + 每次只处理一批数据 + 批次间休息 + Misfire 处理 + 全链路监控。**

> "我之前负责过一个历史订单清理任务，数据量 5 个亿。最开始用的单机定时任务，每次要跑 40 分钟，但调度间隔是 30 分钟，结果任务堆积越来越严重。后来改成分片方案：100 个分片，每台机器每次只跑 2 个分片（每分片 500 万），跑完休息 1 分钟再拉下一个。同时加了任务状态 Redis 记录，第二天调度进来时自动从上次中断的地方继续。改完后每次执行时间控制在 20 分钟左右，还留了一半余量。最关键的经验是：**所有定时任务都要加执行时间监控**，一旦接近调度间隔的 80%，立刻告警，提前发现堆积问题。"
