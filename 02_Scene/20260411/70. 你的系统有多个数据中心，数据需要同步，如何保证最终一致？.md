### 实际落地方案：MySQL + Canal + MQ

```java
// 方案：Canal 监听 binlog → MQ 广播 → 各数据中心消费回放

// 1. Canal 配置（主数据中心）
# canal.properties
canal.serverMode = tcp
canal.destinations = data-sync
# 开启 GTID 模式，支持高可用切换
canal.instance.gtidon = true

// 2. Canal Client 订阅并发送 MQ
@Service
public class DataSyncPublisher {
    
    @Autowired
    private CanalConnector canalConnector;
    @Autowired
    private RocketMQTemplate rocketMQTemplate;
    
    public void startSync() {
        canalConnector.connect();
        canalConnector.subscribe(".*\\..*");  // 监听所有库所有表
        
        while (running) {
            Message message = canalConnector.getWithoutAck(100);  // 100ms 超时
            try {
                List<FlatMessage> flatMessages = messageParser.parse(message.getEntries());
                for (FlatMessage flat : flatMessages) {
                    // 发送跨数据中心同步消息
                    SyncMessage syncMsg = buildSyncMessage(flat);
                    // 广播到所有可用区域
                    broadcastToAllDatacenters(syncMsg);
                }
                canalConnector.ack(message.getId());
            } catch (Exception e) {
                canalConnector.ack(message.getId());  // 异常也 ACK，防止堆积
                log.error("Canal 消息解析异常", e);
            }
        }
    }
    
    private SyncMessage buildSyncMessage(FlatMessage flat) {
        return SyncMessage.builder()
            .tableName(flat.getDatabase() + "." + flat.getTableName())
            .pk(flat.getPrimaryKey())
            .eventType(flat.getType())  // INSERT/UPDATE/DELETE
            .beforeData(flat.getBefore())
            .afterData(flat.getAfter())
            .timestamp(flat.getEventTime())
            .gtid(flat.getGtid())
            .build();
    }
}

// 3. 目标数据中心消费并回放
@Service
public class DataSyncConsumer {
    
    @Autowired
    private JdbcTemplate jdbcTemplate;
    
    @RocketMQMessageListener(
        topic = "data-sync",
        consumerGroup = "data-sync-consumer-group",
        messageModel = MessageModel.BROADCASTING  // 广播模式，每个节点都消费
    )
    public void onMessage(SyncMessage syncMsg) {
        log.info("收到数据同步消息: table={}, pk={}, type={}", 
            syncMsg.getTableName(), syncMsg.getPk(), syncMsg.getEventType());
        
        try {
            // 幂等回放：根据 PK + timestamp 防重
            replayToLocal(syncMsg);
        } catch (DuplicateKeyException e) {
            // 幂等处理：主键冲突 = 已同步，跳过
            log.debug("消息重复，忽略: pk={}", syncMsg.getPk());
        }
    }
    
    private void replayToLocal(SyncMessage syncMsg) {
        String sql;
        switch (syncMsg.getEventType()) {
            case "INSERT":
                sql = buildInsertSql(syncMsg);
                break;
            case "UPDATE":
                sql = buildUpdateSql(syncMsg);
                break;
            case "DELETE":
                sql = buildDeleteSql(syncMsg);
                break;
            default:
                return;
        }
        jdbcTemplate.execute(sql);
    }
}
```

```java
@Service
public class TieredSyncStrategy {
    
    /**
     * 根据数据热度选择同步策略
     */
    public SyncStrategy selectStrategy(String tableName, DataHotness hotness) {
        switch (hotness) {
            case HOT:
                // 用户余额、订单状态 → 强同步 + 本地优先
                return SyncStrategy.SYNC;
            case WARM:
                // 商品信息、配置表 → 半同步 + 版本校验
                return SyncStrategy.SEMI_SYNC;
            case COLD:
                // 历史订单、日志 → 异步 + 定时批量
                return SyncStrategy.ASYNC;
        }
    }
}

// 2. 网络抖动时自动降级
@Component
public class SyncHealthChecker {
    
    private volatile SyncMode currentMode = SyncMode.NORMAL;
    
    @Scheduled(fixedDelay = 5000)
    public void checkSyncHealth() {
        long delay = getCurrentSyncDelay();  // 从监控指标获取
        
        if (delay > 5000 && currentMode == SyncMode.NORMAL) {
            log.warn("同步延迟超过5秒，自动降级为异步模式");
            currentMode = SyncMode.DEGRADED;
            notifyAllNodesDegraded();
        } else if (delay < 1000 && currentMode == SyncMode.DEGRADED) {
            log.info("同步延迟恢复正常，切换回同步模式");
            currentMode = SyncMode.NORMAL;
            // 触发一次全量校验
            triggerFullSync();
        }
    }
}

// 3. 兜底：定期全量校验

    log.info("开始每日全量数据校验");
    List<String> tables = config.getSyncTables();
    for (String table : tables) {

    }
```

## 🔥 扩展追问 2：并发修改导致数据冲突怎么解决？

> **A:** 这是多主写入（Active-Active）的核心难题，常见解法：

```
┌─────────────────────────────────────────────────────────────────┐
│                    数据冲突解决策略                               │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  策略1: Last Write Wins (LWW) - 简单粗暴                        │
│  ┌────────────────────────────────────────────────────────────┐ │
│  │ 冲突时，timestamp 最新的覆盖旧值                             │ │
│  │ 优点: 实现简单                                              │ │
│  │ 缺点: 可能丢失数据                                          │ │
│  │ 适用: 日志、指标等允许少量丢失的场景                          │ │
│  └────────────────────────────────────────────────────────────┘ │
│                                                                 │
│  策略2: 版本向量 (Vector Clock) - 精确追踪                        │
│  ┌────────────────────────────────────────────────────────────┐ │
│  │ 每个节点维护版本向量: {node1:3, node2:1, node3:5}          │ │
│  │ 冲突时根据向量关系判断: 父子/兄弟/ 祖孙                       │ │
│  │ 最终由应用层决定如何合并                                     │ │
│  └────────────────────────────────────────────────────────────┘ │
│                                                                 │
│  策略3: CRDT (Conflict-free Replicated Data Types)              │
│  ┌────────────────────────────────────────────────────────────┐ │
│  │ G-Counter: 只增不减的计数器（库存扣减? 不行）                │ │
│  │ PN-Counter: 支持增减的计数器                               │ │
│  │ LWW-Register: 最新写入胜出                                 │ │
│  │ OR-Set: 增删集合                                            │ │
│  └────────────────────────────────────────────────────────────┘ │
│                                                                 │
│  策略4: 业务层合并 - 最可控                                      │
│  ┌────────────────────────────────────────────────────────────┐ │
│  │ 冲突时记录所有版本 → 告警 → 人工/规则处理                    │ │
│  │ 例子: 订单金额冲突 → 保留两笔，用户确认                       │ │
│  │ 例子: 库存冲突 → 取最小值（防止超卖）                        │ │
│  └────────────────────────────────────────────────────────────┘ │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

```java
// 业务层冲突解决示例：订单金额冲突
@Service
public class OrderConflictResolver {
    
    /**
     * 冲突合并策略：金额以最后一次确权操作（如支付成功）为准
     */
    public Order resolveConflict(Order localVersion, Order remoteVersion) {
        // 金额以最高权威（支付系统）为准
        if (remoteVersion.getPaymentStatus() == PaymentStatus.PAID 
            && localVersion.getPaymentStatus() != PaymentStatus.PAID) {
            // 远端已支付，本地未支付 → 以远端为准
            return remoteVersion;
        }
        
        // 时间戳兜底
        if (remoteVersion.getUpdateTime().isAfter(localVersion.getUpdateTime())) {
            return remoteVersion;
        }
        
        return localVersion;
    }
    
    /**
     * 库存冲突：取最小值（保守策略，防止超卖）
     */
    public Integer resolveStockConflict(Integer localStock, Integer remoteStock) {
        return Math.min(localStock, remoteStock);
    }
}
```

## 🔥 扩展追问 3：MySQL 官方的主从复制够用吗？有什么坑？

> **A:** MySQL 原生复制在多数据中心场景有几个坑：

| 问题 | 表现 | 解决方案 |
|------|------|----------|
| **复制延迟累积** | 写入快、拉取慢，延迟几分钟 | 分库分表 + 延迟监控 + 读本地从库 |
| **主从切换丢数据** | GTID 模式可保证不丢，但切换有窗口期 | 半同步复制（至少1个从库ACK） |
| **从库Binlog relay顺序** | 大事务阻塞后续小事务 | 拆大事务 + parallel replication |
| **跨地域复制带宽** | 100G/天数据跨地域同步成本高 | 压缩 + 按需同步（只同步热点表） |
| **脑裂问题** | 两边都认为自己是主 | 引入仲裁（Zookeeper/ETCD） |

```java
/*
# 主库配置
log-bin = mysql-bin
server-id = 1
rpl-semi-sync-master-enabled = 1
rpl-semi-sync-replica-enabled = 1
rpl-semi-sync-sync-timeout = 2000  -- 2秒超时降级为异步

# 从库配置
log-bin = mysql-bin
server-id = 2
relay-log = relay-bin
rpl-semi-sync-replica-enabled = 1
rpl-semi-sync-master-timeout = 2000
*/

// 切换时保证数据不丢
@Service
public class MySQLFailoverHandler {
    
    @Autowired
    private DataSource dataSource;
    
    /**
     * 主库故障切换流程
     */
    public void failover(String fromMaster, String toSlave) {
        // 1. 确认从库追上主库
        ShowReplicaStatusResult status = getReplicaStatus(toSlave);
        while (status.getSecondsBehindMaster() > 0) {
            log.info("等待从库追上: lag={}s", status.getSecondsBehindMaster());
            Thread.sleep(1000);
        }
        
        // 2. 停止主库写入
        setReadOnly(fromMaster);
        
        // 3. 等待从库消费完 relay log
        status = getReplicaStatus(toSlave);
        while (status.getExecMasterLogPos() < status.getReadMasterLogPos()) {
            Thread.sleep(100);
        }
        
        // 4. 提升从库为主库
        stopReplica(toSlave);
        changeMasterToPrimary(toSlave);
        
        // 5. 通知应用层切主
        notifyAppSwitchMaster(toSlave);
    }
}
```

```java
// Redis 双写 + 冲突解决实现
@Service
public class RedisCrossDCCache {
    
    @Autowired
    private StringRedisTemplate redisTemplate;
    
    /**
     * 写入：本地优先 + 异步广播
     */
    public void set(Long userId, String value) {
        // 1. 先写本地 Redis
        String localKey = buildLocalKey(userId);
        redisTemplate.opsForValue().set(localKey, value, Duration.ofHours(1));
        
        // 2. 记录操作日志（用于冲突解决）
        String logKey = "sync:log:" + userId + ":" + System.currentTimeMillis();
        redisTemplate.opsForHash().putAll(logKey, Map.of(
            "op", "SET",
            "value", value,
            "timestamp", String.valueOf(System.currentTimeMillis()),
            "dc", localDataCenter  // 北京/上海
        ));
        
        // 3. 发送 MQ 广播（最终一致）
        mqTemplate.convertAndSend("cache-sync-topic", 
            new CacheSyncEvent(userId, "SET", value, localDataCenter));
    }
    
    /**
     * 读取：本地优先，过期去远端拉取
     */
    public String get(Long userId) {
        String localKey = buildLocalKey(userId);
        String value = redisTemplate.opsForValue().get(localKey);
        
        if (value != null) {
            return value;
        }
        
        // 本地未命中，去远端拉取并回填
        value = fetchFromRemoteDC(userId);
        if (value != null) {
            redisTemplate.opsForValue().set(localKey, value, Duration.ofMinutes(30));
        }
        
        return value;
    }
    
    /**
     * 冲突解决：LWW + 业务兜底
     */
    public void resolveConflict(String userId, List<CacheSyncEvent> events) {
        // 按时间戳排序，取最新
        events.sort(Comparator.comparing(CacheSyncEvent::getTimestamp).reversed());
        CacheSyncEvent latest = events.get(0);
        
        // 以最新为准，覆盖本地
        String localKey = buildLocalKey(userId);
        redisTemplate.opsForValue().set(localKey, latest.getValue());
    }
}
```

## 🔥 扩展追问 5：如何验证数据最终一致性？

> **A:** 主动校验 + 被动监控双管齐下：

```java
// 1. 定时 checksum 校验
@Service
public class DataConsistencyChecker {
    
    @Scheduled(fixedDelay = 600000)  // 每10分钟
    public void checkConsistency() {
        List<String> tables = getCriticalTables();
        
        for (String table : tables) {
            // 每个数据中心计算 checksum
            Map<String, String> dcChecksums = new HashMap<>();
            for (String dc : allDatacenters) {
                String checksum = calculateTableChecksum(dc, table);
                dcChecksums.put(dc, checksum);
            }
            
            // 对比所有数据中心的 checksum
            Set<String> uniqueChecksums = new HashSet<>(dcChecksums.values());
            if (uniqueChecksums.size() > 1) {
                log.error("数据不一致! table={}, checksums={}", table, dcChecksums);
                alertAndRepair(table, dcChecksums);
            } else {
                log.debug("数据一致: table={}", table);
            }
        }
    }
    
    private String calculateTableChecksum(String dc, String table) {
        // 使用 MySQL CHECKSUM TABLE 或自己计算行数+hash
        String sql = String.format(
            "SELECT COUNT(*) as cnt, GROUP_CONCAT(id, '-', ABS(MD5(CONCAT_ws(',', *))) " +
            "ORDER BY id SEPARATOR ',') as hash FROM %s", table);
        
        // 根据 dc 路由到对应数据源
        return jdbcTemplate.queryForObject(sql, String.class);
    }
}

// 2. 消费端回放计数监控
@Service
public class SyncMonitor {
    
    private ConcurrentHashMap<String, AtomicLong> sequenceMap = new ConcurrentHashMap<>();
    
    // 消费消息时记录序列号
    public void onSyncMessage(SyncMessage msg) {
        String key = msg.getSourceDC() + "-" + msg.getTableName();
        long seq = msg.getSequence();
        sequenceMap.put(key, new AtomicLong(seq));
    }
    
    // 监控：各数据中心序列号差值
    @Scheduled(fixedDelay = 30000)
    public void checkSequenceGap() {
        for (String key : sequenceMap.keySet()) {
            long minSeq = sequenceMap.values().stream()
                .mapToLong(AtomicLong::get).min().orElse(0);
            long maxSeq = sequenceMap.values().stream()
                .mapToLong(AtomicLong::get).max().orElse(0);
            
            if (maxSeq - minSeq > 100) {
                log.warn("同步延迟告警: key={}, gap={}", key, maxSeq - minSeq);
                alertSyncDelay(key, maxSeq - minSeq);
            }
        }
    }
}
```

> "我们之前做过两地三中心的改造，主数据中心在北京，灾备在上海。最开始用的异步复制，结果有一次网络抖动导致上海的数据延迟了 15 分钟才同步过去，中间用户反馈订单状态不对。后来我们加了分层策略：用户余额、订单状态这些热点数据用半同步复制，距离超过 100ms 就自动降级但不丢失；商品详情、配置这些冷数据用纯异步。同时每天凌晨 3 点做一次全量 checksum 校验，7 个月下来数据一致率是 100%。这个改造的经验是：没有银弹，必须根据数据热度分级处理。"
