```java
// ========== 1. 统一通知入口 ==========

@Service
public class NotificationFacade {
    
    @Autowired
    private NotificationDispatcher dispatcher;
    @Autowired
    private TemplateRenderer templateRenderer;
    
    /**
     * 发送通知（同步返回结果）
     */
    public NotifyResult send(NotifyRequest request) {
        // 1. 参数校验
        validateRequest(request);
        
        // 2. 加载模板
        NotifyTemplate template = loadTemplate(request.getTemplateCode());
        
        // 3. 渲染消息
        RenderedMessage rendered = templateRenderer.render(template, request.getParams());
        
        // 4. 发送
        return dispatcher.sendSync(request.getUserId(), template, rendered);
    }
    
    /**
     * 发送通知（异步，推荐）
     */
    public void sendAsync(NotifyRequest request) {
        // 写入 MQ，异步处理
        mqTemplate.convertAndSend("notify.dispatch.queue", request);
    }
    
    /**
     * 批量发送（营销场景）
     */
    public void batchSend(BatchNotifyRequest request) {
        // 分片处理，每批 1000
        List<List<Long>> userIdBatches = Lists.partition(request.getUserIds(), 1000);
        
        for (List<Long> batch : userIdBatches) {
            BatchNotifyMessage msg = new BatchNotifyMessage(
                request.getTemplateCode(), 
                batch, 
                request.getParams());
            
            mqTemplate.convertAndSend("notify.batch.queue", msg);
        }
    }
    
    /**
     * 定时发送
     */
    public void scheduleSend(NotifyRequest request, LocalDateTime scheduleTime) {
        // 先渲染，存储渲染后的内容
        NotifyTemplate template = loadTemplate(request.getTemplateCode());
        RenderedMessage rendered = templateRenderer.render(template, request.getParams());
        
        // 写入定时表
        notifyScheduleMapper.insert(NotifySchedule.builder()
            .messageId(generateMessageId())
            .userId(request.getUserId())
            .channel(request.getChannel().getCode())
            .scheduleTime(scheduleTime)
            .status(0)
            .build());
    }
}

// ========== 2. 消息分发中心 ==========

@Service
public class NotificationDispatcher {
    
    @Autowired
    private UserPreferenceMapper preferenceMapper;
    @Autowired
    private NotifyMessageMapper messageMapper;
    @Autowired
    private ChannelSenderFactory senderFactory;
    
    /**
     * 同步分发（根据用户偏好拆分为多个渠道）
     */
    public NotifyResult sendSync(Long userId, NotifyTemplate template, 
                                  RenderedMessage rendered) {
        // 1. 查询用户通知偏好
        UserNotifyPreference pref = getUserPreference(userId);
        
        // 2. 根据偏好确定发送渠道
        List<NotifyChannel> enabledChannels = resolveChannels(template, pref);
        
        if (enabledChannels.isEmpty()) {
            return NotifyResult.fail("用户未开启任何可用通知渠道");
        }
        
        // 3. 逐个渠道发送
        NotifyResult finalResult = NotifyResult.success();
        for (NotifyChannel channel : enabledChannels) {
            NotifyResult result = sendToChannel(userId, template, rendered, channel);
            if (!result.isSuccess()) {
                finalResult.addError(channel.getName(), result.getMessage());
            }
        }
        
        return finalResult;
    }
    
    /**
     * 异步分发（写入 MQ）
     */
    public void dispatch(NotifyRequest request) {
        // 查询用户偏好
        UserNotifyPreference pref = getUserPreference(request.getUserId());
        NotifyTemplate template = loadTemplate(request.getTemplateCode());
        
        // 根据渠道拆分任务
        for (NotifyChannel channel : NotifyChannel.values()) {
            if (shouldSend(channel, template, pref)) {
                // 为每个渠道生成一条 MQ 消息
                DispatchTask task = DispatchTask.builder()
                    .messageId(generateMessageId())
                    .userId(request.getUserId())
                    .channel(channel)
                    .templateCode(request.getTemplateCode())
                    .params(request.getParams())
                    .priority(template.getPriority())
                    .build();
                
                // 根据优先级发送到不同队列
                sendToQueue(channel, task, template.getPriority());
            }
        }
    }
    
    private List<NotifyChannel> resolveChannels(NotifyTemplate template, 
                                                UserNotifyPreference pref) {
        List<NotifyChannel> channels = new ArrayList<>();
        
        // 检查各渠道是否启用
        if (pref.isLetterEnabled() && template.supportsChannel(NotifyChannel.LETTER)) {
            channels.add(NotifyChannel.LETTER);
        }
        if (pref.isSmsEnabled() && template.supportsChannel(NotifyChannel.SMS)) {
            channels.add(NotifyChannel.SMS);
        }
        if (pref.isEmailEnabled() && template.supportsChannel(NotifyChannel.EMAIL)) {
            channels.add(NotifyChannel.EMAIL);
        }
        if (pref.isPushEnabled() && template.supportsChannel(NotifyChannel.PUSH)) {
            channels.add(NotifyChannel.PUSH);
        }
        
        return channels;
    }
    
    private boolean shouldSend(NotifyChannel channel, NotifyTemplate template,
                               UserNotifyPreference pref) {
        // 模板不支持该渠道
        if (!template.supportsChannel(channel)) {
            return false;
        }
        
        // 用户未开启该渠道
        switch (channel) {
            case LETTER: return pref.isLetterEnabled();
            case SMS: return pref.isSmsEnabled();
            case EMAIL: return pref.isEmailEnabled();
            case PUSH: return pref.isPushEnabled();
        }
        
        return false;
    }
}

@Component
public class ChannelSenderFactory {
    
    private final Map<NotifyChannel, ChannelSender> senders = new ConcurrentHashMap<>();
    
    @Autowired
    private LetterSender letterSender;
    @Autowired
    private SmsSender smsSender;
    @Autowired
    private EmailSender emailSender;
    @Autowired
    private PushSender pushSender;
    
    @PostConstruct
    public void init() {
        senders.put(NotifyChannel.LETTER, letterSender);
        senders.put(NotifyChannel.SMS, smsSender);
        senders.put(NotifyChannel.EMAIL, emailSender);
        senders.put(NotifyChannel.PUSH, pushSender);
    }
    
    public ChannelSender getSender(NotifyChannel channel) {
        return senders.get(channel);
    }
}

@Component
public class SmsSender implements ChannelSender {
    
    @Autowired
    private AliyunSmsClient smsClient;
    
    @Override
    public SendResult send(DispatchContext context) {
        try {
            // 调用阿里云短信
            SendSmsRequest request = new SendSmsRequest();
            request.setPhoneNumbers(context.getPhone());
            request.setSignName("您的应用名");
            request.setTemplateCode(context.getTemplate().getSmsTemplateId());
            request.setTemplateParam(JSON.toJSONString(context.getParams()));
            
            SendSmsResponse response = smsClient.getAcsResponse(request);
            
            if ("OK".equals(response.getCode())) {
                return SendResult.success(response.getBizId());
            } else {
                return SendResult.fail(response.getMessage());
            }
        } catch (Exception e) {
            return SendResult.fail(e.getMessage());
        }
    }
}

@Component
public class LetterSender implements ChannelSender {
    
    @Autowired
    private NotifyMessageMapper messageMapper;
    
    @Override
    public SendResult send(DispatchContext context) {
        // 站内信直接入库
        NotifyMessage message = NotifyMessage.builder()
            .messageId(context.getMessageId())
            .userId(context.getUserId())
            .userName(context.getUserName())
            .title(context.getTitle())
            .content(context.getContent())
            .channel(NotifyChannel.LETTER.getCode())
            .sendStatus(2)  // 直接成功
            .sendTime(new Date())
            .readStatus(0)
            .build();
        
        messageMapper.insert(message);
        
        // 通知用户有新消息（WebSocket 实时推送）
        webSocketServer.sendToUser(context.getUserId(), "new_message", message.getId());
        
        return SendResult.success(message.getId().toString());
    }
}

@Component
public class EmailSender implements ChannelSender {
    
    @Autowired
    private JavaMailSender mailSender;
    
    @Override
    public SendResult send(DispatchContext context) {
        try {
            MimeMessage message = mailSender.createMimeMessage();
            MimeMessageHelper helper = new MimeMessageHelper(message, true, "UTF-8");
            
            helper.setFrom("noreply@example.com");
            helper.setTo(context.getEmail());
            helper.setSubject(context.getTitle());
            helper.setText(context.getContent(), true);  // HTML 格式
            
            mailSender.send(message);
            
            return SendResult.success();
        } catch (Exception e) {
            return SendResult.fail(e.getMessage());
        }
    }
}

@Component
public class PushSender implements ChannelSender {
    
    @Autowired
    private JiguangPushClient jiguangClient;
    
    @Override
    public SendResult send(DispatchContext context) {
        try {
            // 构建推送消息

                .setPlatform(Platform.all())
                .setAudience(Audience.registrationId(context.getDeviceTokens()))
                .setNotification(Notification.newBuilder()

                        .setTitle(context.getTitle())

                        .build())
                    .build())
                .build();
            
            PushResult result = jiguangClient.send(payload);
            
            if (result.isSuccess()) {
                return SendResult.success(result.getMsgId());
            } else {
                return SendResult.fail(result.getErrorMsg());
            }
        } catch (Exception e) {
            return SendResult.fail(e.getMessage());
        }
    }

@Component
@Slf4j
public class NotifyMessageConsumer {
    
    @Autowired
    private ChannelSenderFactory senderFactory;
    @Autowired
    private NotifyMessageMapper messageMapper;
    
    @RabbitListener(queues = "notify.letter.queue")
    public void consumeLetter(DispatchTask task, Channel channel) {
        processMessage(task, NotifyChannel.LETTER);
    }
    
    @RabbitListener(queues = "notify.sms.queue")
    public void consumeSms(DispatchTask task, Channel channel) {
        processMessage(task, NotifyChannel.SMS);
    }
    
    @RabbitListener(queues = "notify.email.queue")
    public void consumeEmail(DispatchTask task, Channel channel) {
        processMessage(task, NotifyChannel.EMAIL);
    }
    
    private void processMessage(DispatchTask task, NotifyChannel channel) {
        log.info("处理通知消息: messageId={}, channel={}", task.getMessageId(), channel);
        
        // 1. 查询消息详情
        NotifyMessage dbMessage = messageMapper.selectByMessageId(task.getMessageId());
        if (dbMessage == null) {
            log.error("消息不存在: {}", task.getMessageId());
            return;
        }
        
        // 2. 状态检查（幂等）
        if (dbMessage.getSendStatus() == 2) {
            log.info("消息已发送，跳过: {}", task.getMessageId());
            return;
        }
        
        // 3. 发送
        DispatchContext context = buildContext(task, dbMessage);
        ChannelSender sender = senderFactory.getSender(channel);
        SendResult result = sender.send(context);
        
        // 4. 更新状态
        if (result.isSuccess()) {
            messageMapper.updateSendSuccess(task.getMessageId(), result.getExternalId());
        } else {
            handleSendFailure(task, dbMessage, result.getMessage());
        }
    }
    
    private void handleSendFailure(DispatchTask task, NotifyMessage dbMessage, 
                                   String errorMsg) {
        int retryCount = dbMessage.getRetryCount() + 1;
        
        if (retryCount < dbMessage.getMaxRetry()) {
            // 重试
            messageMapper.updateRetry(task.getMessageId(), retryCount, errorMsg);
            
            // 延迟重试（指数退避）
            long delay = (long) Math.pow(2, retryCount) * 1000;
            rabbitTemplate.convertAndSend("notify.retry.queue", task, msg -> {
                msg.getMessageProperties().setDelay((int) delay);
                return msg;
            });
            
            log.warn("消息发送失败，进入重试队列: messageId={}, retry={}, delay={}ms",
                task.getMessageId(), retryCount, delay);
        } else {
            // 超过重试次数，标记失败
            messageMapper.updateSendFailed(task.getMessageId(), errorMsg);
            
            // 告警
            alertService.send("消息发送失败", 
                String.format("messageId=%s, error=%s", task.getMessageId(), errorMsg));
        }
    }
}
```

## 🔥 扩展追问 1：如何保证消息不丢失？

> **A:** 消息不丢失需要三层保证：

```
┌─────────────────────────────────────────────────────────────────┐
│              消息不丢失三保险                                    │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│   保险1: MQ 持久化                                              │
│   ┌─────────────────────────────────────────────────────────┐  │
│   │ publisher:                                                 │  │
│   │ channel.queueDeclare("notify.letter", durable=true)      │  │
│   │ basicPublish(..., mandatory=true)                        │  │
│   │                                                          │  │
│   │ broker:                                                   │  │
│   │ - durable=true: 队列持久化                               │  │
│   │ - deliveryMode=2: 消息持久化                              │  │
│   └─────────────────────────────────────────────────────────┘  │
│                                                                 │
│   保险2: 消费确认 (ACK)                                         │
│   ┌─────────────────────────────────────────────────────────┐  │
│   │ - 手动 ACK: 消费成功后再 ACK                            │  │
│   │ - 消费失败: NACK → requeue → 稍后重试                  │  │
│   │                                                          │  │
│   │ @RabbitListener ACK 模式:                               │  │
│   │ - basicAck(deliveryTag, multiple) → 确认成功             │  │
│   │ - basicNack(deliveryTag, requeue=true) → 重新入队       │  │
│   └─────────────────────────────────────────────────────────┘  │
│                                                                 │
│   保险3: 消息表落库                                            │
```

```java
// ========== 消息不丢失实现 ==========

@Service
public class ReliableNotificationService {
    
    @Autowired
    private NotifyMessageMapper messageMapper;
    
    /**
     * 可靠发送：先落库，再发送
     */
    public void reliableSend(NotifyRequest request) {
        // 1. 消息入库（状态=待发送）
        NotifyMessage message = NotifyMessage.builder()
            .messageId(generateMessageId())
            .userId(request.getUserId())
            .templateCode(request.getTemplateCode())
            .title(request.getTitle())
            .content(request.getContent())
            .channel(request.getChannel().getCode())
            .sendStatus(0)  // 待发送
            .retryCount(0)
            .maxRetry(3)
            .build();
        
        messageMapper.insert(message);
        
        // 2. 发送 MQ
        try {
            mqTemplate.convertAndSend("notify.queue", message);
        } catch (Exception e) {
            // MQ 发送失败，更新状态
            messageMapper.updateSendFailed(message.getMessageId(), e.getMessage());
            throw e;
        }
        
        // 3. 发送成功，更新状态
        messageMapper.updateSendSuccess(message.getMessageId(), null);
    }
    
    /**
     * 兜底：定时扫描超时未发送的消息
     */
    @Scheduled(fixedDelay = 60000)
    public void scanPendingMessages() {
        // 查询超过 5 分钟仍未发送的消息
        List<NotifyMessage> pending = messageMapper
            .selectPendingMessages(5, TimeUnit.MINUTES.toSeconds(5));
        
        for (NotifyMessage msg : pending) {
            // 重新发送
            log.warn("发现超时未发送消息，重新投递: {}", msg.getMessageId());
            mqTemplate.convertAndSend("notify.queue", msg);
        }
    }
}
```

```java
@Service
public class IdempotentNotificationService {
    
    /**
     * 幂等发送：基于 messageId 去重
     */
    public SendResult sendWithIdempotency(NotifyRequest request) {
        // 1. 生成幂等 key
        String idempotentKey = generateIdempotentKey(request);
        
        // 2. Redis 检查是否已发送
        Boolean exists = redisTemplate.hasKey("notify:idempotent:" + idempotentKey);
        
        if (Boolean.TRUE.equals(exists)) {
            log.info("消息已发送，幂等跳过: key={}", idempotentKey);
            return SendResult.success("重复发送，已忽略");
        }
        
        // 3. 发送
        SendResult result = doSend(request);
        
        // 4. 发送成功，写入 Redis（TTL = 24小时）
        if (result.isSuccess()) {
            redisTemplate.opsForValue().set(
                "notify:idempotent:" + idempotentKey, 
                "1", 
                Duration.ofHours(24));
        }
        
        return result;
    }
    
    /**
     * 生成幂等 key
     * 规则：bizType + userId + bizId 组合
     */
    private String generateIdempotentKey(NotifyRequest request) {
        return String.format("%s:%d:%s",
            request.getBizType(),
            request.getUserId(),
            request.getBizId());
    }
}

// MQ 消费者幂等处理
@RabbitListener(queues = "notify.letter.queue")
public void consumeWithIdempotency(NotifyMessage message, Channel channel, 
                                   @Header(AmqpHeaders.DELIVERY_TAG) long tag) {
    try {
        // 幂等检查
        if (isAlreadyProcessed(message.getMessageId())) {
            channel.basicAck(tag, false);  // 已处理，直接 ACK
            return;
        }
        
        // 发送
        doSend(message);
        
        // 标记已处理
        markAsProcessed(message.getMessageId());
        
        channel.basicAck(tag, false);  // 消费成功 ACK
        
    } catch (Exception e) {
        channel.basicNack(tag, false, true);  // 失败重试
    }
}
```

```java
@Service
public class NotificationStrategyService {
    
    /**
     * 灰度发送策略
     */
    public void sendWithStrategy(NotifyRequest request, SendStrategy strategy) {
        switch (strategy.getType()) {
            case USER_PERCENT:
                // 百分比灰度
                int percent = MurmurHash3.murmurhash3AsInt(
                    request.getUserId().toString(), 0);
                if (percent % 100 >= strategy.getPercent()) {
                    log.info("用户不在灰度范围内: userId={}", request.getUserId());
                    return;
                }
                break;
                
            case USER_TAG:
                // 标签灰度
                UserTag tag = getUserTag(request.getUserId());
                if (!strategy.getTags().contains(tag)) {
                    return;
                }
                break;
                
            case TIME_RANGE:
                // 定时灰度
                if (!isInTimeRange(strategy.getStartTime(), strategy.getEndTime())) {
                    return;
                }
                break;
        }
        
        // 发送
        sendAsync(request);
    }
    
    /**
     * A/B 测试
     */
    public void sendWithABTest(NotifyRequest request, String templateA, 
                               String templateB) {
        // 按用户 ID 哈希决定组别
        int bucket = MurmurHash3.murmurhash3AsInt(
            request.getUserId().toString(), 0) % 2;
        
        String templateCode = bucket == 0 ? templateA : templateB;
        request.setTemplateCode(templateCode);
        
        sendAsync(request);
        
        // 记录实验分组（用于后续分析）
        recordABTest(request.getUserId(), templateA, templateB, bucket);
    }
```

## 🔥 扩展追问 4：如何处理免打扰时段？

> **A:** 免打扰时段的处理策略：

```java
@Service
public class QuietHoursService {
    
    /**
     * 检查是否在免打扰时段
     */
    public boolean isQuietHours(UserNotifyPreference pref) {
        if (pref.getQuietHoursStart() == null || pref.getQuietHoursEnd() == null) {
            return false;
        }
        
        LocalTime now = LocalTime.now();
        LocalTime start = pref.getQuietHoursStart().toLocalTime();
        LocalTime end = pref.getQuietHoursEnd().toLocalTime();
        
        // 跨午夜的情况
        if (start.isAfter(end)) {
            // 22:00 - 08:00 这种情况
            return now.isAfter(start) || now.isBefore(end);
        } else {
            return now.isAfter(start) && now.isBefore(end);
        }
    }
    
    /**
     * 处理免打扰：延迟到免打扰结束后发送
     */
    public void sendWithQuietHours(NotifyRequest request, UserNotifyPreference pref) {
        if (!isQuietHours(pref)) {
            // 不在免打扰时段，直接发送
            sendAsync(request);
            return;
        }
        
        // 计算延迟时间
        LocalDateTime sendTime = LocalDateTime.now()
            .toLocalDate().atTime(pref.getQuietHoursEnd());
        
        // 防止延迟到第二天
        if (sendTime.isBefore(LocalDateTime.now())) {
            sendTime = sendTime.plusDays(1);
        }
        
        // 定时发送
        scheduleSend(request, sendTime);
    }
    
    /**
     * 紧急消息：绕过免打扰
     */
    public void sendUrgent(NotifyRequest request) {
        // 紧急消息强制发送（短信/电话）
        if (request.isForceSend()) {
            sendSync(request);
        } else {
            // 非强制紧急，优先推送站内信
            request.setChannel(NotifyChannel.LETTER);
            sendSync(request);
        }
    }
}
```

```java
     * 记录点击/打开

            callback.getMessageId(), 
            callback.getClickTime(),
            callback.getClickUrl());

            "messageId", callback.getMessageId(),
            "userId", callback.getUserId(),
            "channel", callback.getChannel()
        ));
    }

     * 统计报表
     */
    public NotifyReport generateReport(Date startDate, Date endDate) {
        // 各渠道发送量
        Map<Integer, Long> sendCount = statisticsMapper.countByChannel(startDate, endDate);
        
        // 各渠道送达量
        Map<Integer, Long> deliveredCount = statisticsMapper.countDeliveredByChannel(startDate, endDate);
        
        // 各渠道点击量
        Map<Integer, Long> clickCount = statisticsMapper.countClickByChannel(startDate, endDate);
        
        // 计算转化率
        Map<Integer, NotifyMetrics> metrics = new HashMap<>();
        for (Integer channel : sendCount.keySet()) {
            long sent = sendCount.getOrDefault(channel, 0L);
            long delivered = deliveredCount.getOrDefault(channel, 0L);
            long clicked = clickCount.getOrDefault(channel, 0L);
            
            metrics.put(channel, NotifyMetrics.builder()
                .sendCount(sent)
                .deliveredCount(delivered)
                .clickCount(clicked)
                .deliveryRate(delivered * 100.0 / sent)
                .clickRate(clicked * 100.0 / delivered)
                .build());
        }
        
        return NotifyReport.builder()
            .startDate(startDate)
            .endDate(endDate)
            .metrics(metrics)
            .build();
    }
}
```
