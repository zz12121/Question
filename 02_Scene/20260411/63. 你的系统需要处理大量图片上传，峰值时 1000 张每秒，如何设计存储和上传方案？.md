```java
// ========== 1. 获取上传凭证（服务端） ==========
@Service
public class OssUploadService {
    
    @Autowired
    private AliOSSProperties ossProperties;
    
    /**
     * 获取直传授权（客户端拿这个直接上传到 OSS）
     * 这样上传不走业务服务器，减轻压力
     */
    public OssPolicyVO getUploadPolicy(String bucket) {
        OSS oss = new OSSClientBuilder().build(
            ossProperties.getEndpoint(),
            ossProperties.getAccessKeyId(),
            ossProperties.getAccessKeySecret()
        );
        
        // 生成唯一文件名
        String key = generateFileKey(".jpg");
        
        // 创建 PostObject 请求的签名
        Date expiration = new Date(System.currentTimeMillis() + 3600 * 1000);
        PolicyConditions conditions = new PolicyConditions();
        conditions.addConditionItem(
            PolicyConditions.COND_CONTENT_LENGTH_RANGE, 
            0, 
            10 * 1024 * 1024  // 最大 10MB
        );
        
        String policy = oss.generatePostPolicy(expiration, conditions);
        String signature = oss.calculatePostSignature(policy);
        
        return OssPolicyVO.builder()
            .accessKeyId(ossProperties.getAccessKeyId())
            .policy(policy)
            .signature(signature)
            .key(key)
            .host("https://" + bucket + "." + ossProperties.getEndpoint())
            .expireTime(expiration.getTime())
            .build();
    }
    
    /**
     * 生成唯一文件路径
     * 格式: images/2026/04/14/yyyyMMddHHmmss_UUID.jpg
     */
    private String generateFileKey(String suffix) {
        LocalDate now = LocalDate.now();
        String uuid = UUID.randomUUID().toString().replace("-", "");
        return String.format("images/%d/%02d/%02d/%s_%s%s",
            now.getYear(), now.getMonthValue(), now.getDayOfMonth(),
            DateUtil.format(new Date(), "yyyyMMddHHmmss"),
            uuid,
            suffix
        );
    }
}

// ========== 2. 前端直传 OSS（不上传到后端！） ==========
// 前端代码示例
async function uploadToOSS(file) {
    // 1. 获取上传凭证
    const policy = await api.getUploadPolicy();
    
    // 2. 直接上传到 OSS
    const formData = new FormData();
    formData.append('key', policy.key);
    formData.append('OSSAccessKeyId', policy.accessKeyId);
    formData.append('policy', policy.policy);
    formData.append('signature', policy.signature);
    formData.append('file', file);
    
    await axios.post(policy.host, formData);
    
    // 3. 上传成功后拿 key 回调业务接口
    await api.saveImageMeta({
        ossKey: policy.key,
        originalName: file.name,
        size: file.size
    });
}
```

```java
// ========== 1. 初始化分片上传 ==========
@Service
public class MultipartUploadService {
    
    @Autowired
    private OSS ossClient;
    
    /**
     * 1. 初始化分片上传
     */
    public InitMultipartVO initMultipartUpload(String fileName, String contentType) {
        String uploadId = UUID.randomUUID().toString();
        String key = generateFileKey(getExtension(fileName));
        
        InitiateMultipartUploadRequest request = new InitiateMultipartUploadRequest(
            bucketName, key);
        request.setObjectMetadata(new ObjectMetadata());
        request.getObjectMetadata().setContentType(contentType);
        
        InitiateMultipartUploadResult result = ossClient.initiateMultipartUpload(request);
        String objectKey = result.getKey();
        String uploadId2 = result.getUploadId();
        
        // 保存上传ID和分片状态到 Redis
        String redisKey = "upload:" + uploadId2;
        redisTemplate.opsForHash().putAll(redisKey, Map.of(
            "objectKey", objectKey,
            "fileName", fileName,
            "uploadedParts", "0"  // 已上传分片数
        ));
        redisTemplate.expire(redisKey, Duration.ofHours(24));
        
        return new InitMultipartVO(objectKey, uploadId2);
    }
    
    /**
     * 2. 获取分片上传地址（签名 URL）
     */
    public String getPartUploadUrl(String objectKey, String uploadId, int partNumber) {
        Date expiration = new Date(System.currentTimeMillis() + 3600 * 1000);
        
        // 生成带签名的上传 URL，客户端直接上传到这个地址
        return ossClient.generatePresignedUrl(
            bucketName, objectKey, expiration,
            HttpMethod.PUT
        ).toString();
    }
    
    /**
     * 3. 上传完成，合并分片
     */
    public CompleteMultipartUploadResult completeMultipartUpload(
            String objectKey, String uploadId, List<PartETag> partETags) {
        
        CompleteMultipartUploadRequest request = 
            new CompleteMultipartUploadRequest(bucketName, objectKey, uploadId, partETags);
        
        CompleteMultipartUploadResult result = 
            ossClient.completeMultipartUpload(request);
        
        // 清理 Redis 状态
        redisTemplate.delete("upload:" + uploadId);
        
        return result;
    }
    
    /**
     * 4. 查询已上传的分片（断点续传用）
     */
    public List<Integer> listUploadedParts(String objectKey, String uploadId) {
        ListPartsRequest request = new ListPartsRequest(bucketName, objectKey, uploadId);
        PartListing result = ossClient.listParts(request);
        
        return result.getParts().stream()
            .map(PartSummary::getPartNumber)
            .collect(Collectors.toList());
    }
}

// ========== 2. 前端分片上传 ==========
class MultipartUploader {
    constructor(file, chunkSize = 4 * 1024 * 1024) {  // 4MB
        this.file = file;
        this.chunkSize = chunkSize;
        this.chunks = Math.ceil(file.size / chunkSize);
        this.uploadedParts = [];
    }
    
    async upload(onProgress) {
        // 1. 初始化
        const { objectKey, uploadId } = await api.initMultipartUpload(this.file.name);
        
        // 2. 并发上传分片
        const uploadPromises = [];
        for (let i = 0; i < this.chunks; i++) {
            uploadPromises.push(this.uploadChunk(i, objectKey, uploadId));
        }
        
        await Promise.all(uploadPromises);
        
        // 3. 合并
        await api.completeMultipartUpload(objectKey, uploadId, this.uploadedParts);
        
        return objectKey;
    }
    
    async uploadChunk(index, objectKey, uploadId) {
        const start = index * this.chunkSize;
        const end = Math.min(start + this.chunkSize, this.file.size);
        const chunk = this.file.slice(start, end);
        
        // 获取签名 URL
        const url = await api.getPartUploadUrl(objectKey, uploadId, index + 1);
        
        // 上传到 OSS
        await axios.put(url, chunk, {
            headers: { 'Content-Type': 'application/octet-stream' }
        });
        
        this.uploadedParts.push({
            partNumber: index + 1,
            etag: response.headers.etag  // 用于合并
        });
        
        onProgress(this.uploadedParts.length / this.chunks * 100);
    }
}
```

```java
        ImageIO.write(thumbnail, "jpg", baos);

            new ByteArrayInputStream(baos.toByteArray()));
```

```java
@Service
public class UploadRateLimiter {
    
    // 用户维度限流：每人每秒最多 5 次请求上传凭证
    private RateLimiter userRateLimiter = RateLimiter.create(5);  // 5 QPS
    
    // 全局限流：整体 2000 QPS（预留一些余量）
    private RateLimiter globalRateLimiter = RateLimiter.create(2000);

        if (!userRateLimiter.tryAcquire(userId.toString())) {

        }
        
        // 2. 全局限流
        if (!globalRateLimiter.tryAcquire()) {
            throw new BizException("系统繁忙，请稍后再试");
        }
        
        return getUploadPolicy();
    }
}

@Configuration

            throw new BizException("存储空间不足");
        }
    }
```

## 🔥 扩展追问 3：如何防止恶意上传和图片安全问题？

```java
// ========== 1. 文件头校验（防止伪装文件） ==========
@Service
public class FileSecurityService {
    
    // 图片魔数（文件头）
    private static final Map<String, String[]> IMAGE_SIGNATURES = Map.of(
        "jpg", new String[]{"FFD8FFE0", "FFD8FFE1", "FFD8FFE8"},
        "png", new String[]{"89504E47"},
        "gif", new String[]{"47494638"}
    );
    
    public boolean validateImageContent(byte[] bytes, String contentType) {
        String header = bytesToHex(bytes, 8);
        
        // 魔数校验
        String[] signatures = IMAGE_SIGNATURES.get(contentType.split("/")[1]);
        if (signatures != null) {
            boolean match = Arrays.stream(signatures)
                .anyMatch(s -> header.startsWith(s));
            if (!match) {
                throw new BizException("文件内容与类型不符");
            }
        }
        
        return true;
    }
```

```java
        String webpKey = sourceKey.replace(".jpg", ".webp");
        
        ProcessObjectRequest request = new ProcessObjectRequest(
            bucketName, sourceKey, webpKey);
        request.setProcess("image/resize,w_500/format,webp/quality,q_80");
        
        ossClient.processObject(request);
    }
    
    /**
     * 2. 按访问频率自动迁移存储类型
     */
    @Scheduled(cron = "0 0 3 * * ?")  // 每天凌晨 3 点
    public void optimizeStorageTier() {
        // 统计访问量，自动迁移
        List<Image> coldImages = imageMapper.selectColdImages(180);
        for (Image img : coldImages) {
            ossClient.setObjectStorageClass(bucketName, img.getOssKey(), 
                StorageClass.Archive);
        }
    }
    
    /**
     * 3. 删除过期文件
     */
    @Scheduled(cron = "0 0 4 * * ?")
    public void cleanupExpiredImages() {
        List<Image> expiredImages = imageMapper.selectExpired(LocalDateTime.now().minusDays(30));
        for (Image img : expiredImages) {
            ossClient.deleteObject(bucketName, img.getOssKey());
            imageMapper.deleteById(img.getId());
        }
    }
}
```

## 一句话总结

> **高并发图片上传核心是"客户端直传 OSS + 服务端授信 + 异步消息处理 + CDN 分发"：前端拿服务端签名直传 OSS 不经过业务服务器，分片上传支持断点续传，上传成功后发 MQ 异步转码生成缩略图，CDN 缓存热数据，OSS 开启生命周期规则自动冷热分层存储，多层限流防刷 + 文件头校验防恶意上传。**

**面试加分话术：**

> "之前公司做秒杀活动，用户需要上传头像，峰值时 QPS 接近 2000，一开始图片都上传到 Nginx 再转存到 OSS，结果 Nginx 所在的服务器磁盘 IO 打满，页面全挂了。后来改成客户端直传 OSS，服务端只生成签名，图片走 CDN 就近接入到 OSS，整个服务器压力降了 80%。另外遇到一个大坑是用户上传 10MB 的图但网络不稳定，总是上传失败。我加了分片上传 + 断点续传，把文件切成 4MB 一片，并发上传，中断了可以从 Redis 里读取已上传的分片列表继续。合并时用 OSS 的 CompleteMultipartUpload 接口，原子操作，不会出现半成品文件。另外每晚有定时任务清理 30 天未访问的冷数据，自动迁移到 OSS 低频存储，成本降了 40%。"
