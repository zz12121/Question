```java
// ❌ 问题配置 1：连接池配置不当
@Configuration
public class RestTemplateConfig {
    @Bean
    public RestTemplate restTemplate() {
        SimpleClientHttpRequestFactory factory = 
            new SimpleClientHttpRequestFactory();
        factory.setConnectTimeout(1000);      // ⚠️ 连接超时 1s 太短
        factory.setReadTimeout(3000);        // ⚠️ 读取超时 3s 太短
        factory.setBufferRequestBody(false);  // ⚠️ 小文件还好，大文件出问题
        
        // ⚠️ 默认连接池：无显式配置，用 SimpleClientHttpRequestFactory
        // → 每次请求创建新连接，不复用
        return new RestTemplate(factory);
    }
}

// ❌ 问题配置 2：HttpClient 连接池耗尽
@Configuration
public class HttpClientConfig {
    @Bean
    public CloseableHttpClient httpClient() {
        PoolingHttpClientConnectionManager cm = 
            new PoolingHttpClientConnectionManager();
        cm.setMaxTotal(50);      // ⚠️ 最大 50 连接
        cm.setDefaultMaxPerRoute(20);  // ⚠️ 每路由 20
        
        // ⚠️ 高并发时连接不够用，获取不到连接的请求会抛异常

            .setConnectionRequestTimeout(500)  // ⚠️ 获取连接超时 500ms 太短
            .setSocketTimeout(3000)
            .build();
        
        return HttpClients.custom()
            .setConnectionManager(cm)
            .setDefaultRequestConfig(config)
            .build();
    }
}
```

```java
// ❌ 问题代码：服务端提前关闭连接
@RestController
public class BadServerController {
    
    // ⚠️ 使用 try-with-resources 关闭 Response
    @GetMapping("/api/data")
    public String getData() {
        CloseableHttpResponse response = null;
        try {
            response = httpClient.execute(request);
            HttpEntity entity = response.getEntity();
            String result = EntityUtils.toString(entity);
            
            // ⚠️ 实体已消费，但 response 还没完全处理
            // 底层可能还在读数据，而连接已经被归还到连接池
            return result;
            
        } catch (Exception e) {
            throw new RuntimeException(e);
        }
        // ⚠️ 如果这里还有逻辑没执行完，连接可能已被服务端关闭
    }
    
    // ⚠️ 读取不完整就关闭
    @GetMapping("/api/stream")
    public void streamData(HttpServletResponse response) {
        // 只读取部分数据
        for (int i = 0; i < 100; i++) {
            response.getWriter().write("data" + i);
        }
        // ⚠️ 客户端在读取剩余数据时，服务端已关闭 → Reset
        response.flushBuffer();
        response.close();
    }
}
```

```java
public class GoodHttpClient {
```

## 五、重试与容错设计

```java
// ✅ 重试机制（处理偶发 Reset）
@Configuration
public class RetryableConfig {
    
    @Bean
    public RetryTemplate retryTemplate() {
        ExponentialBackOffPolicy backOff = new ExponentialBackOffPolicy();
        backOff.setInitialInterval(100);
        backOff.setMultiplier(2);
        backOff.setMaxInterval(3000);
        backOff.setMaxElapsedTime(10000);
        
        SimpleRetryPolicy retry = new SimpleRetryPolicy();
        retry.setMaxAttempts(3);
        
        RetryTemplate template = new RetryTemplate();
        template.setBackOffPolicy(backOff);
        template.setRetryPolicy(retry);
        
        // ⚠️ 配置哪些异常重试
        Map<Class<? extends Throwable>, Boolean> retryable = new HashMap<>();
        retryable.put(SocketException.class, true);         // ✓ 重试
        retryable.put(ConnectException.class, true);        // ✓ 重试
        retryable.put(HttpClientErrorException.class, false); // ✗ 不重试
        
        template.setRetryPolicy(new SimpleRetryPolicy(3, retryable));
        
        return template;
    }
}

// ✅ 使用 Feign 的重试能力
@FeignClient(
    name = "user-service",
    fallback = UserServiceFallback.class,
    configuration = FeignRetryConfig.class
)
public interface UserClient {
    // ...
}

@Configuration
public class FeignRetryConfig {
    @Bean
    public Retryer retryer() {
        // ⚠️ 重试 3 次，每次间隔递增
        return new Retryer.Default(100, 3000, 3);
    }
}

// ✅ RestTemplate 拦截器 + 重试
@Configuration
public class RestTemplateRetryConfig {
    
    @Bean
    public RestTemplate restTemplate() {
        // 配置重试逻辑
        HttpComponentsClientHttpRequestFactory factory = 
            new HttpComponentsClientHttpRequestFactory();
        
        // ⚠️ Apache HttpClient 默认会重试 3 次
        factory.setReadTimeout(10000);
        factory.setConnectTimeout(5000);
        
        // ⚠️ 关闭自动重试（避免非幂等请求被重复发送）
        RequestConfig config = RequestConfig.custom()
            .setMaxRedirects(0)  // 禁止重定向
            .setContentCompressionEnabled(true)
            .build();
        
        return new RestTemplate(factory);
    }
}
```

## 🔥 扩展追问 1：为什么"偶发"而不是"必现"？

> **A:** 偶发性 Connection Reset 的典型场景：

| 场景 | 原因 | 触发条件 |
|------|------|---------|
| **负载均衡心跳** | LB 定期检测后端存活，无响应则剔除 | 后端 CPU 高/GC 时 |
| **连接池空闲回收** | 服务端空闲超时关闭连接，客户端不知情 | 空闲连接复用 |
| **后端滚动重启** | 正在关闭的实例收到请求，发 RST | 灰度发布时 |
| **网络抖动** | 中间网络设备丢包，TCP 重传失败 | 网络不稳定 |
| **线程池抖动** | 后端线程池满，请求被拒绝 | 突发流量 |

```bash
# 定位偶发原因：对比正常请求和 Reset 请求的差异

tcpdump -i eth0 -nn 'tcp[tcpflags] & tcp-rst != 0' -w reset_$(date +%s).pcap

# 分析：找到 Reset 前后的包
# tshark -r reset.pcap -Y "tcp.rst" -T fields -e ip.src -e ip.dst
```

## 🔥 扩展追问 3：HTTP/2 和 HTTP/1.1 在这方面有何不同？

> **A:** 连接管理差异：

| 维度 | HTTP/1.1 | HTTP/2 |
|------|---------|--------|
| 连接复用 | Pipeline（有队头阻塞） | Multiplexing（真并行）|
| 连接关闭 | 一个连接出问题影响大 | 单个 Stream 失败，连接可复用 |
| Reset 处理 | Connection Reset 整个连接 | Stream Reset（RST_STREAM）只影响单个请求 |
| 心跳 | Keep-Alive 超时断开 | PING 帧保活，更稳定 |

```yaml
# Feign HTTP/2 配置
feign:
  http2:
    enabled: true  # 启用 HTTP/2

# Spring Cloud LoadBalancer HTTP/2
spring:
  cloud:
    loadbalancer:
      ribbon:
        enable: true
```

## 一句话总结

> **Connection Reset 的本质是 TCP 连接被一方提前关闭。偶发型主要是：服务端连接超时关闭（keepalive）、负载均衡剔除后端、客户端超时主动断开、网络抖动。排查核心是 tcpdump 抓包看 RST 谁发的，同时检查连接池配置（maxTotal/maxIdle/timeout）、服务端的 keepalive 配置、以及是否有滚动重启/灰度发布。**

## 面试加分话术

> "之前遇到过一个问题，每天固定时间报一批 Connection Reset。一开始怀疑是网络抖动，但 tcpdump 抓包一看，RST 全是服务端发的。后来排查发现，是下游服务用了 Nginx 做负载均衡，`keepalive_timeout` 设置的是 60 秒，而我们客户端的连接池 `maxIdle` 没有配置，默认是空闲 1 分钟就回收。白天流量低的时候，连接被客户端回收了，但服务端 Nginx 还以为连接在，等客户端复用时，服务端已经关闭了，直接发 RST。后来两边 keepalive 时间对齐，再加了个定时保活，就没再出现了。所以这类偶发问题，关键是**抓包看 RST 谁发的**，这个信息比任何日志都有用。"
