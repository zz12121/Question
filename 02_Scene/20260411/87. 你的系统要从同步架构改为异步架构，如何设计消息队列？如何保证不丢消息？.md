## 四、生产者可靠发送

### 4.1 同步发送 + 确认机制

```java
@Service
public class ReliableMessageProducer {
    
    @Autowired private RocketMQTemplate rocketMQTemplate;
    @Autowired private MqMessageLogService messageLogService;
    
    /**
     * 可靠消息发送流程
     */
    public void sendReliableMessage(String topic, String tag, 
            String messageKey, Object messageBody) {
        
        String messageId = UUID.randomUUID().toString();
        
        // 1. 消息入库（发送中状态）
        MqMessageLog log = MqMessageLog.builder()
            .messageId(messageId)
            .topic(topic)
            .tag(tag)
            .messageKey(messageKey)
            .messageBody(JSON.toJSONString(messageBody))
            .status(MessageStatus.SENDING)
            .build();
        messageLogService.save(log);
        
        try {
            // 2. 同步发送 + 超时控制
            SendResult result = rocketMQTemplate.syncSend(
                topic + ":" + tag,
                MessageBuilder.withPayload(messageBody)
                    .setHeader("messageId", messageId)
                    .setHeader("messageKey", messageKey)
                    .build(),
                3000  // 超时 3s
            );
            
            // 3. 发送成功，更新状态
            if (result.getSendStatus() == SendStatus.SEND_OK) {
                log.setStatus(MessageStatus.SENT_SUCCESS);
                messageLogService.updateById(log);
                
                // 异步更新消费状态监听
                // ...后续由 Consumer 回调更新
            } else {
                throw new RuntimeException("Send failed: " + result.getSendStatus());
            }
            
        } catch (Exception e) {
            // 4. 发送失败，更新状态 + 触发重试
            log.setStatus(MessageStatus.SEND_FAILED);
            log.setRetryCount(log.getRetryCount() + 1);
            log.setNextRetryTime(calculateNextRetry(log.getRetryCount()));
            messageLogService.updateById(log);
            
            // 触发重试（定时任务扫描）
            retryScheduler.scheduleRetry(log.getMessageId());
            
            throw e;
        }
    }
    
    private LocalDateTime calculateNextRetry(int retryCount) {
        // 指数退避：1s → 2s → 4s → 8s → 16s → 32s → 1m → 5m
        if (retryCount <= 3) {
            return LocalDateTime.now().plusSeconds((long) Math.pow(2, retryCount));
        } else if (retryCount <= 6) {
            return LocalDateTime.now().plusSeconds((long) Math.pow(2, retryCount - 3) * 30);
        } else {
            return LocalDateTime.now().plusMinutes(5);  // 最大 5 分钟
        }
    }
}
```

### 4.2 事务消息（最佳方案）

```java
@Service
public class TransactionMessageProducer {
    
    @Autowired private RocketMQTemplate rocketMQTemplate;
    @Autowired private OrderService orderService;
    @Autowired private MqMessageLogService messageLogService;
    
    /**
     * 事务消息：订单创建 + 消息发送 原子性
     */
    public void createOrderWithTransaction(OrderDTO order) {
        String transactionId = UUID.randomUUID().toString();
        
        // 1. 开启本地事务
        Transaction transaction = rocketMQTemplate.beginTransaction();
        
        try {
            // 2. 执行本地业务（订单创建）
            orderService.createOrder(order);
            
            // 3. 发送消息（作为事务的一部分）
            // 注意：此时消息是 "prepared" 状态，不会被消费
            SendResult sendResult = rocketMQTemplate.sendMessageInTransaction(
                "order_topic",
                "create_order",
                MessageBuilder.withPayload(order)
                    .setHeader("transactionId", transactionId)
                    .setHeader("messageKey", order.getOrderId())
                    .build(),
                order  // transactionExt: 传递给事务监听器
            );
            
            // 4. 提交本地事务
            transaction.commit();
            
            // 5. 如果发送失败，事务监听器会收到 LocalTransactionState.ROLLBACK_MESSAGE
            // 并删除 prepared 消息
            
        } catch (Exception e) {
            // 6. 回滚本地事务
            transaction.rollback();
            throw new BusinessException("订单创建失败", e);
        }
    }
}

/**
 * 事务消息监听器
 */
@Component
@Slf4j
public class OrderTransactionListener implements RocketMQLocalTransactionListener {
    
    @Autowired private OrderMapper orderMapper;
    
    @Override
    public RocketMQLocalTransactionState executeLocalTransaction(
            Message msg, Object arg) {
        
        try {
            // 1. 检查本地事务是否提交成功
            // (在事务消息中，这里已经执行过 orderService.createOrder)
            
            Order order = (Order) arg;
            Order existing = orderMapper.selectById(order.getOrderId());
            
            if (existing != null && "PAID".equals(existing.getStatus())) {
                // 订单已创建成功，提交消息
                return RocketMQLocalTransactionState.COMMIT;
            } else {
                // 订单未创建成功，回滚消息
                return RocketMQLocalTransactionState.ROLLBACK;
            }
            
        } catch (Exception e) {
            log.error("事务消息执行失败", e);
            return RocketMQLocalTransactionState.UNKNOWN;
        }
    }
    
    @Override
    public RocketMQLocalTransactionState checkLocalTransaction(Message msg) {
        // 事务超时后，MQ 调用此方法检查本地事务状态
        String transactionId = msg.getHeaders().get("transactionId", String.class);
        
        // 查询本地事务是否完成
        Order order = orderMapper.selectByTransactionId(transactionId);
        
        if (order != null) {
            return RocketMQLocalTransactionState.COMMIT;
        } else {
            return RocketMQLocalTransactionState.ROLLBACK;
        }
    }
}
```

## 五、消费者可靠消费

```java
@Service
@RocketMQMessageListener(
    topic = "order_topic",
    consumerGroup = "inventory-consumer-group",
    // ⚠️ 配置手动 ACK
    acknowledgeMode = AckMode.MANUAL,
    maxConcurrentConsumers = 10,
    consumeThreadMin = 10,
    consumeThreadMax = 20
)
@Slf4j
public class InventoryConsumer implements RocketMQListener<OrderMessage> {
    
    @Autowired private InventoryService inventoryService;
    @Autowired private MqConsumerLogService consumerLogService;
    
    @Override
    public void onMessage(OrderMessage message, 
            ConsumeConcurrentlyContext context, 
            AckCallback ackCallback) {
        
        String messageId = message.getMessageId();
        long startTime = System.currentTimeMillis();
        
        try {
            // 1. 幂等检查
            if (isAlreadyProcessed(messageId)) {
                log.info("消息已处理，跳过: messageId={}", messageId);
                ackCallback.ack();
                return;
            }
            
            // 2. 执行业务逻辑
            boolean success = inventoryService.decreaseStock(
                message.getOrderId(),
                message.getProductId(),
                message.getQuantity()
            );
            
            // 3. 记录消费日志
            MqConsumerLog log = MqConsumerLog.builder()
                .consumerGroup("inventory-consumer-group")
                .messageId(messageId)
                .topic(message.getTopic())
                .status(success ? ConsumeStatus.SUCCESS : ConsumeStatus.FAILED)
                .processTime((int) (System.currentTimeMillis() - startTime))
                .build();
            consumerLogService.save(log);
            
            // 4. 手动 ACK
            if (success) {
                ackCallback.ack();
            } else {
                // 业务失败，不 ACK，返回 RECONSUME_LATER
                // 消息会进入重试队列
                ackCallback.ack(new RuntimeException("库存不足"));
            }
            
        } catch (DuplicateKeyException e) {
            // 幂等键冲突，说明已处理，直接 ACK
            log.warn("消息重复消费: messageId={}", messageId);
            ackCallback.ack();
            
        } catch (Exception e) {
            log.error("消费异常: messageId={}", messageId, e);
            
            // 记录异常日志
            MqConsumerLog errorLog = MqConsumerLog.builder()
                .consumerGroup("inventory-consumer-group")
                .messageId(messageId)
                .topic(message.getTopic())
                .status(ConsumeStatus.FAILED)
                .errorMsg(ExceptionUtils.getMessage(e))
                .processTime((int) (System.currentTimeMillis() - startTime))
                .build();
            consumerLogService.save(errorLog);
            
            // 抛出异常，触发 MQ 重试
            throw e;
        }
    }
    
    private boolean isAlreadyProcessed(String messageId) {
        return consumerLogService.existsByMessageId(
            "inventory-consumer-group", messageId);
    }
}
```

```java
// 死信队列消费者（人工处理）
@Service
@RocketMQMessageListener(
    topic = "%DLQ%inventory-consumer-group",  // RocketMQ 死信 Topic
    consumerGroup = "dlq-handler-group"
)
public class DeadLetterQueueHandler implements RocketMQListener<Message> {
    
    @Autowired private AlertService alertService;
    @Autowired private MqMessageLogService messageLogService;
    
    @Override
    public void onMessage(Message message) {
        String messageId = message.getHeaders().get("messageId", String.class);
        String originalTopic = message.getHeaders().get("ORIGIN_TOPIC", String.class);
        
        // 1. 发送告警
        alertService.send("【死信告警】消息处理失败",
            String.format("Topic: %s, MessageId: %s", originalTopic, messageId));
        
        // 2. 记录详情到数据库
        messageLogService.saveDeadLetter(messageId, originalTopic, message);
        
        // 3. 人工补偿入口（可配置是否自动补偿）
        // 这里可以做人工补偿或人工重放
        log.warn("死信消息: topic={}, messageId={}", originalTopic, messageId);
    }
}
```

## 🔥 扩展追问 1：如何保证消息不重复消费（幂等性）？

> **A:** 消费端幂等是消息队列的核心难点：

| 方案 | 实现 | 适用场景 |
|------|------|---------|
| **唯一消息ID + 消费记录表** | 每次消费前查表，存在则跳过 | 通用，依赖 DB |
| **Redis 防重** | 消费前 SET NX EX，存在则跳过 | 高性能，但依赖 Redis |
| **业务状态机** | 订单状态只能 0→1→2，不能反向 | 最可靠 |
| **分布式锁** | 处理前加锁，处理后释放 | 并发控制 |

```java
// 方案1：消费记录表（最常用）
@Service
public class IdempotentConsumer {
    
    @Autowired private ConsumerLogMapper logMapper;
    @Autowired private InventoryService inventoryService;
    
    public void consume(Message message) {
        String idempotentKey = message.getMessageId();
        
        // 1. 幂等检查
        if (logMapper.existsByKey(idempotentKey)) {
            return;  // 已处理，直接返回
        }
        
        // 2. 开启事务
        Connection conn = dataSource.getConnection();
        conn.setAutoCommit(false);
        
        try {
            // 3. 插入处理记录（唯一索引保证原子性）
            logMapper.insert(ConsumerLog.builder()
                .idempotentKey(idempotentKey)
                .status("PROCESSING")
                .build());
            
            // 4. 执行业务
            inventoryService.decreaseStock(message);
            
            // 5. 更新状态
            logMapper.updateStatus(idempotentKey, "SUCCESS");
            
            conn.commit();
            
        } catch (DuplicateKeyException e) {
            // 唯一键冲突，说明并发情况下另一个线程已处理
            conn.rollback();
        } catch (Exception e) {
            conn.rollback();
            logMapper.updateStatus(idempotentKey, "FAILED");
            throw e;
        } finally {
            conn.setAutoCommit(true);
        }
    }
}

// 方案2：状态机幂等（最可靠）
@Service
public class OrderStateMachineConsumer {
    
    public void consume(OrderMessage message) {
        String orderId = message.getOrderId();
        
        // 订单状态机：CREATE → PAID → COMPLETED
        // 只能正向流转，不能反向
        int affected = orderMapper.updateStatus(
            orderId,
            "CREATE",           // 当前状态
            "PAID"              // 目标状态
        );
        
        if (affected == 0) {
            // 更新失败，说明状态已变化（可能已处理过）
            log.info("订单状态不是 CREATE，跳过: orderId={}", orderId);
            return;
        }
        
        // 执行业务
        inventoryService.decreaseStock(message);
    }
}
```

## 🔥 扩展追问 2：RocketMQ 和 Kafka 如何选型？

> **A:** 核心差异对比：

| 维度 | RocketMQ | Kafka |
|------|----------|-------|
| **事务消息** | ✅ 原生支持 | ❌ 需要自研（Kafka Streams） |
| **延迟消息** | ✅ 原生支持（定时） | ❌ 需要自研 |
| **顺序消息** | ✅ 单队列有序 | ✅ 单 Partition 有序 |
| **消费模式** | 推/拉均可 | 拉（Pull） |
| **顺序写** | 需配置 | ✅ 天生顺序写 |
| **延迟级别** | 1s/5s/10s/30s/1m 自定义 | 无 |
| **吞吐量** | 中等（10万/秒） | 高（百万/秒）|
| **消息积压** | 支持亿级 | 支持亿级 |
| **生态** | 阿里巴巴开源 | Apache 顶级项目 |

```
│  │  1. 消费失败 → 写入 delay_message 表                            │   │
│  │  2. 定时任务每分钟扫描 delay_message 表                         │   │
```

```java
// 方案3：数据库 + 定时任务实现延迟重试
@Service
public class DelayMessageScheduler {
    
    @Autowired private DelayMessageMapper delayMapper;
    @Autowired private ReliableMessageProducer producer;
    
    @Scheduled(fixedDelay = 60000)  // 每分钟执行
    public void processDelayMessages() {
        // 1. 查询到达时间的消息
        List<DelayMessage> messages = delayMapper
            .findByDelayTimeBeforeAndStatus(
                LocalDateTime.now(), 
                DelayStatus.PENDING
            );
        
        for (DelayMessage msg : messages) {
            try {
                // 2. 重新发送
                producer.sendReliableMessage(
                    msg.getTopic(),
                    msg.getTag(),
                    msg.getMessageKey(),
                    JSON.parseObject(msg.getMessageBody())
                );
                
                // 3. 更新状态
                delayMapper.updateStatus(msg.getId(), DelayStatus.SENT);
                
            } catch (Exception e) {
                log.error("延迟消息重新发送失败: id={}", msg.getId(), e);
                
                // 4. 如果超过最大重试，标记失败
                if (msg.getRetryCount() >= msg.getMaxRetry()) {
                    delayMapper.updateStatus(msg.getId(), DelayStatus.FAILED);
                    alertService.send("延迟消息重试失败", msg);
                } else {
                    // 5. 更新重试次数和下次重试时间
                    delayMapper.incrementRetry(msg.getId());
                }
            }
        }
    }
}
```

## 🔥 扩展追问 4：如何设计消息的顺序保证？

> **A:** 顺序保证的代价与实现：

```java
// RocketMQ 顺序消息
@Service
public class OrderMessageProducer {
    
    @Autowired private RocketMQTemplate rocketMQTemplate;
    
    /**
     * 发送顺序消息：同一订单的操作必须有序
     * 消息投递顺序：创建订单 → 扣减库存 → 支付 → 发货
     */
    public void sendOrderMessage(OrderDTO order) {
        String orderId = order.getOrderId();
        
        // 1. 所有消息使用相同的 OrderId 作为 Sharding Key
        //    保证同一订单的消息进入同一个 MessageQueue
        Message message = MessageBuilder.withPayload(order)
            .setHeader("orderId", orderId)
            .build();
        
        // 2. 发送顺序消息
        rocketMQTemplate.asyncSendOrderly(
            "order_topic:order",
            message,
            orderId,  // ⚠️ Sharding Key，同一 Key 进入同一队列
            new SendCallback() {
                @Override
                public void onSuccess(SendResult result) {
                    log.info("顺序消息发送成功: orderId={}, queueId={}", 
                        orderId, result.getQueueId());
                }
                
                @Override
                public void onException(Exception e) {
                    log.error("顺序消息发送失败: orderId={}", orderId, e);
                }
            }
        );
    }
}

// 消费者：顺序消费
@Service
@RocketMQMessageListener(
    topic = "order_topic",
    selectorExpression = "order",
    consumeMode = ConsumeMode.ORDERLY  // ⚠️ 顺序消费模式
)
public class OrderMessageConsumer implements RocketMQListener<Message> {
    
    @Override
    public void onMessage(Message message, ConsumeConcurrentlyContext context) {
        // ⚠️ 必须单线程顺序消费
        // ⚠️ 如果某条消息处理失败，整个队列暂停重试
        // ⚠️ 性能较低，只用于必须保证顺序的场景
        
        String orderId = message.getHeaders().get("orderId", String.class);
        
        try {
            processOrderStep(message);
        } catch (Exception e) {
            // ⚠️ 这里不能直接抛异常，否则阻塞后续消息
            // 应该：记录日志 + 触发告警 + 人工处理
            handleFailedOrder(orderId, message);
        }
    }
}
```

> - 不适合高并发场景
