### 2.2 正确方案一：本地消息表 + 定时补偿

```java
// ========== 本地消息表方案 ==========

// 1. 消息表
CREATE TABLE `outbox_message` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `biz_type` VARCHAR(32) NOT NULL COMMENT '业务类型',
  `biz_id` VARCHAR(64) NOT NULL COMMENT '业务ID',
  `payload` TEXT NOT NULL COMMENT '消息内容(JSON)',
  `status` TINYINT NOT NULL DEFAULT 0 COMMENT '状态: 0-待发送 1-已发送 2-确认',
  `retry_count` INT DEFAULT 0,
  `create_at` DATETIME NOT NULL,
  `update_at` DATETIME NOT NULL,
  KEY `idx_status_create` (`status`, `create_at`)
) ENGINE=InnoDB;

// 2. 改造后的下单逻辑（保证本地原子性）
@Service
public class OrderServiceWithOutbox {
    
    @Autowired
    private OrderMapper orderMapper;
    @Autowired
    private OutboxMapper outboxMapper;
    @Autowired
    private StockService stockService;
    
    @Transactional
    public void createOrder(OrderDTO dto) {
        // 1. 创建订单（状态待支付）
        Order order = new Order();
        order.setOrderNo(generateOrderNo());
        order.setStatus(OrderStatus.PENDING);
        order.setSkuId(dto.getSkuId());
        order.setUserId(dto.getUserId());
        orderMapper.insert(order);
        
        // 2. 创建库存扣减消息（和订单在同一事务！）
        OutboxMessage message = new OutboxMessage();
        message.setBizType("STOCK_DECREMENT");
        message.setBizId(order.getOrderNo());
        message.setPayload(JSON.toJSONString(new StockDecrementDTO(dto.getSkuId(), 1)));
        message.setStatus(MessageStatus.PENDING);
        outboxMapper.insert(message);
        
        // 3. 本地事务提交后，消息一定存在
        // 两个操作在同一个 DB 事务中，要么都成功，要么都失败
    }
    
    // 4. 消息发送任务（轮询 outbox 表）
    @Scheduled(fixedRate = 1000)
    public void sendPendingMessages() {
        List<OutboxMessage> messages = outboxMapper.selectPending(100);
        
        for (OutboxMessage msg : messages) {
            try {
                // 发送消息给库存服务
                mqTemplate.convertAndSend("stock-topic", msg.getPayload());
                
                // 标记已发送
                outboxMapper.updateStatus(msg.getId(), MessageStatus.SENT);
                
            } catch (Exception e) {
                // 重试逻辑
                outboxMapper.incrementRetry(msg.getId());
            }
        }
    }
}
```

### 2.3 正确方案二：RocketMQ 事务消息

```java
// ========== RocketMQ 事务消息方案 ==========
@Service
public class TransactionalOrderService {
    
    @Autowired
    private OrderMapper orderMapper;
    @Autowired
    private RocketMQTemplate rocketMQTemplate;
    
    /**
     * RocketMQ 事务消息：半消息发送
     */
    @Transactional
    public void createOrderWithTransaction(OrderDTO dto) {
        // 1. 创建订单（本地事务）
        Order order = buildOrder(dto);
        orderMapper.insert(order);
        
        // 2. 发送事务消息（半消息）
        // 此时消息不会立即投递给消费者
        rocketMQTemplate.sendMessageInTransaction(
            "order-stock-topic",
            MessageBuilder
                .withPayload(new StockDecrementMessage(dto.getSkuId(), 1, order.getOrderNo()))
                .setHeader("orderNo", order.getOrderNo())
                .build(),
            null  // transactionListener 单独配置
        );
        
        // 3. 本地事务提交后，MQ 回调确认，消息才会投递给库存服务
    }
}

// ========== RocketMQ TransactionListener ==========
@Component
public class OrderTransactionListener implements RocketMQLocalTransactionListener {
    
    @Autowired
    private OrderMapper orderMapper;
    @Autowired
    private StockService stockService;
    
    @Override
    public RocketMQLocalTransactionState executeLocalTransaction(Message msg, Object arg) {
        try {
            // 扣减库存
            StockDecrementMessage dto = JSON.parseObject(
                new String((byte[]) msg.getPayload()), 
                StockDecrementMessage.class
            );
            
            stockService.decrement(dto.getSkuId(), dto.getCount());
            
            // 本地事务成功，提交消息
            return RocketMQLocalTransactionState.COMMIT;
            
        } catch (Exception e) {
            log.error("本地事务失败", e);
            // 本地事务失败，回滚消息
            return RocketMQLocalTransactionState.ROLLBACK;
        }
    }
    
    @Override
    public RocketMQLocalTransactionState checkLocalTransaction(Message msg) {
        // MQ 回调检查：订单是否真的创建成功了
        String orderNo = msg.getHeaders().get("orderNo", String.class);
        
        Order order = orderMapper.selectByOrderNo(orderNo);
        if (order != null) {
            // 订单存在，提交消息（让库存服务继续）
            return RocketMQLocalTransactionState.COMMIT;
        } else {
            // 订单不存在，回滚消息
            return RocketMQLocalTransactionState.ROLLBACK;
        }
    }
}
```

### 2.4 正确方案三：TCC 模式

```java
// ========== TCC 模式 ==========
@Service
public class TccOrderService {
    
    @Autowired
    private TccTransactionManager tccManager;
    
    public void createOrder(OrderDTO dto) {
        String txId = tccManager.begin();
        
        try {
            // Try: 预留库存
            stockTccService.tryDecrement(txId, dto.getSkuId(), 1);
            
            // Try: 创建订单
            orderTccService.tryCreate(txId, dto);
            
            // Confirm: 确认扣减
            stockTccService.confirm(txId);
            
            // Confirm: 确认订单
            orderTccService.confirm(txId);
            
        } catch (Exception e) {
            // Cancel: 回滚
            stockTccService.cancel(txId);
            orderTccService.cancel(txId);
        }
    }
}

// TCC 接口定义
@LocalTCC
public interface StockTccService {
    
    @TwoPhaseBusinessAction(name = "decrement", commitMethod = "confirm", rollbackMethod = "cancel")
    boolean tryDecrement(
        @BusinessActionContextParameter(paramName = "txId") String txId,
        @BusinessActionContextParameter(paramName = "skuId") Long skuId,
        int count
    );
    
    boolean confirm(BusinessActionContext context);
    boolean cancel(BusinessActionContext context);
}
```

## 🔥 扩展追问 2：如何保证幂等防止重复扣库存？

> **A:** 三层幂等：消息层 + 业务层 + 库存层。

```java
// 1. 消息层幂等：消息 ID 去重
@RocketMQMessageListener(topic = "stock-topic", concurrencyLevel = 1)
public class StockConsumer {
    
    @Autowired
    private IdempotentService idempotentService;
    
    @Override
    public void consume(Message msg) {
        String msgId = msg.getMsgId();
        
        // 消息幂等：同一个 msgId 只处理一次
        if (!idempotentService.tryAcquire("MSG", msgId, Duration.ofDays(7))) {
            return;  // 已处理过，跳过
        }
        
        // 业务处理...
        processStockDecrement(msg);
    }
}

// 2. 库存层幂等：业务 Key 去重
public void decrement(Long skuId, int count, String orderNo) {
    // 用订单号做幂等 Key，同一订单扣库存只生效一次
    int rows = stockMapper.decrementIfAbsent(
        skuId, count, 
        "order:" + orderNo  // 幂等 Key
    );
    
    if (rows == 0) {
        log.warn("库存扣减幂等命中，订单: {}", orderNo);
    }
}

// 3. Mapper 幂等更新
<update id="decrementIfAbsent">
  UPDATE stock
  SET stock = stock - #{count},
      version = version + 1
  WHERE sku_id = #{skuId}
    AND idempotent_key IS NULL OR idempotent_key = #{idempotentKey}
</update>
```

## 🔥 扩展追问 3：库存扣减和订单创建，哪个先做？

> **A:** 要根据业务场景和补偿能力来决定。

| 方案 | 先做什么 | 优点 | 缺点 |
|------|---------|------|------|
| 先扣库存 | 用户体验好 | 库存有限时先占住 | 库存扣了订单没创建，需要补偿 |
| 先创订单 | 数据一致性容易保证 | 订单创建失败直接返回 | 库存可能被占满但无实际成交 |

**最佳实践：先创订单 + 锁定库存 + 最终一致性**

```java
public void createOrder(OrderDTO dto) {
    // 1. 创建订单（状态 PENDING）
    Order order = orderMapper.insert(buildPendingOrder(dto));
    
    // 2. 锁定活动库存（乐观锁）
    boolean locked = stockService.tryLock(order.getActivityId(), dto.getSkuId(), 1);
    if (!locked) {
        orderMapper.updateStatus(order.getOrderNo(), OrderStatus.STOCK_LOCK_FAILED);
        throw new BizException("库存锁定失败");
    }
    
    // 3. 订单状态更新
    orderMapper.updateStatus(order.getOrderNo(), OrderStatus.STOCK_LOCKED);
    
    // 4. 支付成功后真正扣减库存
    // onPaymentSuccess -> confirmDeduct
}
```

## 🔥 扩展追问 4：如何设计兜底补偿机制？

```java
// 1. 库存订单不一致检测
@Scheduled(cron = "0 */5 * * * ?")  // 每5分钟
public void detectInconsistency() {
    // 查出所有 PENDING/STOCK_LOCKED 状态的订单
    List<Order> pendingOrders = orderMapper.selectPendingOrders();
    
    for (Order order : pendingOrders) {
        // 检查关联的库存记录
        StockLog stockLog = stockLogMapper.selectByOrderNo(order.getOrderNo());
        
        if (stockLog == null) {
            // 库存没扣减记录，但订单存在 → 需要人工介入
            createAlert(AlertType.STOCK_ORDER_MISMATCH, order);
        }
        
        if (stockLog != null && stockLog.getStatus() == 'SUCCESS' 
            && order.getStatus() == 'PENDING') {
            // 库存扣成功但订单未确认 → 可能是 MQ 消费失败
            retryCreateOrder(order);
        }
    }
}

// 2. 库存已扣但订单超时未支付 → 自动退款
@Scheduled(cron = "0 */10 * * * ?")  // 每10分钟
public void refundTimeoutOrders() {
    List<Order> timeoutOrders = orderMapper.selectTimeoutUnpaidOrders(30);
    
    for (Order order : timeoutOrders) {
        // 归还库存
        stockService.release(order.getSkuId(), 1, order.getOrderNo());
        
        // 取消订单
        orderMapper.updateStatus(order.getOrderNo(), OrderStatus.CANCELLED_TIMEOUT);
    }
}

// 3. 告警配置

// - 消息积压告警
// - 订单超时率告警
```

## 🔥 扩展追问 5：Seata AT 模式能解决这个问题吗？

> **A:** 能，但有局限性。Seata AT 适合强一致性场景，但拼团这种高并发+长链路场景不太适合。

```java
// Seata AT 模式示例
@GlobalTransactional(rollbackFor = Exception.class)
public void createOrder(OrderDTO dto) {
    // 1. 扣减库存（Seata 自动记录 undo log）
    stockService.decrement(dto.getSkuId(), 1);
    
    // 2. 创建订单（Seata 自动记录 undo log）
    orderMapper.insert(buildOrder(dto));
    
    // 3. 如果任意一步失败，Seata 会回滚所有分支事务
}

// 问题：
// 1. Seata AT 模式依赖全局锁，高并发下性能差
// 2. 分支事务多时，undo log 膨胀
// 3. 库存服务必须引入 Seata 依赖，增加复杂度
// 4. 不适合这种跨库、异步确认的场景
```

**Seata vs MQ 事务消息对比：**

| 特性 | Seata AT | RocketMQ 事务消息 |
|------|---------|------------------|
| 一致性 | 强一致 | 最终一致 |
| 性能 | 较差（有全局锁） | 好 |
| 侵入性 | 需要引入框架 | 较小 |
| 适用场景 | 单链路、短链路 | 长链路、异步场景 |
| 超时处理 | 自动回滚 | 定时补偿 |

## 一句话总结

> **"库存扣了订单没创建"本质是跨服务分布式事务一致性问题，排查从日志/DB状态/消息轨迹三个入口入手，解决方案有本地消息表+RocketMQ事务消息+TCC三种模式，核心原则是库存扣减和订单创建必须绑定在同一事务边界内，再用幂等+定时补偿兜底，最终保证数据最终一致性。**

**面试加分话术：**

> "之前公司遇到过这个问题：MQ 消费超时导致库存服务扣了库存，但回调通知订单服务创建订单时网络抖动，订单没创建成功。后来我主导重构了这块逻辑，用本地消息表方案：下单时先在同一个 DB 事务里创建订单记录和一条待发送消息，事务提交后由定时任务轮询发送消息，库存服务消费消息后回执，消息状态更新为已确认。如果消息发送失败，定时任务会重试；如果库存服务消费失败，消息会进死信队列告警。另外关键点是所有扣库存操作都必须带幂等 Key（用订单号），防止重复消费导致超卖。现在这套机制跑了两年，日志里偶尔有几笔需要人工补偿的，数量级降到了万分之一以下。"
