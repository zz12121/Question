### 2.2 方案一：乐观锁（适合冲突较少的场景）

```java
// ========== 订单服务 ==========
@Service
@Slf4j
public class OrderServiceWithOptimisticLock {
    
    @Autowired
    private OrderMapper orderMapper;
    @Autowired
    private OrderAmountLogMapper logMapper;
    
    private static final int MAX_RETRY = 3;
    
    /**
     * 修改订单金额（乐观锁）
     */
    @Transactional
    public boolean modifyAmount(Long orderId, BigDecimal newAmount, String reason) {
        for (int i = 0; i < MAX_RETRY; i++) {
            // 1. 查询当前订单
            Order order = orderMapper.selectById(orderId);
            
            // 2. 检查是否可以修改
            if (order.getFinancialStatus() == FinancialStatus.LOCKED) {
                throw new BizException("订单正在财务核算中，无法修改");
            }
            if (order.getFinancialStatus() == FinancialStatus.SETTLED) {
                throw new BizException("订单已结算，无法修改");
            }
            
            // 3. 乐观锁更新
            int rows = orderMapper.updateAmountWithVersion(
                orderId,
                newAmount,
                order.getVersion(),
                order.getFinalAmount()  // 期望的旧值
            );
            
            if (rows == 1) {
                // 成功，记录修改日志
                OrderAmountLog log = new OrderAmountLog();
                log.setOrderNo(order.getOrderNo());
                log.setOperatorType("USER");
                log.setOperatorId(getCurrentUserId());
                log.setBeforeAmount(order.getFinalAmount());
                log.setAfterAmount(newAmount);
                log.setReason(reason);
                log.setVersion(order.getVersion());
                logMapper.insert(log);
                
                log.info("订单金额修改成功: orderNo={}, {} → {}", 
                    order.getOrderNo(), order.getFinalAmount(), newAmount);
                return true;
            }
            
            // 4. 更新失败，version 不匹配，重试
            log.warn("订单金额修改重试: orderId={}, retry={}", orderId, i + 1);
        }
        
        throw new BizException("订单金额修改失败，请稍后重试");
    }
}

// ========== Mapper ==========
@Mapper
public interface OrderMapper {
    
    @Update("<script>" +
        "UPDATE orders SET " +
        "  final_amount = #{newAmount}, " +
        "  version = version + 1, " +
        "  update_version = update_version + 1, " +
        "  last_modify_time = NOW() " +
        "WHERE id = #{orderId} " +
        "  AND version = #{expectedVersion} " +
        "  AND final_amount = #{expectedAmount}" +
        "</script>")
    int updateAmountWithVersion(@Param("orderId") Long orderId,
                                @Param("newAmount") BigDecimal newAmount,
                                @Param("expectedVersion") Integer expectedVersion,
                                @Param("expectedAmount") BigDecimal expectedAmount);
}
```

```java
// ========== 悲观锁修改金额 ==========
@Service
public class OrderServiceWithPessimisticLock {
    
    @Autowired
    private OrderMapper orderMapper;
    
    /**
     * 修改订单金额（悲观锁：FOR UPDATE）
     */
    @Transactional
    public boolean modifyAmount(Long orderId, BigDecimal newAmount, String reason) {
        // 1. 加锁读取（阻塞其他读和写）
        Order order = orderMapper.selectByIdForUpdate(orderId);
        
        // 2. 检查状态
        if (order.getFinancialStatus() == FinancialStatus.LOCKED ||
            order.getFinancialStatus() == FinancialStatus.SETTLED) {
            throw new BizException("订单状态不允许修改");
        }
        
        // 3. 更新金额
        BigDecimal oldAmount = order.getFinalAmount();
        order.setFinalAmount(newAmount);
        order.setVersion(order.getVersion() + 1);
        order.setUpdateVersion(order.getUpdateVersion() + 1);
        order.setLastModifyTime(LocalDateTime.now());
        orderMapper.updateById(order);
        
        // 4. 记录日志
        log.info("订单金额修改成功: orderNo={}, {} → {}", 
            order.getOrderNo(), oldAmount, newAmount);
        
        return true;
    }
}

// ========== Mapper ==========
@Mapper
public interface OrderMapper {
    
    // SELECT ... FOR UPDATE：悲观锁
    @Select("SELECT * FROM orders WHERE id = #{orderId} FOR UPDATE")
    Order selectByIdForUpdate(Long orderId);
}

// ========== 财务核算（也用悲观锁）==========
@Service
public class FinancialService {
    
    @Autowired
    private OrderMapper orderMapper;
    @Autowired
    private FinancialRecordMapper financialRecordMapper;
    
    /**
     * 财务核算（悲观锁保护）
     */
    @Transactional
    public void settleOrder(Long orderId, Long operatorId) {
        // 1. 加锁读取订单
        Order order = orderMapper.selectByIdForUpdate(orderId);
        
        // 2. 检查状态
        if (order.getFinancialStatus() == FinancialStatus.SETTLED) {
            throw new BizException("订单已结算");
        }
        
        // 3. 锁定订单（防止修改）
        order.setFinancialStatus(FinancialStatus.LOCKED);
        orderMapper.updateById(order);
        
        // 4. 执行财务核算
        BigDecimal amount = order.getFinalAmount();
        // ... 核算逻辑
        
        // 5. 记录财务数据
        FinancialRecord record = new FinancialRecord();
        record.setOrderNo(order.getOrderNo());
        record.setOrderId(orderId);
        record.setAmount(amount);
        record.setSnapshotVersion(order.getVersion());  // 记录快照版本
        record.setSettleTime(LocalDateTime.now());
        record.setOperatorId(operatorId);
        financialRecordMapper.insert(record);
        
        // 6. 标记订单已结算
        order.setFinancialStatus(FinancialStatus.SETTLED);
        orderMapper.updateById(order);
        
        log.info("订单结算完成: orderNo={}, amount={}", order.getOrderNo(), amount);
    }
}
```

```java
// ========== 订单修改服务（发消息）==========
@Service
public class OrderMessageService {
    
    @Autowired
    private OrderMapper orderMapper;
    @Autowired
    private RocketMQTemplate rocketMQTemplate;
    
    /**
     * 修改订单金额（异步消息）
     */
    @Transactional
    public void modifyAmountAsync(Long orderId, BigDecimal newAmount, String reason) {
        // 1. 查询当前订单版本
        Order order = orderMapper.selectById(orderId);
        
        // 2. 记录修改申请（预记录）
        OrderAmountChangeRequest request = new OrderAmountChangeRequest();
        request.setRequestId(UUID.randomUUID().toString());
        request.setOrderId(orderId);
        request.setOrderNo(order.getOrderNo());
        request.setOldAmount(order.getFinalAmount());
        request.setNewAmount(newAmount);
        request.setReason(reason);
        request.setCurrentVersion(order.getVersion());
        request.setStatus(ChangeStatus.PENDING);
        requestMapper.insert(request);
        
        // 3. 发送消息（带版本号）
        OrderAmountChangedEvent event = new OrderAmountChangedEvent();
        event.setEventId(UUID.randomUUID().toString());
        event.setOrderId(orderId);
        event.setOrderNo(order.getOrderNo());
        event.setOldAmount(order.getFinalAmount());
        event.setNewAmount(newAmount);
        event.setCurrentVersion(order.getVersion());
        
        // 发送延迟消息（给财务系统留处理时间）
        rocketMQTemplate.asyncSend("order-amount-change-topic", event, 
            new DelayMsg(5, TimeUnit.SECONDS));  // 5秒后送达
        
        // 4. 先更新本地（乐观锁）
        int rows = orderMapper.updateAmountWithVersion(
            orderId, newAmount, order.getVersion(), order.getFinalAmount());
        
        if (rows == 1) {
            request.setStatus(ChangeStatus.APPLIED);
            requestMapper.updateById(request);
        }
    }
}

// ========== 财务核算消费者 ==========
@Service
@Slf4j
public class FinancialConsumer {
    
    @Autowired
    private OrderMapper orderMapper;
    @Autowired
    private FinancialRecordMapper financialRecordMapper;
    @Autowired
    private OrderAmountChangeRequestMapper requestMapper;
    
    @RocketMQMessageListener(topic = "order-amount-change-topic", concurrency = "3-5")
    public void handleAmountChange(OrderAmountChangedEvent event) {
        log.info("收到金额变更事件: orderNo={}, version={}", 
            event.getOrderNo(), event.getCurrentVersion());
        
        // 1. 查询变更申请记录
        OrderAmountChangeRequest request = 
            requestMapper.selectByRequestId(event.getEventId());
        
        // 2. 幂等检查
        if (request.getStatus() == ChangeStatus.PROCESSED) {
            log.info("该变更已处理，跳过: {}", event.getEventId());
            return;
        }
        
        // 3. 加锁查询订单（悲观锁）
        Order order = orderMapper.selectByIdForUpdate(event.getOrderId());
        
        // 4. 版本检查：如果当前版本 > 事件版本，说明有新修改
        if (order.getVersion() > event.getCurrentVersion()) {
            log.warn("订单版本已更新，需要重新核算: orderNo={}, eventVersion={}, currentVersion={}",
                event.getOrderNo(), event.getCurrentVersion(), order.getVersion());
            
            // 重新发起核算流程
            processSettlement(event.getOrderId());
            return;
        }
        
        // 5. 执行财务核算
        if (order.getFinancialStatus() != FinancialStatus.SETTLED) {
            FinancialRecord record = new FinancialRecord();
            record.setOrderNo(order.getOrderNo());
            record.setOrderId(order.getId());
            record.setAmount(order.getFinalAmount());
            record.setSnapshotVersion(order.getVersion());
            financialRecordMapper.insert(record);
            
            order.setFinancialStatus(FinancialStatus.SETTLED);
            orderMapper.updateById(order);
        }
        
        // 6. 更新请求状态
        request.setStatus(ChangeStatus.PROCESSED);
        requestMapper.updateById(request);
        
        log.info("金额变更处理完成: orderNo={}, amount={}", 
            event.getOrderNo(), order.getFinalAmount());
    }
}
```

```java
public enum OrderStatus {
    PENDING(0),      // 待支付
    PAID(1),         // 已支付
    MODIFYING(2),    // 修改中（锁定，财务暂停核算）

@Service
public class FinancialLockService {
    
    /**
     * 申请财务锁定（修改前）
     */
    public boolean tryAcquireFinancialLock(Long orderId) {
        String lockKey = "financial:lock:" + orderId;
        Boolean success = redisTemplate.opsForValue()
            .setIfAbsent(lockKey, "locked", Duration.ofMinutes(5));
        return Boolean.TRUE.equals(success);
    }
    
    /**
     * 释放财务锁定（修改后）
     */
    public void releaseFinancialLock(Long orderId) {
        redisTemplate.delete("financial:lock:" + orderId);
    }
    
    /**
     * 财务核算前检查
     */
    public boolean canSettle(Long orderId) {
        String lockKey = "financial:lock:" + orderId;
        return !Boolean.TRUE.equals(redisTemplate.hasKey(lockKey));
    }
}
```

## 🔥 扩展追问 1：乐观锁的重试机制怎么做？

> **A:** 重试要注意次数限制 + 退避策略 + 记录日志。

```java
// ========== 乐观锁重试工具 ==========
@Service
public class RetryableService {
    
    /**
     * 带退避的乐观锁重试
     */
    public <T> T executeWithRetry(Supplier<T> action, int maxRetries, String operationName) {
        int retryCount = 0;
        
        while (true) {
            try {
                return action.get();
            } catch (OptimisticLockException e) {
                retryCount++;
                
                if (retryCount >= maxRetries) {
                    log.error("乐观锁重试失败: operation={}, retries={}", operationName, retryCount);
                    throw new BizException("操作失败，请稍后重试");
                }
                
                // 指数退避 + 随机 jitter
                long delay = (long) Math.pow(2, retryCount) * 100 
                           + ThreadLocalRandom.current().nextInt(50);
                
                log.warn("乐观锁冲突，重试: operation={}, retry={}, delay={}ms", 
                    operationName, retryCount, delay);
                
                try {
                    Thread.sleep(delay);
                } catch (InterruptedException ie) {
                    Thread.currentThread().interrupt();
                    throw new BizException("操作被中断");
                }
            }
        }
    }
}

// ========== 使用示例 ==========
public void modifyAmount(Long orderId, BigDecimal newAmount) {
    executeWithRetry(() -> {
        Order order = orderMapper.selectById(orderId);
        int rows = orderMapper.updateAmountWithVersion(
            orderId, newAmount, order.getVersion(), order.getFinalAmount());
        
        if (rows == 0) {
            throw new OptimisticLockException("版本不匹配");
        }
        return true;
    }, 3, "modifyAmount");
}
```

## 🔥 扩展追问 2：如何保证财务核算的准确性？

> **A:** 快照 + 核对 + 异步通知。

```java
// ========== 财务核算核心逻辑 ==========
@Service
public class FinancialVerificationService {
    
    @Autowired
    private OrderMapper orderMapper;
    @Autowired
    private FinancialRecordMapper financialRecordMapper;
    
    /**
     * 财务核算（带快照保证准确性）
     */
    @Transactional
    public FinancialSettlementResult settle(Long orderId) {
        // 1. 快照订单数据（防止读取到修改中的数据）
        OrderSnapshot snapshot = orderMapper.selectSnapshotById(orderId);
        
        // 2. 核对金额
        BigDecimal orderAmount = snapshot.getFinalAmount();
        BigDecimal paidAmount = paymentMapper.sumPaidAmountByOrderId(orderId);
        BigDecimal refundAmount = refundMapper.sumRefundAmountByOrderId(orderId);
        
        // 3. 计算应结算金额
        BigDecimal settleAmount = orderAmount.subtract(refundAmount);
        
        if (settleAmount.compareTo(paidAmount) != 0) {
            // 金额不一致，告警并记录
            log.error("结算金额不一致: orderId={}, settleAmount={}, paidAmount={}",
                orderId, settleAmount, paidAmount);
            
            // 发送告警
            alertService.send("FINANCIAL_MISMATCH", 
                String.format("订单 %d 结算金额不一致", orderId));
        }
        
        // 4. 记录快照版本（对账用）
        FinancialRecord record = new FinancialRecord();
        record.setOrderNo(snapshot.getOrderNo());
        record.setOrderId(orderId);
        record.setOrderAmount(snapshot.getFinalAmount());
        record.setPaidAmount(paidAmount);
        record.setRefundAmount(refundAmount);
        record.setSettleAmount(settleAmount);
        record.setSnapshotVersion(snapshot.getVersion());  // 关键：记录版本
        record.setSnapshotHash(calculateHash(snapshot));   // 数据指纹
        financialRecordMapper.insert(record);
        
        return new FinancialSettlementResult(settleAmount, record.getId());
    }
    
    /**
     * 定期核对（保证长期一致性）
     */
    @Scheduled(cron = "0 0 2 * * ?")  // 每天凌晨2点
    public void dailyReconciliation() {
        // 查出最近结算的订单，核对快照
        List<FinancialRecord> records = financialRecordMapper.selectRecentRecords(30);
        
        for (FinancialRecord record : records) {
            Order current = orderMapper.selectById(record.getOrderId());
            
            // 比较版本和数据指纹
            if (current.getVersion() != record.getSnapshotVersion()) {
                log.error("发现数据篡改: orderNo={}, oldVersion={}, currentVersion={}",
                    record.getOrderNo(), record.getSnapshotVersion(), current.getVersion());
                
                // 发送安全告警
                alertService.sendSecurityAlert("DATA_TAMPERING", 
                    String.format("订单 %s 数据被篡改", record.getOrderNo()));
            }
        }
    }
}
```

## 🔥 扩展追问 3：分布式环境下如何保证一致性？

> **A:** 分布式锁 + 消息队列 + 对账机制。

```java
// ========== 分布式锁 + 消息队列 ==========
@Service
public class DistributedOrderService {
    
    @Autowired
    private RedissonClient redisson;
    
    /**
     * 修改金额（分布式锁保护）
     */
    @Transactional
    public boolean modifyAmount(Long orderId, BigDecimal newAmount) {
        // 1. 获取分布式锁
        String lockKey = "order:modify:" + orderId;
        RLock lock = redisson.getLock(lockKey);
        
        if (!lock.tryLock(5, 30, TimeUnit.SECONDS)) {
            throw new BizException("系统繁忙，请稍后重试");
        }
        
        try {
            // 2. 检查财务状态
            Order order = orderMapper.selectById(orderId);
            if (order.getFinancialStatus() == FinancialStatus.SETTLED) {
                throw new BizException("订单已结算，无法修改");
            }
            
            // 3. 执行修改
            return doModifyAmount(order, newAmount);
            
        } finally {
            lock.unlock();
        }
    }
    
    /**
     * 财务核算（分布式锁保护）
     */
    @Transactional
    public void settleOrder(Long orderId) {
        String lockKey = "order:settle:" + orderId;
        RLock lock = redisson.getLock(lockKey);
        
        if (!lock.tryLock(5, 30, TimeUnit.SECONDS)) {
            throw new BizException("系统繁忙，请稍后重试");
        }
        
        try {
            // 财务核算逻辑...
        } finally {
            lock.unlock();
        }
    }
}

// ========== 注意：订单修改锁 和 财务核算锁 要是同一个锁！==========
// 否则会出现：
// 用户获取 lock1 = order:modify:123
// 财务获取 lock2 = order:settle:123
// 死锁！
```

## 🔥 扩展追问 4：如何设计一个"财务安全锁"？

> **A:** 在业务层加逻辑锁，禁止并发操作。

```java
// ========== 财务安全锁设计 ==========
@Service
public class FinancialSafetyService {
    
    /**
     * 修改金额前：检查财务是否空闲
     */
    public void checkBeforeModify(Long orderId) {
        Order order = orderMapper.selectById(orderId);
        
        // 检查财务状态
        if (order.getFinancialStatus() == FinancialStatus.SETTLED) {
            throw new BizException("订单已结算，无法修改");
        }
        
        // 检查是否有正在进行的核算
        if (isFinancialProcessing(orderId)) {
            throw new BizException("财务正在核算中，请稍后再试");
        }
    }
    
    /**
     * 财务核算前：检查是否有修改操作
     */
    public void checkBeforeSettle(Long orderId) {
        Order order = orderMapper.selectById(orderId);
        
        // 检查修改状态
        if (hasPendingModification(orderId)) {
            throw new BizException("订单有待处理的修改，请稍后核算");
        }
        
        // 检查最后修改时间（5分钟内动过的订单谨慎处理）
        if (order.getLastModifyTime() != null) {
            Duration duration = Duration.between(
                order.getLastModifyTime(), LocalDateTime.now());
            
            if (duration.toMinutes() < 5) {
                log.warn("订单最近有修改，谨慎核算: orderId={}, minutes={}", 
                    orderId, duration.toMinutes());
                
                // 可以选择：
                // 1. 拒绝核算，等修改完成
                // 2. 强制核算（用最新金额）
                // 3. 人工介入
            }
        }
    }
    
    /**
     * 财务处理中：禁止修改
     */
    public void markAsProcessing(Long orderId) {
        orderMapper.updateFinancialStatus(orderId, FinancialStatus.PROCESSING);
    }
    
    /**
     * 财务处理完成：恢复
     */
    public void markAsCompleted(Long orderId) {
        orderMapper.updateFinancialStatus(orderId, FinancialStatus.SETTLED);
    }
}
```

## 🔥 扩展追问 5：如何回滚已财务核算的订单？

> **A:** 反向操作 + 冲正记录 + 审批流。

```java
// ========== 财务回滚 ==========
@Service
public class FinancialRollbackService {
    
    @Autowired
    private OrderMapper orderMapper;
    @Autowired
    private FinancialRecordMapper financialRecordMapper;
    @Autowired
    private ApprovalFlowService approvalFlowService;
    
    /**
     * 申请财务回滚（需要审批）
     */
    @Transactional
    public Long applyRollback(Long orderId, String reason) {
        // 1. 查询最新财务记录
        FinancialRecord latestRecord = 
            financialRecordMapper.selectLatestByOrderId(orderId);
        
        if (latestRecord == null) {
            throw new BizException("未找到财务记录");
        }
        
        if (latestRecord.getRollbackStatus() == RollbackStatus.APPLIED) {
            throw new BizException("该订单已有回滚申请");
        }
        
        // 2. 创建回滚申请
        RollbackApplication application = new RollbackApplication();
        application.setOrderId(orderId);
        application.setOriginalRecordId(latestRecord.getId());
        application.setOriginalAmount(latestRecord.getSettleAmount());
        application.setReason(reason);
        application.setApplicantId(getCurrentUserId());
        application.setStatus(RollbackStatus.PENDING_APPROVAL);
        
        // 3. 提交审批流
        approvalFlowService.submit(
            "FINANCIAL_ROLLBACK",
            application,
            List.of("财务主管", "财务总监")  // 审批人
        );
        
        return application.getId();
    }
    
    /**
     * 执行回滚（审批通过后）
     */
    @Transactional
    public void executeRollback(Long rollbackId) {
        RollbackApplication application = rollbackMapper.selectById(rollbackId);
        
        if (application.getStatus() != RollbackStatus.APPROVED) {
            throw new BizException("审批未通过");
        }
        
        // 1. 查询原订单和原财务记录
        Order order = orderMapper.selectByIdForUpdate(application.getOrderId());
        FinancialRecord original = 
            financialRecordMapper.selectById(application.getOriginalRecordId());
        
        // 2. 创建冲正记录（红字）
        FinancialRecord reversal = new FinancialRecord();
        reversal.setOrderNo(order.getOrderNo());
        reversal.setOrderId(order.getId());
        reversal.setAmount(original.getSettleAmount().negate());  // 负数
        reversal.setSource("ROLLBACK");
        reversal.setReferenceRecordId(original.getId());
        reversal.setRollbackRecordId(application.getId());
        financialRecordMapper.insert(reversal);
        
        // 3. 更新原记录状态
        financialRecordMapper.markAsRolledBack(original.getId(), reversal.getId());
        
        // 4. 更新订单状态
        order.setFinancialStatus(FinancialStatus.ROLLED_BACK);
        order.setVersion(order.getVersion() + 1);
        orderMapper.updateById(order);
        
        log.info("财务回滚完成: orderNo={}, originalAmount={}, reversalAmount={}",
            order.getOrderNo(), original.getSettleAmount(), reversal.getAmount());
    }
}
```

## 一句话总结

> **订单修改和财务核算的并发冲突核心是"谁先谁后"的时序问题，解决方案有乐观锁（version+重试）、悲观锁（FOR UPDATE）、消息队列（最终一致）、状态机（财务安全锁）四种，财务核算要记录快照版本和做定期对账防止数据被篡改，回滚要走审批流+冲正记录。**
