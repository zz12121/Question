## 先说结论

购物车设计的核心是**分层聚合模型**——底层用明细表存储SKU级数据（店铺ID、SKU ID、数量、选中状态），上层用店铺维度聚合展示，库存校验走Redis预扣+数据库兜底，价格计算实时查价+异步刷新。关键是**读写分离**：写操作（加购/修改）实时，读操作（价格/库存）可接受短暂延迟。

---

## 🔥 核心概念

**1. 解决什么问题**
- 多店铺：同一用户购物车里有不同店铺的商品，需要按店铺分组结算
- 多规格：同一商品不同SKU（颜色/尺码），价格库存独立
- 库存校验：加购时校验库存，防止超卖
- 价格计算：实时计算总价，支持优惠券/满减

**2. 基本原理/思想**
- **明细存储**：每个SKU一条记录，支持独立数量/选中状态
- **店铺分组**：用店铺ID聚合，支持分店铺结算
- **价格分离**：购物车只存快照，结算时实时查最新价
- **库存异步**：加购不扣库存，结算时校验

```java
@Service
public class CartService {
    
    @Autowired
    private CartMainRepository cartMainRepository;
    
    @Autowired
    private CartItemRepository cartItemRepository;
    
    @Autowired
    private StringRedisTemplate redisTemplate;
    
    @Autowired
    private ProductService productService;
    
    /**
     * 加入购物车
     */
    @Transactional
    public void addToCart(Long userId, AddCartRequest request) {
        Long skuId = request.getSkuId();
        Integer quantity = request.getQuantity();
        
        // 1. 查询SKU信息
        SkuInfo sku = productService.getSkuInfo(skuId);
        if (sku == null || !sku.isOnSale()) {
            throw new BusinessException("商品已下架");
        }
        
        // 2. 库存预校验（Redis）
        String stockKey = "stock:sku:" + skuId;
        Long stock = redisTemplate.opsForValue().increment(stockKey, -quantity);
        if (stock < 0) {
            // 回滚
            redisTemplate.opsForValue().increment(stockKey, quantity);
            throw new BusinessException("库存不足");
        }
        
        // 3. 查询或创建购物车主表
        CartMain cart = cartMainRepository
            .findByUserIdAndShopId(userId, sku.getShopId())
            .orElseGet(() -> createCart(userId, sku));
        
        // 4. 查询或创建明细
        CartItem item = cartItemRepository
            .findByCartIdAndSkuId(cart.getCartId(), skuId)
            .map(existing -> {
                // 已存在，累加数量
                existing.setQuantity(existing.getQuantity() + quantity);
                existing.setUpdateTime(LocalDateTime.now());
                return existing;
            })
            .orElseGet(() -> {
                // 新建明细
                CartItem newItem = new CartItem();
                newItem.setCartId(cart.getCartId());
                newItem.setUserId(userId);
                newItem.setShopId(sku.getShopId());
                newItem.setSkuId(skuId);
                newItem.setSpuId(sku.getSpuId());
                newItem.setSkuName(sku.getSkuName());
                newItem.setSkuImage(sku.getImage());
                newItem.setPriceSnapshot(sku.getCurrentPrice());
                newItem.setQuantity(quantity);
                newItem.setSelected(true);  // 默认选中
                newItem.setValid(true);
                return newItem;
            });
        
        cartItemRepository.save(item);
        
        // 5. 更新购物车统计
        updateCartStatistics(cart.getCartId());
    }
    
    private CartMain createCart(Long userId, SkuInfo sku) {
        CartMain cart = new CartMain();
        cart.setUserId(userId);
        cart.setShopId(sku.getShopId());
        cart.setShopName(sku.getShopName());
        cart.setSelectedAll(true);
        cart.setTotalCount(0);
        return cartMainRepository.save(cart);
    }
}
```

```java
@Service
public class CartQueryService {
    
    @Autowired
    private CartMainRepository cartMainRepository;
    
    @Autowired
    private CartItemRepository cartItemRepository;
    
    @Autowired
    private ProductService productService;
    
    /**
     * 查询用户购物车（按店铺分组）
     */
    public List<CartShopVO> getUserCart(Long userId) {
        // 1. 查询所有购物车
        List<CartMain> carts = cartMainRepository.findByUserId(userId);
        
        return carts.stream().map(cart -> {
            CartShopVO shopVO = new CartShopVO();
            shopVO.setShopId(cart.getShopId());
            shopVO.setShopName(cart.getShopName());
            shopVO.setSelectedAll(cart.getSelectedAll());
            
            // 2. 查询该购物车下的所有明细
            List<CartItem> items = cartItemRepository
                .findByCartIdAndValidTrue(cart.getCartId());
            
            // 3. 批量查询最新价格和库存（优化N+1）
            List<Long> skuIds = items.stream()
                .map(CartItem::getSkuId).collect(Collectors.toList());
            Map<Long, SkuRealtimeInfo> realtimeMap = productService
                .batchGetRealtimeInfo(skuIds);
            
            // 4. 组装VO
            List<CartItemVO> itemVOs = items.stream().map(item -> {
                CartItemVO vo = new CartItemVO();
                vo.setItemId(item.getItemId());
                vo.setSkuId(item.getSkuId());
                vo.setSkuName(item.getSkuName());
                vo.setSkuImage(item.getSkuImage());
                vo.setQuantity(item.getQuantity());
                vo.setSelected(item.getSelected());
                
                // 实时价格（可能和快照不同）
                SkuRealtimeInfo realtime = realtimeMap.get(item.getSkuId());
                vo.setCurrentPrice(realtime.getPrice());
                vo.setPriceChanged(!realtime.getPrice()
                    .equals(item.getPriceSnapshot()));
                
                // 库存状态
                vo.setStockStatus(realtime.getStock() >= item.getQuantity() 
                    ? "SUFFICIENT" : "INSUFFICIENT");
                
                // 商品是否有效
                vo.setValid(realtime.isOnSale() && item.isValid());
                
                return vo;
            }).collect(Collectors.toList());
            
            shopVO.setItems(itemVOs);
            
            // 5. 计算店铺小计
            calculateShopTotal(shopVO);
            
            return shopVO;
        }).collect(Collectors.toList());
    }
    
    /**
     * 计算选中商品的总价（结算用）
     */
    public CartTotalVO calculateTotal(Long userId, List<Long> selectedItemIds) {
        List<CartItem> items = cartItemRepository.findAllById(selectedItemIds);
        
        // 校验是否属于当前用户
        items.forEach(item -> {
            if (!item.getUserId().equals(userId)) {
                throw new BusinessException("非法操作");
            }
        });
        
        // 批量查最新价格
        Map<Long, BigDecimal> priceMap = productService
            .batchGetPrice(items.stream().map(CartItem::getSkuId).collect(Collectors.toList()));
        
        BigDecimal totalAmount = BigDecimal.ZERO;
        int totalCount = 0;
        
        for (CartItem item : items) {
            if (!item.isSelected()) continue;
            
            BigDecimal price = priceMap.getOrDefault(item.getSkuId(), item.getPriceSnapshot());
            totalAmount = totalAmount.add(price.multiply(new BigDecimal(item.getQuantity())));
            totalCount += item.getQuantity();
        }
        
        CartTotalVO total = new CartTotalVO();
        total.setTotalAmount(totalAmount);
        total.setTotalCount(totalCount);
        // 优惠券计算...
        
        return total;
    }
}
```

```java
@Service
public class StockCheckService {
    
    @Autowired
    private StringRedisTemplate redisTemplate;
    
    @Autowired
    private ProductService productService;
    
    /**
     * 结算时库存校验
     */
    public StockCheckResult checkStock(List<CartItem> items) {
        // 1. 先查Redis缓存
        List<String> stockKeys = items.stream()
            .map(i -> "stock:sku:" + i.getSkuId())
            .collect(Collectors.toList());
        
        List<String> stockStrs = redisTemplate.opsForValue().multiGet(stockKeys);
        
        List<CartItem> needDbCheck = new ArrayList<>();
        List<StockCheckResult.Item> results = new ArrayList<>();
        
        for (int i = 0; i < items.size(); i++) {
            CartItem item = items.get(i);
            String stockStr = stockStrs.get(i);
            
            if (stockStr == null) {
                // Redis未命中，需要查DB
                needDbCheck.add(item);
            } else {
                long stock = Long.parseLong(stockStr);
                results.add(new StockCheckResult.Item(
                    item.getSkuId(),
                    item.getQuantity(),
                    stock,
                    stock >= item.getQuantity()
                ));
            }
        }
        
        // 2. 查DB补全
        if (!needDbCheck.isEmpty()) {
            Map<Long, Integer> dbStocks = productService
                .batchGetStock(needDbCheck.stream().map(CartItem::getSkuId).collect(Collectors.toList()));
            
            for (CartItem item : needDbCheck) {
                Integer stock = dbStocks.get(item.getSkuId());
                // 回填Redis
                if (stock != null) {
                    redisTemplate.opsForValue().set(
                        "stock:sku:" + item.getSkuId(), 
                        stock.toString(), 
                        5, TimeUnit.MINUTES
                    );
                }
                
                results.add(new StockCheckResult.Item(
                    item.getSkuId(),
                    item.getQuantity(),
                    stock != null ? stock : 0,
                    stock != null && stock >= item.getQuantity()
                ));
            }
        }
        
        boolean allSufficient = results.stream().allMatch(StockCheckResult.Item::isSufficient);
        
        return new StockCheckResult(allSufficient, results);
    }
}
```
