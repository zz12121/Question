## 先说结论

亿级商品搜索设计的核心是**Elasticsearch + 异构数据同步**——ES负责倒排索引和复杂查询，MySQL保留原始数据，通过Canal/MQ实现实时同步。架构分层：接入层统一查询网关，索引层按类目分片+冷热分离，数据层MySQL+ES双写。关键优化：分词用IK+自定义词库，排序用function_score实现个性化，分页用search_after深分页，筛选用filter缓存加速。

### 三、核心搜索流程

#### 1. 查询构建（DSL）

```java
@Service
public class SearchService {
    
    @Autowired
    private RestHighLevelClient esClient;
    
    /**
     * 商品搜索
     */
    public SearchResult search(ProductSearchRequest request) {
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        // 1. 构建查询条件（must + filter分离）
        BoolQueryBuilder boolQuery = QueryBuilders.boolQuery();
        
        // 1.1 关键词搜索（影响相关性得分）
        if (StringUtils.hasText(request.getKeyword())) {
            MultiMatchQueryBuilder keywordQuery = QueryBuilders
                .multiMatchQuery(request.getKeyword(), 
                    "title^3",      // 标题权重3
                    "brandName^2",  // 品牌权重2
                    "categoryName"  // 类目权重1
                )
                .type(MultiMatchQueryBuilder.Type.BEST_FIELDS)
                .fuzziness(Fuzziness.AUTO);  // 容错匹配
            
            boolQuery.must(keywordQuery);
        }
        
        // 1.2 筛选条件（filter，不计算得分，可缓存）
        List<QueryBuilder> filters = new ArrayList<>();
        
        // 类目筛选
        if (request.getCategoryId() != null) {
            filters.add(QueryBuilders.termQuery("categoryId", request.getCategoryId()));
        }
        
        // 品牌筛选（多选）
        if (!CollectionUtils.isEmpty(request.getBrandIds())) {
            filters.add(QueryBuilders.termsQuery("brandId", request.getBrandIds()));
        }
        
        // 价格区间
        if (request.getMinPrice() != null || request.getMaxPrice() != null) {
            RangeQueryBuilder priceRange = QueryBuilders.rangeQuery("price");
            if (request.getMinPrice() != null) {
                priceRange.gte(request.getMinPrice());
            }
            if (request.getMaxPrice() != null) {
                priceRange.lte(request.getMaxPrice());
            }
            filters.add(priceRange);
        }
        
        // 属性筛选（nested查询）
        if (!CollectionUtils.isEmpty(request.getAttributes())) {
            for (AttributeFilter attr : request.getAttributes()) {
                NestedQueryBuilder nestedQuery = QueryBuilders.nestedQuery(
                    "attributes",
                    QueryBuilders.boolQuery()
                        .must(QueryBuilders.termQuery("attributes.attrId", attr.getAttrId()))
                        .must(QueryBuilders.termsQuery("attributes.attrValue", attr.getValues())),
                    ScoreMode.None
                );
                filters.add(nestedQuery);
            }
        }
        
        // 有货筛选
        if (request.getInStock() != null && request.getInStock()) {
            filters.add(QueryBuilders.rangeQuery("stock").gt(0));
        }
        
        // 所有filter加入boolQuery
        filters.forEach(boolQuery::filter);
        
        sourceBuilder.query(boolQuery);
        
        // 2. 排序
        if ("price_asc".equals(request.getSort())) {
            sourceBuilder.sort("price", SortOrder.ASC);
        } else if ("price_desc".equals(request.getSort())) {
            sourceBuilder.sort("price", SortOrder.DESC);
        } else if ("sales".equals(request.getSort())) {
            sourceBuilder.sort("sales", SortOrder.DESC);
        } else {
            // 默认：综合排序（相关性 + 销量 + 店铺评分）
            FunctionScoreQueryBuilder functionScore = QueryBuilders.functionScoreQuery(
                boolQuery,
                new FunctionScoreQueryBuilder.FilterFunctionBuilder[]{
                    // 销量加权
                    new FunctionScoreQueryBuilder.FilterFunctionBuilder(
                        ScoreBuilders.fieldValueFactorFunction("sales")
                            .factor(0.1f)
                            .modifier(FieldValueFactorFunction.Modifier.LOG1P)
                    ),
                    // 店铺评分加权
                    new FunctionScoreQueryBuilder.FilterFunctionBuilder(
                        ScoreBuilders.fieldValueFactorFunction("shopScore")
                            .factor(0.5f)
                    ),
                    // 新品加权
                    new FunctionScoreQueryBuilder.FilterFunctionBuilder(
                        QueryBuilders.rangeQuery("createTime").gte("now-7d"),
                        ScoreBuilders.weightFactorFunction(1.2f)
                    )
                }
            ).scoreMode(FunctionScoreQuery.ScoreMode.SUM);
            
            sourceBuilder.query(functionScore);
        }
        
        // 3. 分页（深分页优化）
        if (request.getPage() <= 100) {
            // 浅分页用from/size
            sourceBuilder.from((request.getPage() - 1) * request.getSize());
            sourceBuilder.size(request.getSize());
        } else {
            // 深分页用search_after
            sourceBuilder.size(request.getSize());
            if (StringUtils.hasText(request.getSearchAfter())) {
                sourceBuilder.searchAfter(request.getSearchAfter().split(","));
            }
        }
        
        // 4. 高亮
        HighlightBuilder highlight = new HighlightBuilder();
        highlight.field("title").preTags("<em>").postTags("</em>");
        sourceBuilder.highlighter(highlight);
        
        // 5. 聚合（facet筛选）
        // 品牌聚合
        sourceBuilder.aggregation(
            AggregationBuilders.terms("brands").field("brandId").size(20)
        );
        
        // 类目聚合
        sourceBuilder.aggregation(
            AggregationBuilders.terms("categories").field("categoryId").size(20)
        );
        
        // 价格区间聚合
        sourceBuilder.aggregation(
            AggregationBuilders.histogram("price_ranges")
                .field("price")
                .interval(100)
        );
        
        // 6. 执行查询
        SearchRequest searchRequest = new SearchRequest("products");
        searchRequest.source(sourceBuilder);
        
        try {
            SearchResponse response = esClient.search(searchRequest, RequestOptions.DEFAULT);
            return parseResponse(response);
        } catch (IOException e) {
            throw new SearchException("搜索失败", e);
        }
    }
}
```

#### 2. 搜索结果解析

```java
private SearchResult parseResponse(SearchResponse response) {
    SearchResult result = new SearchResult();
    
    // 1. 解析商品列表
    List<ProductDoc> products = Arrays.stream(response.getHits().getHits())
        .map(hit -> {
            ProductDoc doc = JSON.parseObject(hit.getSourceAsString(), ProductDoc.class);
            
            // 高亮标题
            HighlightField highlight = hit.getHighlightFields().get("title");
            if (highlight != null) {
                doc.setHighlightTitle(highlight.fragments()[0].string());
            }
            
            // 记录search_after值（用于深分页）
            doc.setSortValues(hit.getSortValues());
            
            return doc;
        })
        .collect(Collectors.toList());
    
    result.setProducts(products);
    result.setTotal(response.getHits().getTotalHits().value);
    
    // 2. 解析聚合结果（筛选条件）
    Aggregations aggregations = response.getAggregations();
    
    // 品牌聚合
    Terms brandTerms = aggregations.get("brands");
    List<BrandFacet> brandFacets = brandTerms.getBuckets().stream()
        .map(bucket -> new BrandFacet(
            ((Long) bucket.getKey()).intValue(),
            bucket.getDocCount()
        ))
        .collect(Collectors.toList());
    result.setBrandFacets(brandFacets);
    
    // 类目聚合
    Terms categoryTerms = aggregations.get("categories");
    List<CategoryFacet> categoryFacets = categoryTerms.getBuckets().stream()
        .map(bucket -> new CategoryFacet(
            ((Long) bucket.getKey()).intValue(),
            bucket.getDocCount()
        ))
        .collect(Collectors.toList());
    result.setCategoryFacets(categoryFacets);
    
    return result;
}
```

```java
@Service
public class CachedSearchService {
    
    @Autowired
    private StringRedisTemplate redisTemplate;
    
    @Autowired
    private SearchService searchService;
    
    /**
     * 带缓存的搜索
     */
    public SearchResult searchWithCache(ProductSearchRequest request) {
        // 1. 构建缓存Key（去掉用户相关参数）
        String cacheKey = buildCacheKey(request);
        
        // 2. 查缓存
        String cached = redisTemplate.opsForValue().get(cacheKey);
        if (cached != null) {
            return JSON.parseObject(cached, SearchResult.class);
        }
        
        // 3. 执行搜索
        SearchResult result = searchService.search(request);
        
        // 4. 写入缓存（热点查询缓存5分钟）
        if (isHotQuery(request)) {
            redisTemplate.opsForValue().set(cacheKey, 
                JSON.toJSONString(result), 5, TimeUnit.MINUTES);
        }
        
        return result;
    }
    
    /**
     * 缓存失效策略
     */
    @EventListener
    public void onProductChange(ProductChangeEvent event) {
        // 商品变更时，模糊匹配失效相关缓存
        String pattern = "search:*:category:" + event.getCategoryId() + ":*";
        Set<String> keys = redisTemplate.keys(pattern);
        if (!CollectionUtils.isEmpty(keys)) {
            redisTemplate.delete(keys);
        }
    }
}
```
