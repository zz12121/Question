## 二、五种灰度路由策略

### 策略一：用户 ID 哈希（最常用）

```
┌─────────────────────────────────────────────────────────────────────┐
│                    用户 ID 哈希灰度                                  │
│                                                                     │
│  【原理】hash(userId) % 100 < 5 → 灰度用户                           │
│                                                                     │
│  优点：同一用户每次请求都路由到同一版本（体验一致）                    │
│  缺点：用户分布不均时，灰度比例可能有波动                             │
│                                                                     │
│  ┌───────────────────────────────────────────────────────────────┐ │
│  │  userId = 12345                                              │ │
│  │  hash = MurmurHash(12345) % 100 = 37                         │ │
│  │  37 < 5 ? ❌ → 走老版本                                        │ │
│  │                                                               │ │
│  │  userId = 67890                                              │ │
│  │  hash = MurmurHash(67890) % 100 = 3                          │ │
│  │  3 < 5 ? ✅ → 走灰度版本                                       │ │
│  └───────────────────────────────────────────────────────────────┘ │
└─────────────────────────────────────────────────────────────────────┘
```

```java
/**
 * 基于用户 ID 的灰度路由
 */
@Service
public class GrayReleaseService {

    private static final double GRAY_PERCENT = 0.05;  // 5% 灰度

    /**
     * 判断用户是否在灰度人群中
     */
    public boolean isGrayUser(Long userId) {
        // MurmurHash 分布更均匀
        int hash = MurmurHash.hash(userId.toString());
        int bucket = Math.abs(hash % 100);
        return bucket < (GRAY_PERCENT * 100);  // < 5
    }

    /**
     * 网关路由判断
     */
    @Component
    @Slf4j
    public class GrayGatewayFilter implements GlobalFilter {

        @Autowired
        private GrayReleaseService grayService;

        @Override
        public Mono<Void> filter(ServerWebExchange exchange, GatewayFilterChain chain) {
            ServerHttpRequest request = exchange.getRequest();

            // 1. 获取用户 ID（从 JWT / Header / Cookie）
            Long userId = extractUserId(request);

            if (userId != null && grayService.isGrayUser(userId)) {
                // 2. 染色：添加 Header
                ServerHttpRequest mutatedRequest = request.mutate()
                    .header("X-Gray-Version", "v2")  // 标记灰度用户
                    .build();

                log.info("灰度用户路由到新版本, userId={}", userId);

                return chain.filter(
                    exchange.mutate().request(mutatedRequest).build()
                );
            }

            // 3. 普通用户走老版本
            return chain.filter(exchange);
        }
    }
}

/**
 * Spring Cloud Gateway 路由配置
 */
@Configuration
public class GatewayRouteConfig {

    @Bean
    public RouteLocator customRouteLocator(RouteLocatorBuilder builder) {
        return builder.routes()
            // 老版本路由
            .route("order-service-stable", r -> r
                .path("/api/order/**")
                .filters(f -> f.stripPrefix(0))
                .uri("lb://order-service"))
            // 灰度版本路由（Header 染色触发）
            .route("order-service-gray", r -> r
                .path("/api/order/**")
                .filters(f -> f
                    .stripPrefix(0)
                    .addRequestHeader("X-Gray-Version", "v2"))
                .uri("lb://order-service-gray"))
            .build();
    }
}
```

```java
 * 动态切换 Nginx 灰度配置
 */
@Service
public class NginxGrayConfigManager {

    private static final String NGINX_CONF_PATH = "/etc/nginx/conf.d/gray.conf";

    public void enableGray(int percent) {
        String config = String.format("""
            upstream order-stable {
                server order-service-v1:8080 weight=%d;
            }
            upstream order-gray {
                server order-service-v2:8080 weight=%d;
            }
            """, 100 - percent, percent);

        // 写入配置并 reload
        Files.writeString(Path.of(NGINX_CONF_PATH), config);
        execCommand("nginx -s reload");

        log.info("灰度配置已更新, percent={}", percent);
    }

    public void disableGray() {
        Files.deleteIfExists(Path.of(NGINX_CONF_PATH));
        execCommand("nginx -s reload");
        log.info("灰度已关闭，全量切换到老版本");
    }
}
```

```java
/**
 * 自定义灰度负载均衡策略
 */
@Configuration
public class GrayLoadBalancerConfig {

    @Bean
    public ReactorLoadBalancer<ServiceInstance> grayRule(
            Environment environment) {
        return new GrayLoadBalancer(
            new RandomLoadBalancer(),
            environment
        );
    }
}

public class GrayLoadBalancer implements ReactorServiceInstanceLoadBalancer {

    private final ObjectProvider<ServiceInstance> serviceInstanceProvider;
    private final double grayPercent;

    public GrayLoadBalancer(ObjectProvider<ServiceInstance> provider,
                           Environment env) {
        this.serviceInstanceProvider = provider;
        this.grayPercent = Double.parseDouble(
            env.getProperty("gray.percent", "5")
        );
    }

    @Override
    public Mono<Response<ServiceInstance>> choose(Request request) {
        // 获取请求上下文
        ServerWebExchange exchange = request.getNativeRequest();
        HttpHeaders headers = exchange.getRequest().getHeaders();

        // 1. 检查 Header 是否已染色
        String grayVersion = headers.getFirst("X-Gray-Version");
        if ("v2".equals(grayVersion)) {
            return chooseFromGray(request);
        }

        // 2. 检查用户 ID 哈希
        Long userId = extractUserId(headers);
        if (userId != null && isGrayUser(userId)) {
            return chooseFromGray(request);
        }

        // 3. 普通用户走稳定版
        return chooseFromStable(request);
    }

    private boolean isGrayUser(Long userId) {
        return (Math.abs(userId.hashCode()) % 100) < grayPercent;
    }
}
```

```java
/**
 * Feature Flag 灰度实现
 */
@Service
public class FeatureFlagService {

    @Autowired
    private ConfigClient configClient;  // Nacos / Apollo

    /**
     * 判断用户是否开启某功能
     */
    public boolean isFeatureEnabled(String featureKey, User user) {
        // 1. 从配置中心获取开关配置
        FeatureConfig config = configClient.getFeatureConfig(featureKey);

        if (!config.isEnabled()) {
            return false;
        }

        // 2. 白名单优先
        if (config.getUserIds().contains(user.getId())) {
            return true;
        }

        // 3. 版本限制
        if (config.hasVersionConstraint()) {
            if (!checkVersion(user.getAppVersion(), config.getVersionConstraint())) {
                return false;
            }
        }

        // 4. 百分比灰度
        return isInGrayPercentage(user.getId(), config.getPercentage());
    }

    private boolean isInGrayPercentage(Long userId, int percentage) {
        return (Math.abs(userId.hashCode()) % 100) < percentage;
    }
}

/**
 * 灰度功能使用示例
 */
@Service
@Slf4j
public class OrderService {

    @Autowired
    private FeatureFlagService featureFlag;

    @Autowired
    private NewOrderService newOrderService;   // 新版

    @Autowired
    private OldOrderService oldOrderService;   // 旧版

    public Order createOrder(User user, OrderRequest request) {
        // 灰度逻辑
        if (featureFlag.isFeatureEnabled("new_order_flow_v2", user)) {
            log.info("用户 {} 走新订单流程", user.getId());
            return newOrderService.create(request);
        } else {
            log.info("用户 {} 走老订单流程", user.getId());
            return oldOrderService.create(request);
        }
    }
}
```

```java
    private MeterRegistry meterRegistry;

        boolean isGray = featureFlag.isFeatureEnabled(featureKey, user);

        if (isGray) {
            grayRequestCounter.increment();
            Timer.Context timer = grayRequestTimer.start();
            try {
                return grayLogic.get();
            } catch (Exception e) {
                grayErrorCounter.increment();
                throw e;
            } finally {
                timer.stop();
            }
        } else {
            stableRequestCounter.increment();
            return stableLogic.get();
        }
    }
```

## 六、面试加分话术

> **面试官问：灰度过程中新版本有 Bug，如何快速回滚？**
>
> 三层回滚机制：
> 1. **配置中心关闭开关**（Apollo/Nacos → 秒级生效）
> 2. **网关层切回老版本路由**（Nginx reload）
> 3. **数据库灰度数据回滚**（如新功能写了数据，需要清理脚本）

> **追问：灰度用户如何选择？**
>
> 两种策略：
> 4. **随机灰度**：hash(userId) % 100，适合通用场景
> 5. **定向灰度**：VIP 用户、白名单用户、地区用户，适合重点验证

> **再问：灰度和 A/B 测试有什么区别？**
>
> 灰度 = 技术发布策略，关注稳定性
> A/B 测试 = 业务优化策略，关注转化率
> 两者可以结合：灰度放量 + A/B 对比核心业务指标

## 一句话收尾

**灰度发布 = 流量染色 + 规则路由 + 监控对比 + 一键回滚，让新功能"带护栏上线"，出问题秒级切回。**
